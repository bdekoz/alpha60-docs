48/// Torrent swarms are characterized by peer and seed elements.

56{
57 using enum_type = swarm_mode;
58 using enum_map_type = std::map<enum_type, string>;

61 if (enum_map.empty())
62 {
63 enum_map[enum_type::peer] = k::peer_s;
64 enum_map[enum_type::seed] = k::seed_s;

108/// Less than compare for meta_peer via ip address.
109bool
110meta_peer_lt_ip(const meta_peer& m1, const meta_peer& m2)
111{ return std::get<0>(m1) < std::get<0>(m2); };
112
113/// Less than compare for meta_peer via geo.
114bool
115meta_peer_lt_geo(const meta_peer& m1, const meta_peer& m2)
116{
117 bool ret(false);
118 string s1 = std::get<1>(m1);
119 string s2 = std::get<1>(m2);
120 bool t1 = s1 < s2;
121
122 if (t1)
123 ret = t1;
124 else
125 {
126 if (s1 == s2)
127 {
128 string sr1 = std::get<4>(m1);
129 string sr2 = std::get<4>(m2);
130 bool t2 = sr1 < sr2;
131 if (t2)
132 ret = t2;
133 else
134 {
135 if (sr1 == sr2)
136 {
137 string sc1 = std::get<5>(m1);
138 string sc2 = std::get<5>(m2);
139 ret = sc1 < sc2;
140 }
141 }
142 }
143 }
144 return ret;
145};

168/// Cache for whois calls.
169meta_peer
170cached_whois_to_meta_peer(string ip, double& lt, double& lng, string& country)
171{
172 using cached_whois = std::unordered_map<string, meta_peer>;
173 static cached_whois whoiscache;
174
175 string region;
176 string city;
177 auto mp = std::make_tuple(ip, country, lt, lng, region, city);
178
179 meta_peer retval;
180 auto i = whoiscache.find(ip);
181 if (i != whoiscache.end())
182 retval = i->second;
183 else
184 {
185 whois_to_meta_peer(ip, lt, lng, country);
186 auto wmp = std::make_tuple(ip, country, lt, lng, region, city);
187 whoiscache.insert(make_pair(ip, wmp));
188 retval = wmp;
189 }
190 return retval;
191}

196/// Hash specializations.
197namespace std {

199 // tuple types
200 // https://www.variadic.xyz/2018/01/15/hashing-stdpair-and-stdtuple/
201 template <class T>
202 inline void hash_combine(std::size_t& seed, T v)
203 {
204 seed ^= hash<T>()(v) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
205 }
206
207 template<class... TupleArgs>
208 struct hash<tuple<TupleArgs...>>
209 {
210 private:
211 static constexpr size_t N = std::tuple_size<tuple<TupleArgs...>>::value;
212
213 // this is a termination condition
214 // N == sizeof...(TupleTypes)
215 template<size_t Idx = 0, typename... TupleTypes>
216 inline typename enable_if<N == sizeof...(TupleTypes), void>::type
217 hash_combine_tup(size_t&, const tuple<TupleTypes...>&) const
218 { }
219
220 // this is the computation workhorse
221 // N < sizeof...(TupleTypes)
222 template<size_t Idx = 0, typename... TupleTypes>
223 inline typename enable_if<N < sizeof...(TupleTypes), void>::type
224 hash_combine_tup(size_t& seed, const tuple<TupleTypes...>& tup) const
225 {
226 hash_combine(seed, get<Idx>(tup));
227
228 // on to next element
229 hash_combine_tup<Idx+1>(seed, tup);
230 }
231
232 public:
233 size_t operator()(tuple<TupleArgs...> tupleValue) const
234 {
235 size_t seed = 0;
236 hash_combine_tup<0>(seed, tupleValue);
237 return seed;
238 }
239 };

242 template<>
243 struct hash<a60::meta_peer>
244 {
245 std::size_t
246 operator()(const a60::meta_peer& p) const
247 {
248 const std::size_t h1(std::hash<a60::string>{}(get<0>(p)));
249 return h1;
250 }

270/// GeoIP
271meta_peer
272make_meta_peer(string addr, geo::country_map& cmap [[gnu::unused]],
273 geo::city_map& ctymap [[gnu::unused]])
274{
275 // Strip IPV6 encoding from addr, ie:
276 // ::ffff:46.165.244.207
277 const char colon(':');
278 if (addr.front() == colon)
279 addr = addr.substr(addr.rfind(colon) + 1);
280
281 string country;
282 double lt(0);
283 double lng(0);
284 string region;
285 string city;
286
287#ifdef a60_USE_GEO
288 country = cmap.country_code_via_string(addr);
289 geo::city_map::record r = ctymap.open_record(addr);
290 if (r.is_record_open())
291 {
292 lt = r.ptr->latitude;
293 lng = r.ptr->longitude;
294 region = geo::safe_if_empty(r.ptr->region);
295 city = geo::safe_if_empty(r.ptr->city);
296 }
297#endif
298
299 /// Whois
300 /// Do more extensive checks based on netblock here using whois.
301 /// Use RIPE Database search service, returning objects are in RPSL format.
302 /// https://www.irr.net/docs/rpsl.html
303 const bool whoisp = false;
304 const bool is_looking_lostp = country.empty() || (lt == 0 && lng == 0);
305 if (is_looking_lostp && whoisp)
306 whois_to_meta_peer(addr, lt, lng, country);
307
308 // Finalize.
309 return std::make_tuple(addr, country, lt, lng, region, city);
310}

394/// Try again to get geoip information for peers that are returning (0,0).
395vmeta_peers
396rescue_lost_peers(const vmeta_peers& peers)
397{
398 vmeta_peers peeredo(peers);
399
400#ifdef a60_USE_GEO
401
402 // Setup geospatial.
403 geo::country_map cmap;
404 cmap.open();
405 geo::city_map ctymap;
406 ctymap.open(GEOIP_INDEX_CACHE);
407
408 vmeta_peers isla;
409 auto funct = is_meta_peer_lost;
410 auto fit = std::find_if(peeredo.begin(), peeredo.end(), funct);
411 while (fit != peeredo.end())
412 {
413 // Extract out the ip, and try again with address to get geo info.
414 meta_peer p(*fit);
415 string addr(std::get<0>(p));
416 meta_peer pnew = make_meta_peer(addr, cmap, ctymap);
417 isla.push_back(pnew);
418 auto eit = peeredo.erase(fit);
419 fit = std::find_if(eit, peeredo.end(), funct);
420 }
421
422 // Copy back in the known lost island peers.
423 std::copy(isla.begin(), isla.end(), std::back_inserter(peeredo));
424
425#endif
426 return peeredo;
427}

463/// Assume input list is sorted low to high, based on IP4 address.
464vmeta_peers
465make_exit_node_meta_peers(const string nodedumpfile = "current-nodes.txt")
466{
467 vmeta_peers metapeers;
468 auto& rtr = io::get_run_time_resources();
469 string fname = io::end_path(rtr.data) + "tor-exit-nodes/" + nodedumpfile;
470
471#ifdef a60_USE_GEO
472
473 geo::country_map cmap;
474 cmap.open();
475 geo::city_map ctymap;
476 ctymap.open(GEOIP_INDEX_CACHE);
477
478 std::ifstream ifs(fname);
479 if (ifs.good())
480 {
481 // Discard first three lines.
482 string line;
483 getline(ifs, line);
484 getline(ifs, line);
485 getline(ifs, line);
486 getline(ifs, line);
487
488 while (!line.empty() && ifs.good())
489 {
490 std::istringstream iss(line);
491 string addr = line_to_ip_address_as_string(iss);
492 meta_peer pmeta = make_meta_peer(addr, cmap, ctymap);
493 metapeers.push_back(pmeta);
494
495 getline(ifs, line);
496 }
497
498 std::cout << metapeers.size() << " exit nodes found" << std::endl;
499 }
500 else
501 throw std::runtime_error("make_exit_node_meta_peers:: source file error: "
502 + fname);
503#endif
504
505 return metapeers;
506}

520/// Weigh by similar latitude and logitude.
521vwmetai_peers
522weigh_meta_peers_via_geo_coordinates(const vmeta_peers& peers)
523{
524 // Remove each duplicate entry, based on latitude and longitude,
525 // incrementing the weight of the original by one each time.
526 vwmetai_peers wpeersn;
527 vmeta_peers wpeers(peers);
528 for (const meta_peer& wp: wpeers)
529 {
530 auto lwpeereq = [wp](const meta_peer& p)
531 {
532 double lt1 = std::get<2>(wp);
533 double lt2 = std::get<2>(p);
534 double lng1 = std::get<3>(wp);
535 double lng2 = std::get<3>(p);
536 bool t1 = lt1 == lt2;
537 bool t2 = lng1 == lng2;
538 return bool(t1 && t2);
539 };
540
541 int n = std::count_if(wpeers.begin(), wpeers.end(), lwpeereq);
542 if (n > 1)
543 {
544 // Erase.
545 auto fit = std::find_if(wpeers.begin(), wpeers.end(), lwpeereq);
546 while (fit != wpeers.end())
547 {
548 wpeers.erase(fit);
549 fit = std::find_if(wpeers.begin(), wpeers.end(), lwpeereq);
550 }
551 }
552 wpeersn.push_back(std::make_tuple(wp, n));
553 }
554
555 // Sort weighted peer list, the largest weights first so that the
556 // biggest weights are drawn first.
557 auto lpeergt = [](const wmetai_peer& p1, const wmetai_peer& p2)
558 { return std::get<1>(p1) > std::get<1>(p2); };
559 std::sort(wpeersn.begin(), wpeersn.end(), lpeergt);
560
561 return wpeersn;
562}

611/// Normalize value from (min, max) on (nfloor, nceil)
612double
613normalize_on_range(uint value, uint min, uint max, uint nfloor, uint nceil)
614{
615 auto weightn_numer = ((nceil - nfloor) * (value - min));
616 auto weightn_denom = (max - min) + nfloor;
617 auto weightn = std::round(weightn_numer / weightn_denom);
618 return weightn;
619}

622/**
623 Normalize density value to a smaller range.
624
625 To convert a value x from [minimum..maximum] to [floor..ceil]
626 normalize(x) = ((ceil - floor) * (x - minimum))/(maximum - minimum) + floor
627*/
628template<typename W>
629vwmeta_peers<W>
630normalize_weighted_peers(const vwmeta_peers<W>& wpeers,
631 const uint nfloor, const uint nceil)
632{
633 vwmeta_peers<W> wpeersn;
634 if (!wpeers.empty())
635 {
636 auto lpeerlt = [](const wmeta_peer<W>& p1, const wmeta_peer<W>& p2)
637 { return std::get<1>(p1) < std::get<1>(p2); };
638
639 // So... find the min and max of the re-weighted peers.
640 auto dminp = std::min_element(wpeers.begin(), wpeers.end(), lpeerlt);
641 auto dmaxp = std::max_element(wpeers.begin(), wpeers.end(), lpeerlt);
642 auto dmin = std::get<1>(*dminp);
643 auto dmax = std::get<1>(*dmaxp);
644
645 // First, calculate the top (topn) cities for each torrent, and then
646 // normalize re-weighed peers to (nfloor, nceil) range.
647 for (const wmeta_peer<W>& wp : wpeers)
648 {
649 const meta_peer p = std::get<0>(wp);
650 auto weightd = std::get<1>(wp);
651 auto weightn = normalize_on_range(weightd, dmin, dmax, nfloor, nceil);
652 wpeersn.push_back(std::make_tuple(p, weightn));
653 }
654 }
655 return wpeersn;
656}

659template<typename W>
660vwmeta_peers<W>
661unique_weighted_peers_via_geo_names(vwmeta_peers<W>& wpeers)
662{
663 using wpeer_type = wmeta_peer<W>;
664 using wpeers_type = vwmeta_peers<W>;
665
666 wpeers_type wpeersu;
667 for (const wpeer_type& wp: wpeers)
668 {
669 auto lwpeereq = [wp](const wpeer_type& wp2)
670 {
671 const meta_peer& mp1 = std::get<0>(wp);
672 const meta_peer& mp2 = std::get<0>(wp2);
673 const string cp1 = std::get<1>(mp1);
674 const string cp1r = std::get<5>(mp1);
675 const string cp2 = std::get<1>(mp2);
676 const string cp2r = std::get<5>(mp2);
677 return cp1 == cp2 && cp1r == cp2r;
678 };
679
680 auto weightn(0);
681 meta_peer p = std::get<0>(wp);
682 int n = std::count_if(wpeers.begin(), wpeers.end(), lwpeereq);
683 if (n > 1)
684 {
685 // Erase.
686 auto fit = std::find_if(wpeers.begin(), wpeers.end(), lwpeereq);
687 while (fit != wpeers.end())
688 {
689 weightn += std::get<1>(*fit);
690 wpeers.erase(fit);
691 fit = std::find_if(wpeers.begin(), wpeers.end(), lwpeereq);
692 }
693 }
694 else
695 weightn = std::get<1>(wp);
696 wpeersu.push_back(std::make_tuple(p, weightn));
697 }
698
699 // Then, re-sort adescending.
700 auto lpeergt = [](const wpeer_type& p1, const wpeer_type& p2)
701 { return std::get<1>(p1) > std::get<1>(p2); };
702 std::sort(wpeersu.begin(), wpeersu.end(), lpeergt);
703 return wpeersu;
704}
