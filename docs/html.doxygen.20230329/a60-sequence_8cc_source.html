44/**
45 Sequence a directory formatted with half frames such that the
46 frames crawl upwards vertically from the lower half, to the top
47 half, then off screen. As each image moves up, the next image in
48 series is taken from the input directory until none remains.
49
50 idir is a directory of png images, each of which is half the size of a frame
51
52 Common resolutions refreshers:
53 1080p is 1920 x 1080
54 4k is 3840 x 2160 (aka 2 x 2 of 1080p)
55
56 So area = 4k implies an input directory of 2 x (3840 x 1080p)
57 frames stacked vertically.
58
59 This will produce as output numbered frames for as many image
60 combinations as is required to use up all the images in the input
61 directory.
62
63 Then use a script to transform each image to a movie of n seconds, merge.
64*/
65void
66composite_images_vertical_flip_up(string idir, const area<> a,
67 const string basename = "vertical-flip-")
68{
69 strings files = io::populate_files(idir, ".png");
70 std::cout << "composite_images_vertical_flip files found: "
71 << to_string(files.size())
72 << std::endl;
73
74 const auto [ width, height ] = a;
75 const auto hheight(height / 2);
76
77 // Assume each png in idir is 3840 × 1080
78 // Or, the bottom slice of a 4k frame.
79 for (uint i = 0; i < files.size(); ++i)
80 {
81 std::ostringstream ostr;
82 ostr << basename << std::setfill('0') << std::setw(4) << i;
83 svg_element obj(ostr.str(), a);
84
85 // Bottom
86 image_element img;
87 image_element::data di = { files[i], 0, hheight, width, hheight };
88 img.start_element();
89 img.add_data(di);
90 img.finish_element();
91 obj.add_element(img);
92
93 // Top
94 if (i > 0)
95 {
96 image_element imgt;
97 image_element::data dt = { files[i - 1], 0, 0, width, hheight };
98 imgt.start_element();
99 imgt.add_data(dt);
100 imgt.finish_element();
101 obj.add_element(imgt);
102 }
103 }
104}

107void
108composite_images_vertical_wink(string idir, const area<> frame,
109 const color klr = color::white,
110 const double totalsec = 4,
111 const string basename = "vertical-wink-")
112{
113 const auto [ width, height ] = frame;
114 const size_t fps = 30;
115
116 strings files = io::populate_files(idir);
117 std::cout << "composite_images_vertical_flip files found: "
118 << to_string(files.size())
119 << std::endl;
120
121 // Randomness.
122 static std::mt19937_64 rg(std::random_device{}());
123
124 const double minclose(0.6);
125 std::uniform_real_distribution<> dist1(minclose, 1.0);
126
127 const double quickestw(0.3);
128 std::uniform_real_distribution<> dist2(quickestw, totalsec - 0.5);
129
130 uint nframe(0);
131 for (const string& file: files)
132 {
133 // Background image.
134 image_element img;
135 image_element::data di = { file, 0, 0, width, height };
136 img.start_element();
137 img.add_data(di);
138 img.finish_element();
139
140 double maxclose = dist1(rg); // Random range (0.6 to 1.0)
141 double winksec = dist2(rg); // Random range (0.3, totalsec - 0.5)
142 rect_element::data dr = { 0, 0, width, height };
143 strings winks = wink_to_color_seq(dr, klr, fps, totalsec, 0,
144 maxclose, winksec);
145 for (const string& wink : winks)
146 {
147 std::ostringstream ostr;
148 ostr << basename << std::setfill('0') << std::setw(8) << nframe++;
149 svg_element obj(ostr.str(), frame);
150
151 obj.add_element(img);
152 obj.add_raw(wink);
153 }
154 }
155}
