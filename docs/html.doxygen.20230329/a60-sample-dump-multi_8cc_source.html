34/// Sample one locale.
35void
36sample_swarm_1_locale(string& idir)
37{
38 auto now = time_point_as_string(std::chrono::system_clock::now(), true);
39
40 // Gather all valid torrent files in input directory and sample.
41 strings files = io::populate_files(idir, k::torrent_s);
42 if (!files.empty())
43 {
44 // Assign torrent files to separate threads.
45 std::vector<std::thread> tvec;
46 for (string& file: files)
47 tvec.push_back(std::thread(make_session_for_torrent, file, now, 4));
48
49 // Invoke each.
50 for (auto& t : tvec)
51 t.join();
52
53 std::cout << "sample swarm finished for one locale" << std::endl;
54 }
55 else
56 std::cout << "failed to find torrent files in " << idir << std::endl;
57}

86/// Repeat 1 locale above with delay PAUSE over not more than duration INTERVAL.
87void
88sample_swarm_pause_for(string& idir, std::chrono::minutes pause,
89 std::chrono::minutes interval)
90{
91 using namespace std::chrono;
92
93 // Time arithmetic.
94 system_clock::time_point tstart = system_clock::now();
95 system_clock::time_point tend(tstart + interval);
96
97 // Setup duration subdirectory for output.
98 string odir = io::get_output_directory(time_point_as_string(tstart));
99 io::sanity_check_output_directory(odir);
100
101 // Loop.
102 uint iterations(0);
103 long samplesec(0);
104 long remainsec(0);
105 do
106 {
107 // Do it.
108 system_clock::time_point tpre = system_clock::now();
109 sample_swarm_1_locale(idir);
110 std::this_thread::sleep_for(pause);
111 system_clock::time_point tpost = system_clock::now();
112 ++iterations;
113
114 // Loop if under.
115 samplesec = duration_cast<seconds>(tpost - tpre).count();
116 remainsec = duration_cast<seconds>(tend - tpost).count();
117 }
118 while (remainsec > 0 && remainsec - (1 * samplesec) > 0);
119
120 // Teardown subdirectory.
121 io::set_output_prefix(io::get_output_prefix());
122
123 // Output useful diagnostics.
124 std::cout << "sample_swarm_pause_for: " << iterations << std::endl;
125}

128/// Repeat 3 locale above n times at intervals of no less than
129/// (interval) minutes.
130void
131sample_swarm_repeat(string& idir, uint repeater, std::chrono::minutes interval)
132{
133 using namespace std::chrono;
134
135 for (uint i = 0; i < repeater; ++i)
136 {
137 // Do it.
138 system_clock::time_point pre = system_clock::now();
139 sample_swarm_3_locale(idir);
140 system_clock::time_point post = system_clock::now();
141
142 system_clock::time_point next(pre + interval);
143 if (post < next)
144 std::this_thread::sleep_for(next - post);
145 }
146}

149/// Repeat above n times at intervals of no less than duration D
150void
151sample_swarm_repeat_repeat(string& idir, uint repeater,
152 std::chrono::hours interval, uint rinterval = 6)
153{
154 using namespace std::chrono;
155
156 string dir = io::get_output_prefix();
157 for (uint i = 0; i < repeater; ++i)
158 {
159 // Setup.
160 system_clock::time_point pre = system_clock::now();
161 string odir = io::get_output_directory(time_point_as_string(pre));
162 io::sanity_check_output_directory(odir);
163
164 // Do it.
165 sample_swarm_repeat(idir, rinterval, minutes(10));
166
167 system_clock::time_point post = system_clock::now();
168
169 // Reset toplevel directory.
170 io::set_output_prefix(dir);
171
172 // Wait until required interval expires.
173 system_clock::time_point next(pre + interval);
174 if (post < next && i + 1 < repeater)
175 std::this_thread::sleep_for(next - post);
176 }
177}

180// Pass in directory containing torrent files (idir). This directory
181// is scanned for torrent files, and in the modes where peer swarms
182// are sampled multiple times over a specified time duration, re-read
183// (idir) at each sample, so that torrent files can be continuously
184// added as the sampling progresses.
185
186// Pass in the output directory (odir), terminated with '/'.
187
188int main(int argc, char* argv[])
189{
190 using namespace std::chrono;
191
192 // Sanity check.
193 if (argc < 2 || argc > 4)
194 {
195 fputs(usage().c_str(), stderr);
196 return 1;
197 }
198
199 // Sanity check specified input torrent directory.
200 string idir(argv[1]);
201 io::sanity_check_input(idir);
202
203 // Set up output environment.
204 // Iff argument is passed for output directory, use it. Else, pwd + tmp.
205 string odir;
206 if (argc == 3)
207 {
208 odir = argv[2];
209 io::get_output_prefix(odir);
210 }
211 else
212 odir = io::get_output_directory("tmp");
213 io::sanity_check_output_directory(odir);
214
215 // Report.
216 std::cout << "sample using: " << odir << " for output" << std::endl;
217
