27/// Base Radial Visualization.
28///
29/// There are two main variants.
30/// - radial and overlapping
31/// - radial and non-overlapping "Yayoi Kusama-inspired" satellites
32/// - sized by value or not
33/// - display as list, second-layer higher orbit-satellites, arcs, etc.
34///
35/// The following three functions all support the generic display of
36/// identifiers around a circle. This
37
38/// Setup background, any other pertinent info, per dimension of identity.
39///
40/// cmarkp == if center mark should be shown.
41/// tagcountp == if title for tag and count should be shown.
42void
43render_id_base(svg_element& obj, const string tag, ulong count,
44 const point_2t origin, const int radius, const int rspace,
45 const double opacity = 1.0, const bool cmarkp = true,
46 const bool tagcountp = false)
47{
48 // Make center mark.
49 if (cmarkp)
50 {
51 // Base circle
52 if (false)
53 {
54 auto cradius = radius / 10;
55 style styl = get_metadata_style();
56 styl._M_fill_opacity = opacity;
57 point_to_circle(obj, origin, styl, cradius);
58 }
59
60 // Direction glyph.
61 if (true)
62 {
63 auto cradius = radius;
64 style stylarc = get_metadata_style(true);
65 stylarc._M_fill_opacity = 1.0;
66 stylarc.set_colors(color::wcag_lgray);
67 const auto rrangeold = set_radial_range(20, 340);
68 direction_arc_at(obj, origin, cradius, stylarc, cradius * 0.1);
69 set_radial_range(std::get<0>(rrangeold), std::get<1>(rrangeold));
70 }
71 }
72
73 // Tag and count below
74 if (tagcountp)
75 {
76 // Find max value, outermost point on the ring glyph.
77 const auto startlenm = (radius * 6) + radius;
78 const int amax = k::age_value_to_multiple(k::age_values.back());
79 const int gmax = get_min_ring_size();
80 const ulong rlenmax = std::max(std::max(amax, gmax), startlenm);
81
82 // rspace llen rspace
83 auto [xo, yo] = origin;
84 const ulong rradius = rlenmax + 2 * rspace + 2 * rspace;
85 yo += rradius;
86
87 typography typo = get_metadata_typo(12, true, true);
88 typo._M_style._M_fill_color = color::wcag_lgray;
89 sized_text(obj, typo, 18, tag, xo, yo);
90 yo += 30;
91 sized_text(obj, typo, 12, to_string(count), xo, yo);
92 }
93}

124/// If kusamap == true, use kusama with (optional) collision avoidance.

133 render_id_base(obj, tag, count, origin, radius, rspace);

220/// Darker and lighter versions of red.
221svg::color_qi
222age_value_to_color_tint_shade(const string& value)
223{
224 using svg::color;
225
226 // Numeric range is [0-11], twelve steps center at 5 == red.
227 // 6 == red
228 // 5 == combine_to_color_qi(red, .8, white, .2);
229 // 4 == combine_to_color_qi(red, .68, white, .32);
230 // 3 == combine_to_color_qi(red, .56, white, .44);
231 // 2 == combine_to_color_qi(red, .44, white, .56);
232 // 1 == combine_to_color_qi(red, .32, white, .68);
233 // 0 == combine_to_color_qi(red, .2, white, .8);
234 svg::color_qi klr;
235
236 // Compute klr.
237 // white == 255, 255, 255
238 // black == 0, 0, 0
239 // 90% light == 255, 204, 204 (lightest)
240 // 60% light == 255, 51, 51
241 // 50% light == 255, 0, 0 (aka red)
242 // 40% light == 204, 0, 0
243 // 25% light == 128, 0, 0 (darkest)
244 // -> Implies using 65% of the shades/tints of red.
245
246 const ushort val = k::age_value_to_multiple(value, true);
247 switch (val)
248 {
249 case 0:
250 case 1:
251 case 2:
252 case 3:
253 case 4:
254 case 5:
255 {
256 // Going from g,b 204 to 0 over 6 steps, 204 / 6 == 34.
257 const ushort step = 34;
258 const ushort oklr = 204 - (val * step);
259 klr = { 255, oklr, oklr };
260 break;
261 }
262 case 6:
263 klr = color::red;
264 break;
265 case 7:
266 case 8:
267 case 9:
268 case 10:
269 case 11:
270 case 12:
271 {
272 // Going from r 255 to 126 over 6 steps, 127 / 6 == 21.
273 const ushort darkerstep = 21;
274 const ushort oklr = 255 - ((val - 6) * darkerstep);
275 klr = { oklr, 0, 0};
276 break;
277 }
278 default:
279 throw std::runtime_error("age_value_to_color_tint_shade:: out of range");
280 break;
281 }
282
283 return klr;
284}

287/// Find orbit of ring, given radius, rspace, and number ring.
288/// Assume:
289/// ringsn starts with 1 for values and has a maximum value of 13.
290/// ringsn zero is unknown, which hass a ring_radius value less than radius.
291constexpr uint
292ring_radius(const int radius, const int rspace, const int ringsn)
293{
294 uint ret(0);
295 if (rspace < radius)
296 ret = radius + (rspace * (ringsn - 1));
297 return ret;
298}

301/// Grows out from center 'as if' a tree growing a tree ring.
302/// Spans unknown to [00-01, ..., 90-99] to 110+ values.
303/// Rings grow out from origin by radius + ringsn * space
304/// Iff total 8 of castmembers, 0.125 * 8 gets to 1.0 opacity, if overlayed.
305void
306ring_glyph(svg_element& obj, const point_2t& origin,
307 const int radius, const int rspace, const uint ringsn, style rstyl,
308 const uint showringsn = 1, const double opacitymax = 1.0,
309 const double strokemultiplier = 1.0)
310{
311 auto [ x, y ] = origin;
312 if (static_cast<bool>(ringsn))
313 {
314 uint j = 1;
315 for (uint i = 1; i <= ringsn; ++i)
316 {
317 // Fade up to the last (showringsn) rings, with the last ring wider.
318 if (i + showringsn > ringsn)
319 {
320 rstyl._M_stroke_opacity = opacitymax * j / showringsn;
321 if (i == ringsn)
322 rstyl._M_stroke_size *= strokemultiplier;
323 const uint rradius = ring_radius(radius, rspace, i);
324 point_2d_to_circle(obj, x, y, rstyl, rradius);
325 ++j;
326 }
327 }
328 }
329 else
330 {
331 // Assuming unknown.
332 // Unknown is mapped to zero, so fill in center with white circle.
333 svg::style ustyl = rstyl;
334 ustyl._M_fill_opacity = opacitymax;
335 ustyl._M_stroke_opacity = 0;
336 auto uradius = std::round(radius / 3);
337 point_2d_to_circle(obj, x, y, ustyl, uradius);
338 }
339}

342/// Ring glyph with arc missing on the bottom.
343void
344partial_ring_glyph(svg_element& obj, const point_2t& origin, style rstyl,
345 double rradius, const bool showvaluesp,
346 const double adelta = 25)
347{
348 if (showvaluesp)
349 {
350 // Make arc not circle, cut out bottom.
351 const double astart = zero_angle_north_cw(180);
352 const double adist = adjust_angle_at_orbit_for_distance(rradius, adelta);
353 auto p1 = get_circumference_point_d(astart + adist / 2, rradius, origin);
354 auto p2 = get_circumference_point_d(astart - adist / 2, rradius, origin);
355
356 string arc = make_path_arc_circumference(p1, p2, rradius, 1, 0);
357
358 path_element parc(true);
359 path_element::data da = { arc, 0 };
360 parc.start_element("arc-segment-" + std::to_string(rradius));
361 parc.add_data(da);
362 parc.add_style(rstyl);
363 parc.finish_element();
364 obj.add_element(parc);
365 }
366 else
367 point_to_circle(obj, origin, rstyl, rradius);
368}

372/// Weighted ring glyph.
373/// Not opacity, but breadth. More weight, wider bands around ring_radius length.
374void
375ring_w_glyph(svg_element& obj, const point_2t& origin,
376 const int radius, const int rspace, const string agev, style rstyl,
377 const int weight, const bool showvaluesp)
378{
379 // Error check computed spaces, if weight > rspace then this is not
380 // going to work.
381 if (weight > rspace)
382 {
383 string m("ring_w_glyph: error with weight supplied and rspace distance");
384 m += k::newline;
385 m += "weight: ";
386 m += to_string(weight);
387 m += k::newline;
388 m += "rspace: ";
389 m += to_string(rspace);
390 m += k::newline;
391 throw std::runtime_error(m);
392 }
393
394 // Find starting ring radius.
395 // When @ringsn == 0 -> unknown. All other values are linear.
396 const uint ringsn = k::age_value_to_multiple(agev);
397 double rradius(0.0);
398 if (ringsn > 1)
399 {
400 rradius = ring_radius(radius, rspace, ringsn);
401 partial_ring_glyph(obj, origin, rstyl, rradius, showvaluesp);
402 }
403 else if (ringsn == 1)
404 {
405 rradius = ring_radius(radius, rspace, ringsn);
406 partial_ring_glyph(obj, origin, rstyl, rradius, showvaluesp, rradius * 2);
407 }
408 else
409 {
410 // Assuming unknown.
411 // Unknown is mapped to zero, so fill in center with white circle.
412 svg::style ustyl = rstyl;
413 ustyl._M_fill_opacity = 1.0;
414 ustyl._M_stroke_opacity = 0;
415 rradius = std::round(radius / 3);
416 point_to_circle(obj, origin, ustyl, rradius);
417 }
418
419 // If weighted, inscribe orbits (or partial orbits for unknown or 00-09 values.
420 const double echom(double(rspace) / (weight + 1));
421 for (int i = 1; i < weight; ++i)
422 {
423 rradius += echom;
424 if (ringsn > 1)
425 partial_ring_glyph(obj, origin, rstyl, rradius, showvaluesp);
426 else
427 partial_ring_glyph(obj, origin, rstyl, rradius, showvaluesp, rradius * 2);
428 }
429
430 // Show value, if requested.
431 if (showvaluesp)
432 {
433 typography typo = get_radial_typo(5); // If other radial typo is 6
434 typo._M_baseline = typography::baseline::text_top;
435
436 auto rr = ring_radius(radius, rspace, ringsn);
437 auto [ ax, ay ] = origin;
438 point_2t vp = { ax, ay + rr + (typo._M_size / 2) };
439 styled_text(obj, agev, vp, typo);
440 }
441}

478/// Ages and ages playing composited into one ring glyph.
479/// Select, sort, and composite if necessary the various age variants
480/// like playing and multiples of both natural and playing ages.
481///
482/// Create a cocentric number of rings around center radius,
483/// with constant spacing rspace between the rings
484///
485/// agev is one value from the 'cast-lead-1-ages' field in the data file.
486/// radius is the smallest circle (aka age 00-09)'s radius.
487/// rspace is the distance between rings
488/// linewidth is the stroke size of each ring
489void
490composite_age_glyph(svg_element& obj, const point_2t& origin, const string agev,
491 const int radius, const int rspace = 12,
492 const double opacitymax = 1.0, const double linewidth = 3,
493 const color klr = color::black,
494 const color fadeklr = color::white)
495{
496 const auto playingpos = agev.find(k::playing);
497 if (playingpos == string::npos)
498 {
499 // Natural age, same as playing age, black hairline.
500 svg::style rstyl = get_metadata_style(false);
501 rstyl._M_stroke_size = linewidth;
502 rstyl._M_fill_opacity = 0.0;
503
504 // Set special style for "100+" age value.
505 ulong m = k::age_value_to_multiple(agev);
506 if (m == 13)
507 {
508 svg::style rwstyl = get_metadata_style();
509 rwstyl._M_fill_opacity = 0.0;
510
511 rwstyl._M_stroke_size = 3 * linewidth;
512 ring_glyph(obj, origin, radius, rspace, m, rwstyl, 1);
513
514 rstyl._M_stroke_size = linewidth;
515 ring_glyph(obj, origin, radius, rspace, m, rstyl, 1, 1.0);
516 }
517 else
518 ring_glyph(obj, origin, radius, rspace, m, rstyl, 1, opacitymax, 1.0);
519 }
520 else
521 {
522 strings agesvo;
523 strings agesplaying;
524 split_playing_value(agev, agesvo, agesplaying);
525
526 // Make radial blur for each playing value.
527 for (const string& agepv : agesplaying)
528 {
529 // Compute ring radius in the same way as ring_glyph.
530 ulong m = k::age_value_to_multiple(agepv);
531 const uint rradius = ring_radius(radius, rspace, m);
532 const double rrblur = radius * 0.3;
533 // Draw circle with gradients outside and inside circumference r.
534 point_to_ring_halo(obj, origin, rradius, rrblur, klr, fadeklr);
535 }
536
537 // Make ring for each natural value.
538 for (const string& agenv : agesvo)
539 composite_age_glyph(obj, origin, agenv, radius, rspace,
540 opacitymax, linewidth);
541 }
542}

599/// Entry point specialized for age dimension for media object.
600void
601render_ids_ages(svg_element& obj, const string tag, ulong count,
602 const sids& ids, const point_2t origin, const int radius,
603 const int rspace = 12, const double opacitymax = 1.0,
604 const bool showtagp = true)
605{
606 // Make circle at point (xoff, yoff)
607 if (showtagp)
608 render_id_base(obj, tag, count, origin, radius, rspace, 0, false);
609
610 double opacity(opacitymax);
611 if (opacitymax > 1.0 || opacitymax < 0.0)
612 opacity = 1.0;
613
614 // Sanity check.
615 if (count)
616 {
617 for (const id_type& id : ids)
618 {
619 auto [ weight, agev ] = id;
620
621 // Fill in for larger weights by increasing opacity and radius.
622 double ototal(opacity * weight);
623 if (ototal < 1.0)
624 composite_age_glyph(obj, origin, agev, radius, rspace, ototal);
625 else
626 {
627 // Find weight that trips 100% opacity (aka 1.0 value).
628 const double saturationn = 1.0 / opacity;
629
630 const double rspacedelta(rspace / 10);
631
632 int radiusb(radius);
633 while (weight > 0)
634 {
635 double eweight(std::min(double(weight), saturationn));
636 double ctotal(opacity * eweight);
637 composite_age_glyph(obj, origin, agev, radiusb, rspace,
638 ctotal);
639 weight -= eweight;
640 radiusb += rspacedelta;
641 }
642 }
643 }
644 }
645}

648/// Entry point specialized for age dimension for aggregate.
649void
650render_ids_ages(svg_element& obj, const string tag, ulong count,
651 const sids& ids, const sids& idsp,
652 const point_2t origin, const int radius,
653 const int rspace, const double /* opacitymax */ = 1.0,
654 const bool showtagp = true)
655{
656 // Show age values as well as glyph.
657 const bool showvp = true;
658
659 // Make circle at point (xoff, yoff)
660 if (showtagp)
661 render_id_base(obj, tag, count, origin, radius, rspace, 0, false);
662
663 // Sanity check.
664 if (count)
665 {
666 // Simplify all compound values playing.
667 umids vmids;
668 umids vpmids;
669 for (const id_type& id : idsp)
670 {
671 const auto [ weight, agev ] = id;
672
673 strings lhs;
674 strings rhs;
675 split_playing_value(agev, lhs, rhs);
676
677 // LHS values from split
678 for (const string& s : lhs)
679 insert_ids_into(vmids, make_tuple(weight, s));
680
681 // RHS playing values from split
682 for (const string& s : rhs)
683 insert_ids_into(vpmids, make_tuple(weight, s));
684 }
685 for (const id_type& id : ids)
686 {
687 const auto [ weight, agev ] = id;
688 insert_ids_into(vmids, make_tuple(weight, agev));
689 }
690
691 // Find centers.
692 auto [ acp, apcp ] = diptych_xdipoles(origin, radius, rspace, 0.5);
693 if (vmids.empty() || vpmids.empty())
694 {
695 acp = origin;
696 apcp = origin;
697 }
698
699 const auto subsetsz = 10;
700 typography typos = get_metadata_typo(subsetsz, false, false);
701
702 // Set a small line width to avoid overlapping.
703 const double linewidth = 0.5;
704
705 // Render (values playing).

1060/// Text grow-up columns, horizontal with values (LHS) and values playing (RHS).
1061/// Entry point specialized for values/valuesplaying 2-column, side-by-side.
1062/// values are right aligned
1063/// valuesplaying are left aligned
1064/// if either ids/idps are empty, move to centered 1-column.
1065/// Minimum and maximum text size are @minsz and @maxsz
1066/// NB: dimension with values (ids) and values playing (idps) separated.
1067void
1068render_ids_text_x(svg_element& obj, ulong count, const point_2t origin,
1069 const auto& ids, const auto& idps,
1070 const int radius, const int rspace,
1071 const uint minsz = 7, const uint maxsz = 24)
1072{
1073 // Find range of weights.
1074 uint maxweight(0);
1075 for (const id_type& id : ids)
1076 {
1077 auto [weight, name] = id;
1078 maxweight = std::max(maxweight, weight);
1079 }
1080 for (const id_type& id : idps)
1081 {
1082 auto [weight, name] = id;
1083 maxweight = std::max(maxweight, weight);
1084 }
1085
1086 // Find range of y values given radius, ids, scaled weights.
1087 uint ydelta(0);
1088 for (const id_type& id : ids)
1089 {
1090 auto [weight, name] = id;
1091 uint sz = normalize_value_on_range(weight, 1, count, minsz, maxsz);
1092 ydelta += sz;
1093 }
1094 uint ydeltap(0);
1095 for (const id_type& id : idps)
1096 {
1097 auto [weight, name] = id;
1098 uint sz = normalize_value_on_range(weight, 1, count, minsz, maxsz);
1099 ydeltap += sz;
1100 }
1101 ydelta = std::max(ydelta, ydeltap);
1102
1103 // Sanity check total distance isn't too large, it will overlap labels.
1104 // If this is the case, use a mulitplier.
1105 // from render_id_base, tag title is at 5.5 * radius.
1106 double yadjust(1.0);
1107 const double ydeltamax(2 * radius);
1108 if (ydelta > ydeltamax)
1109 {
1110 ydelta = static_cast<uint>(ydeltamax);
1111 yadjust = ydeltamax / ydelta;
1112 }
1113
1114 // Setup origins for v, vplaying...
1115 auto [ xo, yo ] = origin;
1116
1117 // Set vertical - y values
1118 // Grow text from bottom up, so sink from middle to start text..
1119 const auto y = yo + std::round(double(ydelta) / 2);
1120
1121 // Set horizontal - x values.
1122 // Only do dipoles if both values and values playing.
1123 double xv(0);
1124 double xvp(0);
1125 bool onlyonep = false;
1126 if (ids.empty() || idps.empty())
1127 {
1128 xv = xo;
1129 xvp = xo;
1130 onlyonep = true;
1131 }
1132 else
1133 {
1134 xv = xo - rspace;
1135 xvp = xo + rspace;
1136 }
1137
1138 // Draw text values from bottom up.
1139 if (count)
1140 {
1141 // Values.
1142 auto yv = y;
1143 typography typov = get_metadata_typo(minsz, true, true);
1144 if (!onlyonep)
1145 {
1146 typov._M_anchor = svg::typography::anchor::end;
1147 typov._M_align = svg::typography::align::right;
1148 }
1149 style stylv = get_metadata_style(true);
1150 typov._M_style = stylv;
1151 for (const id_type& id : ids)
1152 {
1153 auto [weight, v] = id;
1154 uint sz = normalize_value_on_range(weight, 1, count,
1155 minsz, maxsz);
1156
1157 const point_2t p = { xv, yv + sz / 2};
1158 uint yoff = text_grow_up_from(obj, p, v, sz * yadjust, typov);
1159 yv -= yoff;
1160 }
1161
1162 // Values playing.
1163 auto yvp = y;
1164 typography typovp = get_metadata_typo(minsz, true, true);
1165 if (!onlyonep)
1166 {
1167 typovp._M_anchor = svg::typography::anchor::start;
1168 typovp._M_align = svg::typography::align::left;
1169 }
1170 style stylvp = get_metadata_style(false);
1171 typovp._M_style = stylvp;
1172 typovp._M_style._M_fill_color = color::wcag_gray;
1173 for (const id_type& id : idps)
1174 {
1175 auto [weight, v] = id;
1176 uint sz = normalize_value_on_range(weight, 1, count,
1177 minsz, maxsz);
1178 const point_2t p = { xvp, yvp + sz / 2};
1179 const string sv = shorten_playing_value(v);
1180 uint yoff = text_grow_up_from(obj, p, sv, sz * yadjust, typovp);
1181 yvp -= yoff;
1182 }
1183 }
1184}
