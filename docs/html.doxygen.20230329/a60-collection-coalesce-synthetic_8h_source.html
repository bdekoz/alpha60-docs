27/**
28 Sanity check BTIH array inputs and mark duplicates, exclude
29 corrupt files, subset to a single episode (if desired).
30
31 Returns a vector where each element is a BTIH containing vector
32 pruned torrent file name matches.
33*/
34vvstrings
35find_duplicate_btih_in_torrent_collection(const string& idir,
36 const string matchepisode = "")
37{
38 // Use a multimap of (BTIH, file).
39 using mmtype = std::multimap<std::string, std::string>;
40 using mmrange = std::pair<mmtype::iterator, mmtype::iterator>;
41 mmtype mm;
42
43 // Populate the multimap with this torrent collection's BTIHA.
44 strings files = io::populate_files(idir);
45 std::cout << "find_duplicate_btih_in_torrent_collection: " << std::endl
46 << files.size() << " files in collection " << std::endl
47 << idir << std::endl;
48 std::cout << std::endl;
49
50 // BTIHA.
51 strings names;
52
53 uint filen(0);
54 strings errorfiles;
55 for (const std::string& f : files)
56 {
57 string name = io::prune_path(f, true);
58
59 // Check to make sure the extension is valid (torrent).
60 auto postor = name.rfind(k::torrent_s);
61 if (postor == string::npos)
62 {
63 std::cout << "error: cannot find extension " << k::torrent_s
64 << " in file " << name << std::endl;
65 errorfiles.push_back(name);
66 continue;
67 }
68 else
69 {
70 // Clip extension.
71 name = name.substr(0, postor);
72 }
73
74 // Check to see if multiepisode or sub-collection.
75 if (!matchepisode.empty())
76 {
77 auto pmatchepisode = name.find(matchepisode);
78 if (pmatchepisode == string::npos)
79 {
80 std::cout << "error: multiepisode issue for match "
81 << matchepisode << " in file: " << name << std::endl;
82 errorfiles.push_back(name);
83 continue;
84 }
85 }
86
87 std::cout << "file " << filen++ << " <--> " << name << std::endl;
88
89 // Make torrent info.
90 lt::error_code ec;
91 lt::add_torrent_params p;
92 p.ti = std::make_shared<lt::torrent_info>(f, std::ref(ec));
93 if (ec)
94 {
95 std::cout << "error: invalid torrent file: " << name << std::endl;
96 errorfiles.push_back(name);
97 }
98 else
99 {
100 auto& ti = *(p.ti);
101 std::ostringstream outs;
102 outs << ti.info_hash();
103 std::string tih = outs.str();
104 mm.insert(std::make_pair(tih, name));
105 }
106 }
107 std::cout << std::endl;
108
109 // Summary, look for duplicates for given key (BTIH) and save off key.
110 usstrings btihs;
111 for (const auto& p : mm)
112 {
113 std::string key = p.first;
114 uint k = mm.count(key);
115 if (k > 1)
116 btihs.insert(key);
117 }
118 std::cout << "duplicate BTIH: " << btihs.size() << std::endl;
119
120 uint errorn = errorfiles.size();
121 std::cout << "torrent errors: " << errorn << std::endl;
122 if (errorn > 0)
123 {
124 for (const string& s : errorfiles)
125 std::cout << "error: " << s << std::endl;
126 }
127 std::cout << std::endl;
128
129 // Each entry is a vector of pruned filenames that have the same BTIH.
130 vvstrings duplicates;
131 if (btihs.empty())
132 return duplicates;
133
134
135 // Detail, look for duplicates for given key (BTIH) and print matches.
136 uint countn(0);
137 for (const auto& key : btihs)
138 {
139 std::cout << countn++ << " " << key << std::endl;
140
141 // Extract from multimap for all filenames mapped to this BTIH.
142 vstrings samebtihnames;
143 mmrange mr = mm.equal_range(key);
144 for (mmtype::iterator it = mr.first; it != mr.second; ++it)
145 {
146 samebtihnames.push_back(it->second);
147 std::cout << " " << it->second << std::endl;
148 }
149 std::cout << std::endl;
150 duplicates.push_back(samebtihnames);
151 }
152
153 std::sort(duplicates.begin(), duplicates.end());
154 return duplicates;
155}

161 const std::string dir = cll.torrent_directory();

166/// Using previously coalesced / cachedfiles for input,
167/// re-partition torrent collection to include only unique BTIH,
168/// replacing any duplicate BTIH with a synthetic multi torrent that
169/// incorporates all duplicates.
170///
171/// Reserialize with syntheic meta torrents to "tmp" output directory
172/// with a different naming scheme (prefixed with "unique-btih").
173///
174/// durationm is either
175/// quanta (a60::k::cumulative_s, day_s, week_s) to match.
176/// finite (a day_s + a unit like "day-00001")
177///
178/// durationprev
179/// if not empty, then can assume durationm is just a quanta
180/// this is durationm + optional digits (so can do at_and_previous)
181cachecll
182coalesce_with_unique_btiha(const collection cll, const string& durationm,
183 const string durationprev = "",
184 const bool equal_and_previous_p = false,
185 geo::meta_mask mask = cachecll::default_mask(),
186 const bool reserializep = false)
187{
188 // Most specialized duration quanta to expect.
189 const string durationmost = durationprev.empty() ? durationm : durationprev;
190 const auto dcomparefn = equal_and_previous_p ? equal_at_and_previous : equal_at;
191
192 const string prefix("unique-btiha-");
193 const string udurationm(prefix + durationmost);
194
195 // Setup logging.
196 const string odir = io::get_output_prefix("tmp");
197 io::sanity_check_output_directory(odir);
198 const string pl = cll.match + '-' + udurationm + ".log";
199 std::ofstream ofsp(odir + pl, std::ios_base::app | std::ios_base::out);
200 ofsp << "amass_unique_btih_each_" << durationmost << std::endl;
201
202 // Partition collection into uniques and duplicates...
203 collection cllu(cll);
204 collection clldup(cll);
205
206 // Find set of most specialized duration cache files...
207 const auto& rtr = io::get_run_time_resources();
208 const string cdir = rtr.cache;
209 strings cfiles = cllu.cache_files_by(cdir, durationmost);
210 ofsp << cfiles.size() << " cache files found for " << durationm << std::endl;
211 if (cfiles.empty())
212 {
213 // There are no cache files, because perhapse the durationm
214 // string doesn't exist (ie, looking for week-00009 in a 8 week
215 // sample set....
216 // Return empty.
217 return cachecll(cllu);
218 }
219
220 // IFF duration at and previous only, expand set of found cache files.
221 if (equal_and_previous_p)
222 {
223 strings cfilesall = cllu.cache_files_by(cdir, durationm);
224 strings cfilesprev;
225 for (const string& f: cfilesall)
226 if (equal_at_and_previous(cll, f, durationprev))
227 cfilesprev.push_back(f);
228 cfiles = cfilesprev;
229 }
230
231 // Next, find duplicates in existing BTIHA
232 // Flatten to unordered set of duplicated names.
233 vvstrings duplicates = find_duplicate_btih_in_torrent_collection(cll);
234 cached_strings cacheds;
235 for (const vstrings& vs : duplicates)
236 std::copy(vs.begin(), vs.end(), std::inserter(cacheds, cacheds.end()));
237
238 // Iff no duplicates, just return cached collection.
239 if (cacheds.empty())
240 {
241 cachecll wcllu(cllu);
242 wcllu.key_mask = mask;
243 wcllu.add_geo_peers_from_glob_files(cfiles);
244
245 ofsp << "no duplicate btih found: " << cllu.members.size() << std::endl;
246 return wcllu;
247 }
248 else
249 {
250 ofsp << "total # duplicated names: " << cacheds.size()
251 << " of " << cll.torrent_files().size() << std::endl;
252 }
253
254 // Else, there were some duplicates. Partition into:
255 // wcllu == cachecll unique, no duplicate btih
256 // cachecll duplicates
257 a60::collection::tmarkups tmu = clldup.partition_by_names(cacheds);
258 if (tmu.empty())
259 {
260 string m("coalesce_with_unique_btiha for " + durationm);
261 m += " could not repartition btiha for duplicates of size: ";
262 m += to_string(cacheds.size());
263 throw std::runtime_error(m);
264 }
265 cllu.members = tmu;
266 cachecll wcllu(cllu);
267 wcllu.key_mask = mask;
268
269 ofsp << "cache btihs size " << clldup.members.size() << std::endl;
270 ofsp << "cache unique size " << wcllu.members.size() << std::endl;
271
272 // If there are files to cache, then do so...
273 wcllu.add_geo_peers_from_glob_files(cfiles);
274
275 // Reserialize if be.
276 // Re-save unique cache files with udurationm marking for convenience...
277 string durationmatch("-" + durationm + "-");
278 string udurationmatch("-" + udurationm + "-");
279 for (uint i = 0; i < wcllu.members.size() && reserializep; ++i)
280 {
281 ulong stid = wcllu.members[i].tid;
282 string stmangle(wcllu.mangle_tid_match_id(stid, durationm));
283 for (const string& f : cfiles)
284 {
285 string m;
286 if (f.find(stmangle) != string::npos)
287 {
288 auto pos = f.find(durationmatch);
289 if (pos != string::npos)
290 {
291 string fprime(f);
292 fprime.replace(pos, durationmatch.size(), udurationmatch);
293 fprime = io::prune_path(fprime);
294
295 // No need to parse file, just copy to new name.
296 std::ifstream ifs(f);
297 std::ofstream ofs(odir + fprime);
298 if (ifs.good() && ofs.good())
299 {
300 ofs << ifs.rdbuf();
301 break;
302 }
303 else
304 {
305 m = "coalesce_with_unique_btiha: file copy fail";
306 m += k::newline;
307 m += f;
308 m += k::newline;
309 m += fprime;
310 m += k::newline;
311 }
312 }
313 else
314 {
315 m = "coalesce_with_unique_btiha: not found " + durationmatch;
316 m += " in ";
317 m += f;
318 }
319 }
320
321 // Error path.
322 if (!m.empty())
323 {
324 m += k::newline;
325 throw std::runtime_error(m);
326 }
327 }
328 }
329
330 // Re-cache duplicated cache files into synthetic unique BTIH.
331 const string dompeert = "peers";
332 const string domseedt = "seeds";
333 for (uint i = 0; i < duplicates.size(); ++i)
334 {
335 // Name for newly-generated synthetic torrent.
336 // NB: The duplicates vector is sorted so front is shortest.
337 vstrings& samebtihnames = duplicates[i];
338 const string ubasename = samebtihnames.front();
339 std::ostringstream oss;
340 oss << "multi-btih-" << i << "-" << samebtihnames.size() << "-x-"
341 << ubasename;
342
343 meta_torrent synthtor;
344 synthtor.name = oss.str();
345 synthtor.tid = wcllu.set_tid(synthtor.name);
346
347 string lcs = to_lowercase_string(synthtor.name);
348 synthtor.resolution = wcllu.get_resolution(lcs, ubasename);
349 synthtor.encoding = wcllu.get_encoding(lcs);
350
351 wcllu.members.push_back(synthtor);
352
353 // Then, for each muti-btiha, compress into one unique result with a
354 // new name and id.
355 usmeta_peers upeers;
356 usmeta_peers useeds;
357 for (const string& s : samebtihnames)
358 {
359 // For each group of duplicates:
360 // Remove duplicate torrents, and find the superset for all
361 // unique peers for duplicate 1
362 // unique peers for duplicate 2
363 // unique peers for duplicate n
364 ulong stid = wcllu.get_tid(s);
365 string stmangle(wcllu.mangle_tid_match_id(stid));
366 for (const string& f : cfiles)
367 {
368 const bool tidmatchp = f.find(stmangle) != string::npos;
369 const bool durationmatchp = dcomparefn(cll, f, durationmost);
370 if (tidmatchp && durationmatchp)
371 {
372 rj::Document dom = deserialize_json_to_dom_object(f);
373 vmeta_peers apeers = search_dom_for_meta_peers(dom, dompeert);
374 vmeta_peers aseeds = search_dom_for_meta_peers(dom, domseedt);
375
376 upeers.insert(apeers.begin(), apeers.end());
377 useeds.insert(aseeds.begin(), aseeds.end());
378
379 ofsp << s << std::endl;
380 ofsp << f << std::endl;
381 ofsp << k::tab << "peers: " << apeers.size()
382 << k::tab << "seeds: " << aseeds.size() << std::endl;
383 ofsp << std::endl;
384 }
385 }
386 }
387
388 ofsp << i << k::tab << synthtor.name << std::endl;
389 ofsp << "total unique peers: " << upeers.size() << std::endl;
390 ofsp << "total unique seeds: " << useeds.size() << std::endl;
391
392 // Now add in synth tor to cached uniques, wcllu.
393 cachecll::umgeo_peers pgeopeers;
394 cachecll::umgeo_ranks pweights;
395
396 cachecll::umgeo_peers sgeopeers;
397 cachecll::umgeo_ranks sweights;
398
399 // Have a set of unique peers, add in all from each file.
400 wcllu.insert_meta_peers_by_meta_mask(upeers, pgeopeers, pweights,
401 wcllu.key_mask);
402 wcllu.peer_geos.push_back(pgeopeers);
403 wcllu.peer_weights.push_back(pweights);
404 wcllu.peer_totals.push_back(upeers.size());
405 wcllu.peer_wkeys.push_back(cachecll::swgeo_peers());
406
407 wcllu.insert_meta_peers_by_meta_mask(useeds, sgeopeers, sweights,
408 wcllu.key_mask);
409 wcllu.seed_geos.push_back(sgeopeers);
410 wcllu.seed_weights.push_back(sweights);
411 wcllu.seed_totals.push_back(useeds.size());
412 wcllu.seed_wkeys.push_back(cachecll::swgeo_peers());
413
414 // Reserialize the de-duplicated swarm_gcache_object.
415 if (reserializep)
416 {
417 reserialize_sample_json(upeers, useeds, wcllu, odir, synthtor.tid,
418 udurationm, 0, wcllu.datestamp);
419 }
420 }
421 return wcllu;
422}
