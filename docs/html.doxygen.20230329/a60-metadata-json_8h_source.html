29/// Convert a vector of strings to a flattened and
30/// comma-space-separated list of strings.
31const string
32strings_to_cssv_string(const vstrings& in)
33{
34 string flat;
35 uint i = 0;
36 for (const string& s : in)
37 {
38 if (i > 0)
39 flat += ", ";
40 flat += s;
41 i++;
42 }
43 return flat;
44}

47/// As above, but only enough of the input vector to make maxlen string.
48/// Argument is not const: modified to have unused strings vector returned.
49const string
50strings_to_cssv_string_len(vstrings& in, uint maxlen)
51{
52 string ret;
53 if (!in.empty())
54 {
55 const uint cslen = 2;
56
57 uint len(0);
58 auto iter = in.begin();
59 for (; iter != in.end(); ++iter)
60 {
61 const string s = *iter;
62 const uint addlen = s.size() + cslen;
63 if (addlen > maxlen)
64 {
65 string m("strings_to_cssv_string_len:: ");
66 m += s;
67 m += " has length with comma and space of ";
68 m += to_string(addlen);
69 m += ", ";
70 m += "which is larger than the maximum line length ";
71 m += to_string(maxlen);
72 m += k::newline;
73 throw std::runtime_error(m);
74 }
75 if (len + addlen <= maxlen)
76 len += addlen;
77 else
78 break;
79 }
80
81 // Remove part used from originating string.
82 if (iter != in.end())
83 {
84 vstrings partial(in.begin(), iter);
85 ret = strings_to_cssv_string(partial);
86 in.erase(in.begin(), iter);
87 }
88 else
89 {
90 ret = strings_to_cssv_string(in);
91 in.clear();
92 }
93 }
94 return ret;
95}

98/// Convert a comma-space-separated list of strings to a vector of
99/// strings
100vstrings
101cssv_string_to_strings(string& in)
102{
103 strings agesv;
104 auto commapos = in.find(k::comma);
105 while (commapos != string::npos)
106 {
107 string agepv = in.substr(0, commapos);
108 agesv.push_back(agepv);
109 in = in.substr(commapos + 1 + 1);
110 commapos = in.find(k::comma);
111 }
112 agesv.push_back(in);
113 return agesv;
114}

117/// Input string vector @values is the vector of string field values
118/// from one field, say character-lead-1-ages
119/// Separated values set to arguments @values and @valuesplaying.
120///
121/// Default:
122/// @values are all values without "playing"
123/// or the prefix part of compound values containing prefix.
124/// @valuesplaying are all postfix values containing "playing"
125///
126/// IFF onlyplayingp
127/// @values is null if any playing
128/// @valuesplaying only value containing "playing"
129void
130separate_playing_values(strings& valueso, strings& valuesplaying)
131{
132 // Actor or individual age values, say.
133 strings values;
134
135 // Split out values from values playing.
136 for (const string& v : valueso)
137 {
138 const auto playingpos = v.find(k::playing);
139 if (playingpos == string::npos)
140 values.push_back(v);
141 else
142 valuesplaying.push_back(v);
143 }
144
145 valueso = values;
146}

149/// Split playing value into
150/// actor -> LHS of playing "person"
151/// role -> RHS of playing "persona"
152/// NB: Each side can be mulitple values.
153void
154split_playing_value(const string vo, strings& vactor, strings& vrole)
155{
156 // Find the playing age or ages, aka after "playing", after a space.
157 // Look for commas and multiple values.
158 const auto playingpos = vo.find(k::playing);
159 if (playingpos != string::npos)
160 {
161 string agevo = vo.substr(0, playingpos - 1);
162 vactor = cssv_string_to_strings(agevo);
163 string ageplaying = vo.substr(playingpos + k::playing.size() + 1);
164 vrole = cssv_string_to_strings(ageplaying);
165 }
166}

169// Get canonical media object name.
170// If season, return name with season. If episode, return episode.
171// NB: used to be name-key, or name-key-canonical.
172string
173canonical_media_name(const rj::Document& dom)
174{
175 string name("");
176 const string sn("name");
177 const string snkey("name-key");
178 const string snkeyc("name-key-canonical");
179 const string snid("id");
180
181 if (!dom.HasParseError() && dom.HasMember(snkey.c_str()))
182 {
183 const rj::Value& nv = dom[snkey.c_str()];
184 if (nv.IsArray())
185 {
186 // If name-key-canonical, then use that.
187 const rj::Value& nvc = dom[snkeyc.c_str()];
188 string namec = nvc.GetString();
189 if (namec.empty())
190 {
191 // Else, name (minus the) + id fields.
192 string namep = dom[sn.c_str()].GetString();
193 string namelc = to_replace_string(to_lowercase_string(namep));
194 if (!namelc.empty())
195 {
196 // Remove leading "the "
197 auto thepos = namelc.find("the-");
198 if (thepos != string::npos && thepos == 0
199 && namelc.size() > 4)
200 namelc = namelc.substr(4);
201
202 // Remove or change punctuation like ':' or '/'
203 string np1 = to_erase_string(namelc, k::colon);
204 string np2 = to_erase_string(np1, k::period);
205 string np3 = to_replace_string(np2, '/');
206
207 // Finalize text part of name.
208 name = np3;
209
210 // Append season part of name, if be.
211 string season = search_dom_for_string_field(dom, snid);
212 if (!season.empty())
213 {
214 std::ostringstream ostr;
215 ostr << k::hyphen;
216 ostr << std::setfill('0') << std::setw(2) << stoi(season);
217 season = ostr.str();
218 name += season;
219 }
220 }
221 }
222 else
223 {
224 // Use canonical name.
225 name = namec;
226 }
227 }
228 else
229 {
230 // Isn't array, so use un-altered name-key field..
231 name = nv.GetString();
232 }
233 }
234
235 return name;
236}

239/// Find metadata files containing.
240/// Find field of name 'field' in JSON file and return vector of
241/// name-keys (namekeyp == true) or file paths matching (namekeyp == false).
242strings
243files_with_field_matching(const strings files,
244 const string field, const string match,
245 const bool namekeyp = false)
246{
247 strings nmatching;
248 for (const string& f: files)
249 {
250 rj::Document dom = deserialize_json_to_dom_object(f);
251 if (!dom.HasParseError() && dom.HasMember(field.c_str()))
252 {
253 const rj::Value& a = dom[field.c_str()];
254
255 if (a.IsArray())
256 {
257 for (uint i = 0; i < a.Size(); ++i)
258 {
259 const rj::Value& v = a[i];
260 string distrot;
261 if (v.IsString())
262 distrot = v.GetString();
263 else
264 distrot = std::to_string(v.GetInt());
265 if (distrot == match)
266 {
267 if (namekeyp)
268 {
269 string snk = canonical_media_name(dom);
270 nmatching.push_back(snk);
271 }
272 else
273 nmatching.push_back(f);
274 }
275 }
276 }
277 }
278 else
279 {
280 string m("files_with_field_matching:: ");
281 if (dom.HasParseError())
282 {
283 m += "error: json parse failed in file: ";
284 m += f;
285 m += k::newline;
286 }
287 else
288 {
289 if (!dom.HasMember(field.c_str()))
290 {
291 m += "skipped, no member field: ";
292 m += k::quote;
293 m += field;
294 m += k::quote;
295 m += " in file: ";
296 m += f;
297 m += k::newline;
298 }
299 }
300 std::clog << m << std::endl;
301 throw std::runtime_error(m);
302 }
303 }
304 return nmatching;
305}

369/// Find name 'field' in JSON file and return vector of values
370strings
371values_from_field(const rj::Document& dom, const string field,
372 const string wfield = "", const string wvalue = "")
373{
374 strings values;
375
376 // Parse metadata file if @field is found.
377 if (!dom.HasParseError() && dom.HasMember(field.c_str()))
378 {
379 // If WITH wfield and wvalue then make sure match condition is true.
380 bool xtractwp = true;
381 if (!wfield.empty() && !wvalue.empty() && dom.HasMember(wfield.c_str()))
382 {
383 const rj::Value& aw = dom[wfield.c_str()];
384 if (aw.IsArray())
385 {
386 bool findmatchp = false;
387 for (uint i = 0; i < aw.Size() && !findmatchp; ++i)
388 {
389 const rj::Value& v = aw[i];
390 string value = v.GetString();
391 if (!value.empty() && value == wvalue)
392 findmatchp = true;
393 }
394
395 // If no matches found then don't extract.
396 if (!findmatchp)
397 xtractwp = false;
398 }
399 }
400
401 // Iff extract condition met, return values.
402 if (xtractwp)
403 {
404 const rj::Value& a = dom[field.c_str()];
405 if (a.IsArray())
406 {
407 for (uint i = 0; i < a.Size(); ++i)
408 {
409 const rj::Value& v = a[i];
410 string value = v.GetString();
411 if (!value.empty())
412 values.push_back(value);
413 }
414 }
415 }
416 }
417 return values;
418}

437/// Find the values/values playing for fields containing agg.
438void
439values_from_fields_matching(const rj::Document& dom, const string agg,
440 msstrings& v, msstrings& vplaying,
441 const string wagg = "", const string wvalue = "")
442{
443 for (uint i = 1; i < 9; ++i)
444 {
445 const string field("cast-lead-" + to_string(i) + "-" + agg);
446 const string wfield("cast-lead-" + to_string(i) + "-" + wagg);
447 strings values = values_from_field(dom, field, wfield, wvalue);
448
449 // If no "playing" values, then assume actor & cast are the same.
450 strings valuesplaying;
451 separate_playing_values(values, valuesplaying);
452 v.insert(values.begin(), values.end());
453 vplaying.insert(valuesplaying.begin(), valuesplaying.end());
454 }

608/**
609 From @files to ranked/ordered vsids for aggregate.
610
611 For each file:
612 Slurp in all the individual cast 1 fields
613 Slurp in all the individual cast 2 fields
614 ...
615 Slurp in all the individual cast 8 fields
616
617 Then return a vector[id_dimensions.size()] of aggregate id_types,
618 where each dimension's vids is sorted from most used value to least.
619
620 The @dimensions argument is the empty vsids vector to be populated
621 by natural values, aka non-value-playing.
622
623 The @dimensionsp argument is the empty vsids vector to be populated
624 by playing values, aka "10-19 playing 20-29".
625
626 Two condition fields default to an empty string, which is the default.
627
628 When @wfield and @wvalue are not empty, then only populate the
629 return vectors for values where wfield-n variants have @wvalue.
630 So, with @wfield set to "genders" and @wvalue set to "female" then
631 the return vectors will only be populated for fields
632 WITH cast-n-"genders" AS "female".
633 */
634void
635get_vsids_from_metadata(const strings& files,
636 vsids& dimensions, vsids& dimensionsp,
637 const string wfield = "", const string wvalue = "")
638{
639 // Collect data in multisets (one for each dimension).
640 std::vector<msstrings> vvalues(k::id_dimensions.size());
641 std::vector<msstrings> vvaluesplaying(k::id_dimensions.size());
642
643 for (const string& f : files)
644 {
645 rj::Document dom = deserialize_json_to_dom_object(f);
646
647 uint i = 0;
648 for (const string& idtag : k::id_dimensions)
649 {
650 msstrings& values = vvalues[i];
651 msstrings& valuesplaying = vvaluesplaying[i];
652 values_from_fields_matching(dom, idtag, values, valuesplaying,
653 wfield, wvalue);
654 ++i;
655 }
656 }
657
658 // Convert
659 for (uint j = 0; j < k::id_dimensions.size(); ++j)
660 {
661 // Values.
662 msstrings& values = vvalues[j];
663 msstrings& valuesplaying = vvaluesplaying[j];
664
665 sids vids = multiset_to_set_of_ids(values);
666 dimensions.push_back(vids);
667
668 // Values playing.
669 sids vplayingids = multiset_to_set_of_ids(valuesplaying);
670 dimensionsp.push_back(vplayingids);
671 }
672}

675/// Find data files in idir and return a cached (id, count) for each dimension.
676/// Returns:
677/// Vector with cumulative maps of name to count for each dimension
678/// in k::id_dimensions.
679vumids
680cache_metadata(const strings& files,
681 const string wfield = "", const string wvalue = "")
682{
683 // Vector with cumulative maps of name to count for each dimension
684 // in k::id_dimensions.
685 vumids cumulative(k::id_dimensions.size());
686
687 // Add to cumulative.
688 for (const string& f: files)
689 {
690 const vsids dimensions = get_vsids_from_metadata(f, wfield, wvalue);
691 for (uint index(0); index < k::id_dimensions.size(); ++index)
692 {
693 const sids ids = dimensions[index];
694 umids& cumuids = cumulative[index];
695
696 for (const id_type& id : ids)
697 {
698 auto [ count, name ] = id;
699
700 auto fiter = cumuids.find(name);
701 if (fiter != cumuids.end())
702 {
703 // Update count to old count plus new count and re-insert.
704 auto [ samename, oldcount ] = *fiter;
705 cumuids.erase(fiter);
706 cumuids.insert(make_pair(name, oldcount + count));
707 }
708 else
709 {
710 // Insert id and count if name doesn't exist...
711 cumuids.insert(make_pair(name, count));
712 }
713 }
714 }
715 }
716 return cumulative;
717}

736/// Writes JSON file with details about aggregate metadata.
737void
738serialize_metadata_aggregate(const vsids& cumulativeranked,
739 const strings& namedkeys, const string aggname)
740{
741 rj::StringBuffer sb;
742 rj::PrettyWriter<rj::StringBuffer> writer(sb);
743 writer.StartObject();
744
745 // Serialize metadata set or subset.
746 if (!namedkeys.empty())
747 {
748 writer.String("name-aggregate");
749 writer.String(aggname);
750
751 writer.String("name-keys");
752 writer.StartArray();
753 for (const string& namekey : namedkeys)
754 {
755 std::cout << namekey << std::endl;
756 writer.String(namekey);
757 }
758 writer.EndArray();
759 }
760
761 // Serialize tag data.
762 uint index(0);
763 for (const string& tag : k::id_dimensions)
764 {
765 const sids& csids = cumulativeranked[index];
766 serialize_id_types_json(writer, csids, tag);
767 ++index;
768 }
769 writer.EndObject();
770
771 string outname = mangle_aggregate(aggname);
772 serialize_buffer(sb, outname + ".json");
773}
