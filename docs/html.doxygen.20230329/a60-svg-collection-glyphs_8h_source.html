33/**
34 Graphic element for torrent details. Aggregated, they form a legend
35 to the given torrent map.
36*/
37struct glyph
38{
39 using colort = svg::color_qi;
40
41 string name;
42 string sname;
43 string infohash;
44 ulong tid;
45 resolution_mode resolution;
46 encoding_mode encoding;
47 colort klr;
48
49 string d1name;
50 string d2name;
51 string d3name;
52 ulong d1; // peers
53 ulong d2; // seeds
54 ulong d3; // downloaded
55
56 // If zero, then unused. If not, then part of a larger set of
57 // glyphs, and this value represents the normalized value.
58 ulong weight;
59
60 ulong bytes;
61 bool is_private;
62 bool is_ssl;
63 bool is_i2p;
64
65 glyph(const string n, string ih, ulong t, resolution_mode res,
66 encoding_mode enc, colort c, string d1, string d2, string d3,
67 ulong pd1, ulong pd2, ulong pd3, ulong w)
68 : name(n), sname(""), infohash(ih), tid(t), resolution(res), encoding(enc),
69 klr(c), d1name(d1), d2name(d2), d3name(d3), d1(pd1), d2(pd2), d3(pd3),
70 weight(w), bytes(0), is_private(false), is_ssl(false), is_i2p(false)
71 { }
72
73 glyph(const string n, colort c, string d1, string d2, string d3,
74 ulong pd1, ulong pd2, ulong pd3)
75 : name(n), sname(""), infohash(""), tid(0),
76 resolution(resolution_mode::none), encoding(encoding_mode::none), klr(c),
77 d1name(d1), d2name(d2), d3name(d3), d1(pd1), d2(pd2), d3(pd3),
78 weight(0), bytes(0), is_private(false), is_ssl(false), is_i2p(false)
79 { }
80
81 glyph(const string n, colort c, string d1, string d2, ulong pd1, ulong pd2)
82 : name(n), sname(""), infohash(""), tid(0),
83 resolution(resolution_mode::none), encoding(encoding_mode::none), klr(c),
84 d1name(d1), d2name(d2), d3name(""), d1(pd1), d2(pd2), d3(0),
85 weight(0), bytes(0), is_private(false), is_ssl(false), is_i2p(false)
86 { }
87
88 glyph()
89 : name(""), sname(""), infohash(""), tid(0),
90 resolution(resolution_mode::none), encoding(encoding_mode::none),
91 klr(svg::color::black),
92 d1name(""), d2name(""), d3name(""), d1(0), d2(0), d3(0),
93 weight(0), bytes(0), is_private(false), is_ssl(false), is_i2p(false)
94 { }
95
96 ulong
97 value(const swarm_mode mode) const
98 {
99 ulong ret(0);
100 if (mode == swarm_mode::peer)
101 ret = d1;
102 if (mode == swarm_mode::seed)
103 ret = d2;
104 return ret;
105 }
106};

112/// Shell for titles page.
113glyphs
114make_glyphs_basic(const collection& cll)
115{
116 using namespace svg;
117 string tdir = cll.torrent_directory();
118
119 // Load up torrent and decode metadata.
120 lt::session_proxy sp;
121 lt::settings_pack settings = make_settings_pack();
122 lt::session s(settings);
123 configure_session(s, settings);
124
125 lt::torrent_handle th;
126 lt::add_torrent_params p;
127 p.save_path = io::get_output_directory().c_str();
128
129 const render_state& tr = get_render_state();
130
131 glyphs glys;
132 for (uint i = 0; i < cll.members.size(); ++i)
133 {
134 const collection::tmarkup& mrkup = cll.members[i];
135
136 // Full filename to trimmed, partial name.
137 string filef = mrkup.name;
138 string filetrim = cll.trim_string(filef, "", cll.multiepisodep);
139
140 glyph g(filef, tr.get_color(filef), "peers", "seeds", 0, 0);
141
142 // Add peer glyph info.
143 g.name = filef;
144 g.sname = filetrim;
145 g.tid = mrkup.tid;
146 g.resolution = mrkup.resolution;
147 g.encoding = mrkup.encoding;
148
149 // If this is a synthetic torrent, strip filename to torrent base.
150 auto fpos = filef.find("multi-btih-");
151 if (fpos != string::npos)
152 {
153 // Strip filef to after "multi-bih-" + .. + "x-" + torrent name.
154 auto fpos2 = filef.find("x-");
155 if (fpos2 != string::npos)
156 filef = filef.substr(fpos2 + 2);
157 }
158
159 // Find the rest from torrent meta data.
160 string fqfilef(io::end_path(tdir) + filef + ".torrent");
161 lt::error_code ec;
162 p.ti = std::make_shared<lt::torrent_info>(fqfilef, std::ref(ec));
163 if (ec)
164 {
165 std::string s("make_glyphs_basic:: error or invalid torrent file");
166 std::cout << s << std::endl << fqfilef << std::endl;
167 std::cout << ec.message() << std::endl;
168 }
169 else
170 {
171 th = s.add_torrent(p, ec);
172 auto& ti = *(th.torrent_file());
173
174 std::ostringstream oss;
175 oss << ti.info_hash();
176 g.infohash = oss.str();
177 g.is_private = ti.priv();
178 g.is_ssl = ti.ssl_cert() != "";
179 g.is_i2p = ti.is_i2p();
180 g.bytes = ti.total_size();
181
182 s.remove_torrent(th);
183 }
184 glys.push_back(g);
185 }
186 return glys;
187}

190/// Expensive. Should this be part of a weighed collection?
191glyphs
192make_glyphs(const swarm_gcache& wcll)
193{
194 glyphs glys = make_glyphs_basic(wcll);
195 for (uint i = 0; i < wcll.members.size(); ++i)
196 {
197 glyph& g = glys[i];
198 g.d1 = wcll.peer_totals[i];
199 g.d2 = wcll.seed_totals[i];
200 }
201 return glys;
202}

205/// Eraser.
206glyphs
207remove_empty_glyphs(glyphs& glysin, swarm_mode m)
208{
209 // Only augment non-zero size glyphs.
210 auto lnonzeropeer = [](const glyph& g) { return g.d1 > 0; };
211 auto lnonzeroseed = [](const glyph& g) { return g.d2 > 0; };
212 auto lnonzero = m == swarm_mode::peer ? lnonzeropeer : lnonzeroseed;
213
214 glyphs glys;
215 copy_if(glysin.begin(), glysin.end(), back_inserter(glys), lnonzero);
216 return glys;
217}

220/// Cache glyphs.
221const glyphs&
222cached_glyphs(swarm_gcache& wcll)
223{
224 static glyphs glys;
225 if (glys.empty())
226 {
227 glyphs g1 = make_glyphs(wcll);
228 glyphs g2 = remove_empty_glyphs(g1, wcll.sort_mode);
229 glys = g2;
230 }
231 return glys;

235/// Sum over vector glyphs, return as new summary glyph.
236glyph
237sum_glyphs(const glyphs& glys, const string name = "all")
238{
239 // Add in summary info for combined torrent collection.
240 glyph sumg = { };
241 sumg.name = name;
242 sumg.klr = svg::color::gray50;
243
244 if (!glys.empty())
245 {
246 for (const glyph& g: glys)
247 {
248 sumg.d1 += g.d1; // peer
249 sumg.d2 += g.d2; // seed
250 sumg.d3 += g.d3; // downloads
251
252 sumg.bytes += g.d1 * g.bytes;
253 }
254 sumg.d1name = glys[0].d1name;
255 sumg.d2name = glys[0].d2name;
256 sumg.d3name = glys[0].d3name;
257 }
258
259 return sumg;
260}

263/// Weigh a vector of glyphs via the number of current peers.
264void
265weigh_glyphs_via_mode(glyphs& glys, const swarm_mode mode,
266 const uint maxh, const uint maxw)
267{
268 auto lglyphascending = [mode](const glyph& g1, const glyph& g2)
269 { return g1.value(mode) > g2.value(mode); };
270 std::sort(glys.begin(), glys.end(), lglyphascending);
271
272 // Remove zero element glyph elements.
273 glyphs glysnu;
274 for (glyph& g: glys)
275 {
276 if (g.value(mode) != 0)
277 glysnu.push_back(g);
278 }
279 glys = glysnu;
280
281 uint maxv = glys.front().value(mode);
282 uint minv = glys.back().value(mode);
283 uint nceil = std::max(maxh, maxw);
284 uint nfloor = 10;
285
286 // Weigh.
287 for (glyph& g: glys)
288 {
289 ulong v = g.value(mode);
290 auto n = normalize_on_range(v, minv, maxv, nfloor, nceil);
291 g.weight = static_cast<ulong>(n);
292 }
293
294 // Sort glyphs in decreasing size, so that biggest layer is lowest.
295 auto lglyphweigh = [](const glyph& g1, const glyph& g2)
296 { return g1.weight > g2.weight; };
297 std::sort(glys.begin(), glys.end(), lglyphweigh);
298}

303/// bndlsize is the subgrouping of names into bundles of 10
304/// max bundles per row is 5 for vertical stacks, 1 for horizontal stacks.
305ulong
306compute_number_of_glyph_rows(const ulong ntorrents, const uint maxbundlesrow)
307{
308 const uint maxperrow = bndlsize * maxbundlesrow;
309 double dtorrents(ntorrents);
310 double d(dtorrents / maxperrow);
311 ulong dceil = std::ceil(d);
312
313 std::cout << "d: " << d << " dceil: " << dceil
314 << " ntorrents: " << ntorrents
315 << " maxperrow: " << maxperrow << std::endl;
316 return dceil;
317}

339/// For ovoid representation.
340void
341augment_weighed_glyph_v2(svg_element& obj, const glyph& g, ulong vmax,
342 bool textp)
343{
344 svg::typography typo = svg::k::apercu_typo;
345
346 // Dead center.
347 auto [ midx, midy ] = obj.center_point();
348 uint radius = g.weight / 4;
349
350 if (textp)
351 {
352 // Find percentage.
353 uint perct = g.d1 * 100 / vmax;
354
355 // Draw circle.
356 svg::style tstyl = svg::k::w_style;
357 tstyl._M_fill_opacity = 0.0;
358 tstyl._M_stroke_opacity = 0.2;
359 tstyl._M_stroke_size = 0.5;
360 point_2d_to_circle(obj, midx, midy, tstyl, radius);
361
362 // Name
363 svg::typography typon = typo;
364 typon._M_anchor = svg::typography::anchor::end;
365 typon._M_align = svg::typography::align::right;
366 sized_text_r(obj, typon, 14, g.name,
367 midx - radius + carto::fspacer, midy - carto::fspacer, 90);
368
369 // Percentage
370 svg::typography typog = typo;
371 typog._M_anchor = svg::typography::anchor::start;
372 typog._M_align = svg::typography::align::left;
373 typog._M_style._M_fill_color = svg::color::gray50;
374 std::ostringstream outs;
375 outs << perct << '%';
376 sized_text_r(obj, typog, 28, outs.str(),
377 midx - radius + carto::fspacer, midy + carto::fspacer, 90);
378 }
379 else
380 {
381 svg::style styl = svg::k::b_style;
382 styl._M_fill_color = g.klr;
383 styl._M_fill_opacity = 1.0;
384
385 styl._M_stroke_color = g.klr;
386 styl._M_stroke_opacity = 0.0;
387 styl._M_stroke_size = 0.20;
388
389 point_2d_to_circle(obj, midx, midy, styl, radius);
390 }

1023/// v3
1024/// by day/week/hour
1025// Compact steep left to right glyphs that staircase up and to the right.
1026// Assume that there is only one row of glyphs, but each column has a
1027// variable size, perhaps pertaining to a partition size.
1028// Uses torrent info instead of name.
1029void
1030augment_torrent_collection_glyphs_v3(svg_element& obj, svg::typography typo,
1031 const glyphs& glys, swarm_mode m,
1032 const int yoff = 2)
1033{
1034 // Offset (delta) and calculated initial position for legend (lx).
1035 const int xdelta = 10;
1036 const int ydelta = 20;
1037 const int xdeltarow = 130;
1038
1039 // Torrent glyphs.
1040 group_element gpre;
1041 gpre.start_element();
1042 obj.add_element(gpre);
1043
1044 // Height range from 10 to 25.
1045 uint maxheight(25);
1046 static std::mt19937_64 rg(std::random_device{}());
1047 auto disti = std::uniform_int_distribution<>(10, maxheight);
1048 uint maxlen = disti(rg);
1049
1050 // Maximum number of staggered groups in one row.
1051 uint groupsn(0);
1052 const uint maxgroupsinrow(13);
1053
1054 int tx = 2 * carto::frame_margin();
1055
1056 // Vertical offset.
1057 int ty = obj._M_area._M_height - (yoff * carto::frame_margin());
1058
1059 int lx = tx;
1060 int ly = ty;
1061
1062 int xcol0 = lx;
1063 for (uint n(0); n < glys.size();)
1064 {
1065 // Walk portrait landscape across horizontal.
1066 for (uint sn(0); sn < maxlen && n < glys.size(); ++sn)
1067 {
1068 const glyph& g = glys[n++];
1069 augment_torrent_glyph_v3(obj, typo, g, lx, ly, m);
1070 lx += xdelta;
1071 ly -= ydelta;
1072 }
1073 ++groupsn;
1074
1075 if (groupsn == maxgroupsinrow)
1076 {
1077 ty -= ((ydelta * maxheight) + (carto::frame_margin()));
1078 ly = ty;
1079 lx = 2 * carto::frame_margin();
1080 xcol0 = lx;
1081 groupsn = 0;
1082 }
1083 else
1084 {
1085 ly = ty;
1086 lx = xcol0 + xdeltarow;
1087 xcol0 = lx;
1088 }
1089
1090 // Margin requires shorter length for the last element in a row.
1091 if (groupsn == maxgroupsinrow - 1)
1092 maxlen = 10;
1093 else
1094 maxlen = disti(rg);
1095 }
1096
1097 group_element gpost;
1098 gpost.finish_element();
1099 obj.add_element(gpost);
1100}
