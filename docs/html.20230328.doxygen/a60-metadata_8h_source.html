49/// Find index for specific value in id_dimensions.
50uint
51find_id_index(const string tag)
52{
53 using citerator = strings::const_iterator;
54 const citerator idend = xdimensions.cend();
55 const citerator fiter = find(xdimensions.cbegin(), idend, tag);
56 if (fiter != idend)
57 return distance(xdimensions.cbegin(), fiter);
58 else
59 {
60 string m("find_id_index:: tag ");
61 m += tag;
62 m += " is not in xdimensions";
63 m += newline;
64 throw std::runtime_error(m);
65 }
66}

69/// Age value constants by decile.
70constexpr std::array<const char*, 13> age_values =
71 {
72 "unknown",
73 "00-09", "10-19", "20-29", "30-39", "40-49",
74 "50-59", "60-69", "70-79", "80-89", "90-99",
75 "100+",
76 "ambiguous"
77 };
78
79/**
80 Age values
81
82 unknown = radius / 3;
83 00-09 = radius + 1 rspace
84 10-19 = radius + 2 rspace
85 ...
86 90-99 = radius + 10 rspace
87 100+ = radius + 11 or 13 rspace
88 ambiguous = radius + 12 or 16 rspace
89
90 Split playing and all compount values before this call,
91 aka 30-30 playing 100+
92**/
93uint
94age_value_to_multiple(const string& value, const bool linearp = false)
95{
96 uint ret(0);
97 switch (value.size())
98 {
99 case 4:
100 {
101 // 100+ -> 13
102 if (linearp)
103 ret = 11;
104 else
105 ret = 13;
106 break;
107 }
108 case 5:
109 {
110 // 00-09 -> 1
111 // 10-19 -> 2
112 // ...
113 // 90-99 -> 10
114 string s = value.substr(0, 1);
115 ret = std::stoi(s);
116 ret++;
117 break;
118 }
119 case 7:
120 // unknown -> 0
121 break;
122 case 9:
123 {
124 // ambiguous -> outermost
125 if (linearp)
126 ret = 12;
127 else
128 ret = 15;
129 break;
130 }
131 default:
132 {
133 string m("age_value_to_multiple:: age value not known: ");
134 m += value;
135 m += k::newline;
136 m += "age size: ";
137 m += std::to_string(value.size());
138 m += k::newline;
139 throw std::logic_error(m);
140 break;
141 }
142 }
143 return ret;
144}

181/// Each idenity_type is an occurrence and an identifier, like (4, male).

184/// Each dimension of a composite identity as a vector of id_types.
185/// aka details for single dimensions so ethnicities { (5,asian), (2,african) }

188/// All dimensions of a composite identity as a vector of vids.

212/// From multiset of strings to ordered elements of id (aka count, identifier).
213sids
214multiset_to_set_of_ids(const msstrings& values)
215{
216 sids ids;
217
218 using mssiter = msstrings::iterator;
219 mssiter iter = values.begin();
220 mssiter iend = values.end();
221 auto lmss = [&ids, &values, &iter] (const std::string& word)
222 {
223 if (!word.empty())
224 {
225 auto p = values.equal_range(word);
226 uint count = static_cast<int>(std::distance(p.first, p.second));
227 ids.insert(make_tuple(count, word));
228 std::advance(iter, count - 1);
229 }
230 };
231
232 std::for_each<mssiter&>(iter, iend, lmss);
233 return ids;
234}

248/// Add id and weight to existing umids type.
249void
250insert_ids_into(umids& um, const id_type& idt)
251{
252 auto [ weight, id ] = idt;
253 if (um.count(id))
254 {
255 // Add to existing value.
256 um[id] += weight;
257 }
258 else
259 um.insert(make_pair(id, weight));
260}

303/// Substitue "->" for "playing" in input string s.
304string
305shorten_playing_value(const string s)
306{
307 string ret;
308 auto pos = s.find(k::playing);
309 if (pos != string::npos)
310 {
311 string lhs = s.substr(0, pos - 1);
312 string rhs = s.substr(pos + k::playing.size() + 1);
313 ret = lhs + k::playingabv + rhs;
314 }
315 return ret;
316}

319/// Combine value/valueplaying sids into one sid and shorten playing in name.
320sids
321strings_to_set_of_ids(const sids& ids, const sids& idsp)
322{
323 // Combine ids and idsp, they don't interesect.
