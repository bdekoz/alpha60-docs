55template <typename Type, typename Func, typename ... Args>
56inline void try_allocate(Type& task, Func&& f, Args&& ... args)
57{
58 try{
59 task = new typename std::remove_pointer<Type>::type(std::bind(std::forward<Func>(f), std::forward<Args>(args)...));
60 } catch (const std::exception& e) {
61 if(task != nullptr)
62 delete task;
63 throw e;
64 }

126// Implementation:
127template<typename Func, typename... Args>
128auto static_pool::enqueue(Func &&f, Args &&... args)
129-> std::future<typename std::result_of<Func(Args...)>::type> {
130 using return_type = typename std::result_of<Func(Args...)>::type;
131 std::packaged_task<return_type()> *task = nullptr;
132 try_allocate(task, std::forward<Func>(f), std::forward<Args>(args)...);
133 auto result = task->get_future();
134 {
135 // Critical region.
136 std::lock_guard<std::mutex> lock(m_shared_src->queue_mu);
137 m_shared_src->queue.push([task]() {
138 (*task)();
139 delete task;
140 });
141 }
142 m_shared_src->to_finish.fetch_add(1);
143 m_shared_src->cv.notify_one();
144 return result;
145}

147static_pool::static_pool(std::size_t sz)
148 : m_shared_src(std::make_shared<pool_src>()), size(sz) {
149 for (int i = 0; i < sz; ++i) {
150 std::thread(
151 [this](std::shared_ptr<pool_src> ptr) {
152 while (true) {
153 std::function<void()> task;
154 // >>> Critical region => Begin
155 {
156 std::unique_lock<std::mutex> lock(ptr->queue_mu);
157 ptr->cv.wait(
158 lock, [&] { return ptr->shutdown or !ptr->queue.empty(); });
159 if (ptr->shutdown and ptr->queue.empty())
160 return; // Conditions to let the thread go.
161 task = std::move(ptr->queue.front());
162 ptr->queue.pop();
163 }
164 // >>> Critical region => End
165 task();
166 if (ptr->to_finish.fetch_add(-1) == 1)
167 ptr->wait_cv.notify_one();
168 }
169 },
170 m_shared_src)
171 .detach();
172 }
173}

175void static_pool::wait() {
176 std::unique_lock<std::mutex> lock(m_shared_src->wait_mu);
177 m_shared_src->wait_cv.wait(
178 lock, [this] { return m_shared_src->to_finish.load() == 0; });
179}

181static_pool::~static_pool() {
182 m_shared_src->shutdown = true;
183 std::atomic_signal_fence(std::memory_order_seq_cst);
184 m_shared_src->cv.notify_all();
185}
