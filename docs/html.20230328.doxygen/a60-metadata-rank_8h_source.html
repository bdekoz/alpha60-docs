22/// POD type for ranked media objects.

35/// Greater sorting for ranked media objects.
36inline bool
37operator>(const ranked_media_object& lhs, const ranked_media_object& rhs)
38{
39 bool ret(false);
40 if (lhs.percent >= rhs.percent)
41 {
42 if (lhs.percent == rhs.percent)
43 {
44 // Level 2.
45 if (lhs.count >= rhs.count)
46 {
47 if (lhs.count == rhs.count)
48 {
49 // Level 3.
50 if (lhs.name > rhs.name)
51 ret = true;
52 }
53 else
54 ret = true;
55 }
56 }
57 else
58 ret = true;
59 }
60 return ret;
61}

64/// Output ranked list in text format.
65void
66serialize_ranked_media_object(const rmediaobjs rankedmos, const string rdimname,
67 const uint sz, const string rmatch = "",
68 const uint topn = 50)
69{
70 string ofprefix("media-objects-" + std::to_string(sz));
71 string ofsuffix("-ranked-by-");
72 ofsuffix += to_replace_string(rdimname);
73 if (!rmatch.empty())
74 {
75 ofsuffix += k::hyphen;
76 string rmatchsafe = to_replace_string(to_replace_string(rmatch), '/');
77 ofsuffix += rmatchsafe;
78 }
79 string ofname(ofprefix + ofsuffix);
80 std::ofstream ofs(ofname + ".txt");
81 uint n(0);
82 for (const auto& rmo: rankedmos)
83 {
84 if (n++ == topn)
85 break;
86
87 auto [ percentage, count, name ] = rmo;
88 if (percentage != 0.0)
89 {
90 ofs << std::fixed << std::setprecision(2) << double(percentage)
91 << k::comma << k::space;
92 }
93 ofs << count << k::comma << k::space
94 << name << std::endl;
95 }
96}

99/// rdim == "ethnicities" etc
100/// rmatch == match specific tag like "black"
101void
102rank_metadata_by_dimension_and_match(const strings& files,
103 const string rdimname,
104 const string rmatch, const uint topn = 50)
105{
106 rmediaobjs rankedmos;
107 for (const string& f : files)
108 {
109 rj::Document dom = deserialize_json_to_dom_object(f);
110 string cname = canonical_media_name(dom);
111
112 // For top-level dimension tag (ethnicities), search for
113 // count of total character entries [1-n] with any values
114 ulong countn = count_values_in_fields_matching(dom, rdimname);
115
116 if (!cname.empty() && countn != 0)
117 {
118 // Count of say "black" matching ethnicities
119 msstrings etags = values_from_fields_matching(dom, rdimname, rmatch);
120 if (!etags.empty())
121 {
122 sids esids = multiset_to_set_of_ids(etags);
123 for (const auto& idt : esids)
124 {
125 auto [ count, id ] = idt;
126 if (id == rmatch)
127 {
128 double pcnt(count);
129 pcnt /= countn;
130 rmediaobj obj = { pcnt, count, cname };
131 rankedmos.insert(obj);
132 break;
133 }
134 }
135 }
136 }
137 }
138 serialize_ranked_media_object(rankedmos, rdimname, files.size(),
139 rmatch, topn);
140}

143/// rdim == "bias-tags" etc
144void
145rank_metadata_by_dimension_count(const strings& files, const string rdimname,
146 const uint topn = 50)
147{
148 rmediaobjs rankedmos;
149 for (const string& f : files)
150 {
151 rj::Document dom = deserialize_json_to_dom_object(f);
152 string cname = canonical_media_name(dom);
153 strings values = values_from_field(dom, rdimname);
154 ulong countn = values.size();
155
156 if (!cname.empty() && countn != 0)
157 {
158 rmediaobj obj = { 0.0, countn, cname };
159 rankedmos.insert(obj);
160 }
161 }
162 serialize_ranked_media_object(rankedmos, rdimname, files.size(), "", topn);
163}
