100/// Info about torrent_info, serialized to json
101string
102serialize_torrent_handle_json(const lt::torrent_handle& th, bool geoip)
103{
104 using namespace libtorrent;
105 using namespace rapidjson;
106 using a60::string;
107
108 StringBuffer sb;
109 jsonstream writer(sb);
110
111 writer.StartObject();
112
113 // Torrent general info.
114 auto tf = th.torrent_file();
115 if (tf == NULL)
116 return string("no torrent metadata");
117 auto& ti = *(th.torrent_file());
118 {
119 writer.String("name");
120 writer.String(ti.name());
121
122 writer.String("private");
123 writer.Bool(ti.priv());
124 writer.String("ssl");
125 writer.Bool(ti.ssl_cert() != "");
126 writer.String("i2p");
127 writer.Bool(ti.is_i2p());
128
129 //writer.String("info hash");
130 //writer.String(ti.info_hash().to_string());
131 }
132
133 // Time, date, and geolocation info.
134 {
135 auto nowtp = std::chrono::system_clock::now();
136 auto nowf = a60::time_point_as_string(nowtp, false);
137 writer.String("date");
138 writer.String(nowf);
139
140 auto nowstmp = a60::time_point_as_string(nowtp, true);
141 writer.String("datestamp");
142 writer.String(nowstmp);
143
144 const string& ip = host_ip_address();
145
146 meta_peer p;
147 if (geoip)
148 {
149#ifdef a60_USE_GEO
150 geo::country_map cmap;
151 cmap.open();
152 geo::city_map ctymap;
153 ctymap.open(GEOIP_INDEX_CACHE);
154 p = make_meta_peer(ip, cmap, ctymap);
155#endif
156 }
157 else
158 p = make_meta_peer(ip);
159
160 serialize_meta_peer_json(writer, p);
161 }
162
163 // Torrent file info.
164 const file_storage& st = ti.files();
165 {
166 double tsize = ti.total_size();
167 tsize /= a60::k::b_in_gb;
168 writer.String("size-total");
169 writer.Double(tsize);
170
171 writer.String("files-total");
172 writer.Int(ti.num_files());
173
174 writer.String("files");
175 writer.StartArray();
176
177 uint idx = 1;
178 for (auto const i : st.file_range())
179 {
180 writer.StartObject();
181 writer.String("number");
182 writer.Int(idx++);
183 writer.String("size");
184 writer.Double(double(st.file_size(i)) / a60::k::b_in_gb);
185 writer.String("path");
186 writer.String(st.file_path(i));
187 writer.EndObject();
188 }
189 writer.EndArray();
190 }
191
192 // Torrent status info.
193 std::vector<announce_entry> ae_vec = ti.trackers();
194 torrent_status ts = th.status();
195 {
196 writer.String("completed");
197 writer.Int(ts.num_complete);
198 writer.String("incomplete");
199 writer.Int(ts.num_incomplete);
200 writer.String("peers");
201 writer.Int(ts.num_peers);
202 writer.String("seeds");
203 writer.Int(ts.num_seeds);
204 writer.String("found_peers");
205 writer.Int(ts.list_peers);
206 writer.String("found_seeds");
207 writer.Int(ts.list_seeds);
208 writer.String("trackers");
209 writer.Int(ae_vec.size());
210 }
211
212 // Connected peer info (seeders are connected and complete).
213 std::vector<lt::peer_info> peers;
214 th.get_peer_info(peers);
215 vmeta_peers metapeers = make_meta_peers(peers);
216 {
217 writer.String("connected peers total");
218 writer.Int(metapeers.size());
219
220 writer.String("connected peer");
221 writer.StartArray();
222 uint countp(1);
223 for (meta_peer& p: metapeers)
224 {
225 writer.StartObject();
226
227 writer.String("number");
228 writer.Int(countp++);
229 serialize_meta_peer_json(writer, p);
230
231 writer.EndObject();
232 }
233 writer.EndArray();
234 }
235
236 // Found or Known peer info (peers).
237 std::vector<string> fullpeers;
238 auto tsp = th.native_handle();
239 tsp->get_full_peer_ip_list(fullpeers);
240 vmeta_peers metafullpeers = make_meta_peers(fullpeers);
241 {
242 writer.String("peers total");
243 writer.Int(metafullpeers.size());
244
245 writer.String(k::peer_s);
246 writer.StartArray();
247 uint countfp(1);
248 for (meta_peer& p: metafullpeers)
249 {
250 writer.StartObject();
251
252 writer.String("number");
253 writer.Int(countfp++);
254 serialize_meta_peer_json(writer, p);
255
256 writer.EndObject();
257 }
258 writer.EndArray();
259 }
260
261 writer.EndObject();
262
263 return sb.GetString();
264}

267/// Deserialize input file.
268rj::Document
269deserialize_json_to_dom(string input_file)
270{
271 // Deserialize input file.
272 std::ifstream ifs(input_file);
273 string json;
274 if (ifs.good())
275 {
276 std::ostringstream oss;
277 oss << ifs.rdbuf();
278 json = oss.str();
279 }
280 else
281 {
282 std::cerr << "error: cannot open input file " << input_file << std::endl;
283 }
284
285 // Validate json file.
286
287 // DOM
288 rj::Document dom;
289 dom.Parse(json.c_str());
290 if (dom.HasParseError())
291 {
292 std::cerr << "error: cannot parse input file " << input_file << std::endl;
293 std::cerr << rj::GetParseError_En(dom.GetParseError()) << std::endl;
294 std::cerr << dom.GetErrorOffset() << std::endl;
295 }
296
297 return dom;
298}

330/// Search DOM for meta_peer(s).
331vmeta_peers
332search_dom_for_meta_peers(const rj::Document& dom, string search_key)
333{
334 std::vector<meta_peer> peers;
335
336 if (!dom.HasParseError() && dom.HasMember(search_key.c_str()))
337 {
338 const rj::Value& a = dom[search_key.c_str()];
339 if (a.IsArray())
340 {
341 for (uint i = 0; i < a.Size(); ++i)
342 {
343 const rj::Value& v = a[i];
344 if (!v.IsObject())
345 {
346 std::cerr << "error: expected v to be an object" << std::endl;
347 }
348
349 string addr = v["ip"].GetString();
350 string country = v["country"].GetString();
351 double lt = v["latitude"].GetDouble();
352 double lng = v["longitude"].GetDouble();
353 string region = v["region"].GetString();
354 string city = v["city"].GetString();
355
356 meta_peer p(make_tuple(addr, country, lt, lng, region, city));
357 peers.push_back(p);
358 }
359 }
360 }
361 return peers;
362}

397/// Search files for field value, return sum.
398ulong
399sum_int_field_in_node_across_files(const string idir, const string field)
400{
401 const strings& files = io::populate_files(idir, ".json");
402
403 // Cumulative summary JSON schema changed at the beginning of 2021.
404 const string node1("collection-unique-btiha-total");
405 const string node2("collection-unique-btiha");
406 ulong count(0);
407 for (const string& file: files)
408 {
409 rj::Document dom = deserialize_json_to_dom(file);
410 if (dom.HasParseError())
411 return count;
412
413 rj::Value node;
414 if (dom.HasMember(node1.c_str()))
415 node = dom[node1.c_str()];
416 if (dom.HasMember(node2.c_str()))
417 node = dom[node2.c_str()];
418
419 if (node.HasMember(field.c_str()))
420 {
421 const rj::Value& a = node[field.c_str()];
422 if (a.IsInt())
423 count += a.GetInt();
424 }
425 }
426
427 return count;
428}

431/// Search files for field value, return std::set of tuple<value, name>
432snstring
433get_int_field_in_node_across_files(const string idir, const string field)
434{
435 const strings& files = io::populate_files(idir, ".json");
436
437 // Cumulative summary JSON schema changed at the beginning of 2021.
438 const string cname("collection-name");
439 const string node1("collection-unique-btiha-total");
440 const string node2("collection-unique-btiha");
441
442 snstring ret;
443 for (const string& file: files)
444 {
445 rj::Document dom = deserialize_json_to_dom(file);
446 if (dom.HasParseError())
447 continue;
448
449 rj::Value node;
450 if (dom.HasMember(node1.c_str()))
451 node = dom[node1.c_str()];
452 if (dom.HasMember(node2.c_str()))
453 node = dom[node2.c_str()];
454
455 ulong count(0);
456 if (node.HasMember(field.c_str()))
457 {
458 const rj::Value& a = node[field.c_str()];
459 if (a.IsInt())
460 count = a.GetInt();
461 }
462
463 string name;
464 if (dom.HasMember(cname.c_str()))
465 {
466 string prename;
467 const rj::Value& a = dom[cname.c_str()];
468 if (a.IsString())
469 prename = a.GetString();
470
471 const rj::Value& a2 = dom["collection-id"];
472 if (a2.IsString())
473 {
474 string modif = a2.GetString();
475 if (!modif.empty())
476 {
477 prename += k::space;
478 prename += modif;
479 }
480 }
481 name = prename;
482 }
483 ret.insert(make_tuple(count, name));
484 }
485
486 return ret;
487}

490/// Search DOM for a range of integer values.
491string
492sample_range_for_field(const strings& dfiles, string match)
493{
494 // Find sample duration min/max size.
495 std::set<int> srange;
496 for (const string& s: dfiles)
497 {
498 rj::Document dom = deserialize_json_to_dom(s);
499 int subsz = search_dom_for_int_field(dom, match);
500 srange.insert(subsz);
501 }
502
503 string subsz = std::to_string(*(srange.begin()));
504 if (srange.size() > 1)
505 subsz += (" to " + std::to_string(*(--srange.end())));
506
507 return subsz;
508}

511/// Search DOM for type and duration metadata.
512string
513sample_and_subsample_type_and_size_metadata(const strings& dfiles,
514 bool subsp = true)
515{
516 // Find sample type.
517 rj::Document dom = deserialize_json_to_dom(dfiles.front());
518 string samplet = search_dom_for_string_field(dom, "sample-type");
519
520 // Find sample min/max duration.
521 string samplesz = sample_range_for_field(dfiles, "sample-size");
522
523 // Find subsample min/max duration.
524 string subsamplesz;
525 int subsize = search_dom_for_int_field(dom, "sub-sample-size");
526 if (subsp && subsize > 0)
527 {
528 subsamplesz = sample_range_for_field(dfiles, "sub-sample-size");
529 subsamplesz += " samples over ";
530 }
531
532 string substr(subsamplesz + samplesz + " " + samplet + "(s)");
533
534 if (samplesz.find(" to ") != string::npos)
535 substr += ", varies per torrent";
536
537 return substr;
538}

542string
543sample_universe_size_metadata(const strings& dfiles, swarm_mode m)
544{
545 // Switch by mode.

586/**
587 ISO-3166 mapping from country codes to country names.
588
589 Bi-directional mapping from the Alpha-3 country code to the
590 Fully-qualified country name.
591
592 Argument true = alpha3, country
593 Argument false = country, alpha3
594
595 For ISO-3 codes, see
596 https://en.wikipedia.org/wiki/ISO_3166-1_alpha-3/
597
598 For ISO-3 to country name mapping resources, see
599 https://github.com/lukes/ISO-3166-Countries-with-Regional-Codes
600*/
601void
602insert_to_iso_3166(umstrings& umap, const string s1, const string s2,
603 const bool code2name)
604{
605 if (code2name)
606 umap.insert(make_pair(s1, s2));
607 else
608 umap.insert(make_pair(s2, s1));
609}

612/// ISO-3 to Country Name.
613umstrings
614geo::deserialize_json_to_iso_3166_3_map(const bool code2name,
615 const bool lowercase)
616{
617 auto& rtr = io::get_run_time_resources();
618 string ddir = io::end_path(rtr.data);
619 ddir += "iso-3166/";
620 string jfile = "slim-3.json";
621 string file(ddir + jfile);
622 io::sanity_check_input(file);
623
624 rj::Document dom = deserialize_json_to_dom(ddir + jfile);
625 std::cerr << "iso-3661 alpha-3 to country file: " << file << std::endl;
626
627 std::cerr << "is object: " << dom.IsObject() << std::endl;
628 std::cerr << "is array: " << dom.IsArray() << std::endl;
629
630 umstrings umap;
631 if (dom.IsArray())
632 {
633 for (uint i = 0; i < dom.Size(); ++i)
634 {
635 const rj::Value& v = dom[i];
636 if (!v.IsObject())
637 {
638 std::cerr << "error: expected v to be an object" << std::endl;
639 }
640
641 string country = v["name"].GetString();
642 string alpha3 = v["alpha-3"].GetString();
643
644 if (lowercase)
645 {
646 string lc = to_lowercase_string(country);
647 country = lc;
648 }
649 insert_to_iso_3166(umap, alpha3, country, code2name);
650 }
651 }
652
653 // Extras values.
654 insert_to_iso_3166(umap, "XXX",
655 lowercase ? "unknown" : "Unknown", code2name);
656 insert_to_iso_3166(umap, "XXA",
657 lowercase ? "stateless" : "Stateless", code2name);
