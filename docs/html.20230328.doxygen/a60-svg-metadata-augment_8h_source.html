27/// Base plate color, identity graphics, svg file.
28svg_element
29augment_metadata_background(const string name, const area<> a,
30 const color klr = color::white,
31 const bool inversep = false,
32 const bool logop = false)
33{
34 // Make SVG file.
35 svg_element obj(name, a);
36 const auto [width, height] = a;
37
38 // Put down klr as background.
39 style styl = get_metadata_style(inversep);
40 styl._M_fill_color = klr;
41 point_2d_to_rect(obj, 0, 0, styl, width, height);
42
43 // LOGO
44 // Add in MMRL logo
45 if (logop)
46 {
47 auto& rtr = io::get_run_time_resources();
48
49 const size_type iw = 108;
50 const size_type ih = 20;
51 area<>::atype ihoffset = height - k::metamargin - 48;
52 // center on width - 200, but centered so 100..
53 area<>::atype iwoffset = width - 100;
54
55 const string logofile1 = inversep
56 ? "identity/mmrl-logo-dots-inverse.svg" : "identity/mmrl-logo-dots.svg";
57 string ifile1(io::end_path(rtr.data) + logofile1);
58
59 image_element::data di1 = { ifile1, iwoffset - iw/2 , ihoffset, iw, iw };
60 image_element i1;
61 i1.start_element();
62 i1.add_data(di1);
63 i1.finish_element();
64 obj.add_element(i1);
65
66 // 54 mm x 20 mm high
67 const string logofile2 = "identity/mmrl-logo-type-paths.svg";
68 string ifile2(io::end_path(rtr.data) + logofile2);
69 ihoffset -= ih;
70 image_element::data di2 = { ifile2, iwoffset - iw/2, ihoffset, 108, 40 };
71 image_element i2;
72 i2.start_element();
73 i2.add_data(di2);
74 i2.finish_element();
75 obj.add_element(i2);
76
77 // 427.1 W x 276 H pixels
78 const string logofile3 = "identity/mmrl-logo-pink-octo-flying-small.svg";
79 string ifile3(io::end_path(rtr.data) + logofile3);
80 ihoffset -= 1.35 * ih;
81 auto iwoffsetmod = iwoffset - 126 / 2;
82 image_element::data di2b = { ifile3, iwoffsetmod, ihoffset, 126, 81 };
83 image_element i2b;
84 i2b.start_element();
85 i2b.add_data(di2b);
86 i2b.finish_element();
87 obj.add_element(i2b);
88
89 area<>::atype dy2 = height - k::metamargin - 40;
90 image_element::data di2c = { ifile2, k::metamargin, dy2, 108, 40 };
91 image_element i2c;
92 i2c.start_element();
93 i2c.add_data(di2c);
94 i2c.finish_element();
95 obj.add_element(i2c);

113/// Base info for per-media object metadata visualizations.
114svg_element
115augment_metadata_base_text(const string f, const string modifier,
116 const area<> a, const color klr = color::white,
117 const bool inversep = false)
118{
119 // Type constants
120 // Text augmentation: name, tags, etc.
121 typography typo = get_metadata_typo();
122 typography typom = get_metadata_typo(14, true);
123 typography typomg(typom);
124 typomg._M_style._M_fill_color = color::wcag_gray;
125
126 // Make SVG file.
127 rj::Document dom = deserialize_json_to_dom_object(f);
128 const string namef = canonical_media_name(dom);
129 svg_element obj = augment_metadata_background(namef + "-" + modifier,
130 a, klr, inversep);
131
132 // Type sizes for title, subtitle, data.
133 const uint nsz = 18;
134 const uint mtsz = 6;
135 const uint vsz = 7; // was mtsz
136
137 // Starting positions....
138 // lower left bottom row 3x3.
139 const auto delta = cell_size_3x3_matrix(a, k::xdimensions.size(), k::metamargin);
140 const point_2t cp = to_point_in_3x3_matrix(a, 8, k::metamargin);
141 const auto [ cpx, cpy ] = cp;
142 uint yoff = cpy - delta / 2;
143 uint xoff = cpx - delta / 2;
144
145 // Tag types.
146
147 // IMDB / wiki
148 sized_text(obj, typomg, mtsz, "imdb", xoff, yoff);
149 yoff += vsz;
150 string idimdb = search_dom_for_string_field(dom, "id-imdb");
151 sized_text(obj, typo, vsz, idimdb, xoff, yoff);
152 yoff += (vsz * 2);
153
154 sized_text(obj, typomg, mtsz, "wikipedia", xoff, yoff);
155 yoff += vsz;
156 string idw = search_dom_for_string_field(dom, "id-wikipedia");
157 sized_text(obj, typo, vsz, idw, xoff, yoff);
158 yoff += (vsz * 2);
159
160 // TAGS (genre, plot, distribution)
161 const uint ntsz = nsz / 2;
162 const auto deltalen = delta / (ntsz / 2);
163 const strings meta_tags { "genre-tags", "plot-tags", "distribution-tags" };
164 for (const string& tag : meta_tags)
165 {
166 // Remove redundant "-tags" from label.
167 const string stag = tag.substr(0, tag.size() - 5);
168 sized_text(obj, typomg, mtsz, stag, xoff, yoff);
169 yoff += vsz;
170
171 strings tagt = values_from_field(dom, tag);
172 std::sort(tagt.begin(), tagt.end());
173 string tagtflat = strings_to_cssv_string(tagt);
174 point_2t p { xoff, yoff };
175 yoff = text_line_n(obj, p, tagtflat, typo, vsz, deltalen);
176 yoff += (vsz * 2);
177 }
178 yoff += (vsz * 2);
179
180 // Name variant/modifier.
181 uint yoff2 = a._M_height - k::metamargin;
182 yoff2 -= nsz;
183
184 // Perhaps "media object" subtitle....
185 if (false)
186 {
187 yoff2 -= vsz;
188 point_2t pmod { xoff, yoff2 };
189 yoff2 = text_line_n(obj, pmod, modifier, typomg, mtsz, delta);
190 yoff2 += nsz - mtsz;
191 }
192
193 // Name Title
194 const string moname = search_dom_for_string_field(dom, "name");
195 const string season = search_dom_for_string_field(dom, "id");
196 const string namefull(moname + k::space + season);
197 sized_text(obj, typom, nsz, namefull, xoff, yoff2);
198
199 return obj;
200}

203/// Base info for per-media object metadata visualizations, rotated.

206 const area<> a, const color klr = color::white,
207 const bool inversep = false)

213 typography typomg(typom);

215 typomg._M_w = typography::weight::normal;
216 typomg._M_p = typography::property::italic;

236 uint xoff = cpx + delta / 2;
237

249 xoff -= nsz;
250 xoff -= xdelta;
251
252 // Tag types.
253 const uint ntsz = nsz / 2;
254 const auto deltalen = delta / (ntsz / 2);
255
256 // IMDB / wiki
257 sized_text_r(obj, typomg, mtsz, "imdb", xoff, yoff, -90);
258 xoff -= xdelta;
259 string idimdb = search_dom_for_string_field(dom, "id-imdb");
260 point_2t p1 { xoff, yoff };
261 xoff = text_line_n_r(obj, p1, idimdb, typo, vsz, deltalen, 3);
262 xoff -= (xdelta * 2);
263
264 sized_text_r(obj, typomg, mtsz, "wikipedia", xoff, yoff, -90);
265 xoff -= xdelta;
266 string idw = search_dom_for_string_field(dom, "id-wikipedia");
267 point_2t p2 { xoff, yoff };
268 xoff = text_line_n_r(obj, p2, idw, typo, vsz, deltalen, 3);
269 xoff -= (xdelta * 2);
270
271 // TAGS (genre, plot, distribution)
272 const strings meta_tags { "genre-tags", "plot-tags", "distribution-tags" };
273 for (const string& tag : meta_tags)
274 {
275 // Remove redundant "-tags" from label.
276 const string stag = tag.substr(0, tag.size() - 5);
277 sized_text_r(obj, typomg, mtsz, stag, xoff, yoff, -90);
278 xoff -= xdelta;
279
280 strings tagt = values_from_field(dom, tag);
281 std::sort(tagt.begin(), tagt.end());
282 string tagtflat = strings_to_cssv_string(tagt);
283 point_2t p { xoff, yoff };
284 xoff = text_line_n_r(obj, p, tagtflat, typo, vsz, deltalen, 3);
285 xoff -= (xdelta * 2);
286 }
287
288 // Paint edges.
289 paint_edges_with_char_index(obj, a, namef.front());
290
291 return obj;
292}

295/// Base for metadata visualizations, individual and media object only.
296uint
297augment_metadata_individual(svg_element& obj, const string f,
298 const string castid)
299{
300 // TAGS cast-lead-n-(id-tags)
301 constexpr uint vsz = 14;
302 constexpr uint mtsz = 16;
303 constexpr uint off = 140;
304
305 const typography typo = get_metadata_typo();
306 const typography typom = get_metadata_typo(12, true);
307
308 const auto [width, height] = obj._M_area;
309
310 typography typomv(typom);
311 typomv._M_style._M_stroke_color = color::gray40;
312 typomv._M_style._M_stroke_size = 0.5;
313 typomv._M_style._M_stroke_opacity = 0.0;
314 typomv._M_style._M_fill_color = color::gray40;
315 typomv._M_style._M_fill_opacity = 1.0;
316
317 constexpr uint csz(72);
318 uint startx = width - k::metamargin - csz;
319 sized_text_r(obj, typomv, csz, castid, startx, k::metamargin, 90);
320
321 rj::Document dom = deserialize_json_to_dom_object(f);
322
323 startx -= 35;
324
325 // Extract name.
326 strings ndimensions(k::id_dimensions);
327 ndimensions.push_back("names");
328 for (const string& tag : ndimensions)
329 {
330 const string ctag(castid + svg::k::hyphen + tag);
331 strings ctagt = values_from_field(dom, ctag);
332 if (!ctagt.empty())
333 {
334 sized_text_r(obj, typom, mtsz, tag, startx, k::metamargin, 90);
335 string ctagtflat = strings_to_cssv_string(ctagt);
336 sized_text_r(obj, typo, vsz, ctagtflat, startx, off, 90);
337 startx -= (mtsz * 1.25);
338 }
339 }
340 return startx;
341}

344/// Entry point for per-character-leads, each on separate page.
345void
346render_cast_leads(const string f, const area<> a)
347{
348 // Go through cast 1 - n (currently 6) x dimensions, pick out values, render.
349 /*
350 "cast-lead-1-ethnicities": ["white"],
351 "cast-lead-1-nationalities": ["usa", "esp", "deu"],
352 "cast-lead-1-genders": ["cis", "male"],
353 "cast-lead-1-sexualities": ["straight"],
354 "cast-lead-1-ages": ["31-40"],
355 "cast-lead-1-z": ["human"],
356 */
357 rj::Document dom = deserialize_json_to_dom_object(f);
358 const string jprefix("cast-lead-");
359 for (uint i = 1; i < 9; ++i)
360 {
361 // Look for identity fields, build up in-memory representation.
362 // castcache = vector[id_tag.size()] of set<string> values.
363 const string prefix(jprefix + to_string(i));
364 vsstrings castcache;
365 ulong count(0);
366 for (const string& idtag : k::id_dimensions)
367 {
368 const string dfield(prefix + svg::k::hyphen + idtag);
369 strings foundv = values_from_field(dom, dfield);
370 std::set<string> s(foundv.begin(), foundv.end());
371 castcache.push_back(s);
372 count += foundv.size();
373 }
374
375 if (count > 0)
376 {
377 svg_element obj = augment_metadata_base_text(f, prefix, a);
378 render_individual(obj, castcache);
379 augment_metadata_individual(obj, f, prefix);
380 }
381 }
382}

385/// Entry point for per-character-leads, composited on single page.
386/// Assume f has non-zero entries.
387void
388render_cast_leads_grid(const string f, const area<> a,
389 const area<> ableed = svg::k::no_area,
390 const svg::k::select slxt = svg::k::select::none)
391{
392 // Set up svg frame for all inserted svg elements...
393 const string oname("cast-leads-composite");
394 rj::Document dom = deserialize_json_to_dom_object(f);
395 const string namef = canonical_media_name(dom);
396
397 const string prefix(namef + k::hyphen + oname);
398 const bool bleedp = ableed._M_width != 0 && ableed._M_height != 0;
399 svg_element obj = bleedp ? svg_element(prefix, a, false)
400 : augment_metadata_background(prefix, a, color::wcag_lgray);
401 if (bleedp)
402 composite_bleed_areas(obj, slxt, 0.125, 0.5);
403
404 // Given a 3x3 grid, want each inserted svg to be one cell in that grid.
405 const space_type delta = cell_size_3x3_matrix(a, k::xdimensions.size(),
406 k::metamargin);
407 const area<>::atype deltah = delta / 2;
408 const area<> inarea = { delta, delta };
409
410 // Go through cast 1 - n (currently 6) x dimensions, pick out values, render.
411 /*
412 "cast-lead-1-ethnicities": ["white"],
413 "cast-lead-1-nationalities": ["usa", "esp", "deu"],
414 "cast-lead-1-genders": ["cis", "male"],
415 "cast-lead-1-sexualities": ["straight"],
416 "cast-lead-1-ages": ["31-40"],
417 "cast-lead-1-z": ["human"],
418 */
419 const string jprefix("cast-lead-");
420 for (uint i = 1; i < 9; ++i)
421 {
422 // Look for identity fields, build up in-memory representation.
423 // castcache = vector[id_tag.size()] of set<string> values.
424 const string prefix(jprefix + to_string(i));
425 vsstrings castcache;
426 ulong count(0);
427 for (const string& idtag : k::id_dimensions)
428 {
429 const string dfield(prefix + svg::k::hyphen + idtag);
430 strings foundv = values_from_field(dom, dfield);
431 std::set<string> s(foundv.begin(), foundv.end());
432 castcache.push_back(s);
433 count += foundv.size();
434 }
435
436 if (count > 0)
437 {
438 // Insertion point on the enclosing svg...
439 auto p = to_point_in_3x3_matrix(a, i, k::metamargin,
440 matrix_variant::centerweightsvg);
441 auto [ px, py ] = p;
442
443 // Adjust insertion point to center nested svg..
444 px -= deltah;
445 py -= deltah;
446 const point_2t pneu = { px, py };
447
448 // Create nested svg_element.
449 svg_element obj_nested(prefix, inarea, false);
450 obj_nested.start_element(pneu, inarea);
451 render_individual(obj_nested, castcache);
452 obj_nested.finish_element();
453
454 obj.add_element(obj_nested);
455 }
456 }
457
458 // Finish up placement of area on larger bleed area, if be.
459 if (bleedp)
460 {
461 obj.finish_element();
462 svg_element objb = augment_metadata_background(prefix, ableed,
463 color::wcag_lgray);
464 objb.add_element(obj);
465 }
466}

501/**
502 Entry point for output, per media object
503 Input @dimensions, @dimensionsp are vsids.
504 Input @a is final size, @ableed is area with bleed for to-edge images.
505 Input @slxt is where the @a is placed on @ableed.
506*/
507void
508render_vsids(const string f,
509 const vsids& dimensions, const vsids& dimensionsp,
510 const vul& counts, const int r, const area<> a,
511 const area<> ableed = svg::k::no_area,
512 const svg::k::select slxt = svg::k::select::none)
513{
514 // Sanity check, if no count in identity 3t, then skip.
515 auto anyatall = std::accumulate(counts.begin(), counts.end(), ulong(0));
516 const bool bleedp = ableed._M_width != 0 && ableed._M_height != 0;
517 if (anyatall != 0)
518 {
519 const string prefix("media-object");
520
521 // If bleed, return nested object.
522 svg_element obj = bleedp ? svg_element(prefix, a, false)
523 : augment_metadata_base_text_r(f, prefix, a);
524 if (bleedp)
525 composite_bleed_areas(obj, slxt, 0.125, 0.5);
526
527 for (uint idx = 0; idx < k::id_dimensions.size(); idx++)
528 {
529 const string tag = k::id_dimensions[idx];
530 const uint tagidx = k::find_id_index(tag);
531 point_2t origin = to_point_in_matrix(a, tagidx, k::metamargin);
532
533 const auto& ids = dimensions[idx];
534 const auto& idsp = dimensionsp[idx];
535 const ulong count = counts[idx];
536
537 // Switch rendering based on id_tag specified.
538 if (tag == "ages")
539 {
540 const auto ar = 14;
541 const auto aspace = std::min(8, ar / 2);
542 render_ids_ages(obj, tag, count, ids, idsp, origin, ar, aspace);
543 }
544 else if (tag == "nationalities")
545 {
546 auto dmin = 2; // 10, r / 3
547 const int rspace = r / 5; // -> 12
548 render_id_base(obj, tag, count, origin, r, rspace, 0, false);
549 render_ids_text_y(obj, count, origin, ids, idsp, dmin);
550 }
551 else
552 {
553 // Combine ids and idsp, they don't interesect.
554 sids idsflat = strings_to_set_of_ids(ids, idsp);
555 if (!idsflat.empty())
556 {
557 constexpr bool showuniquevalue(true);
558 if (showuniquevalue)
559 render_ids_uvalue(obj, tag, count, idsflat, origin, r, r/5);
560 else
561 render_ids_value(obj, tag, count, idsflat, origin, r);
562 }
563 }
564 }
565
566 // Biases
567 // Reception.
568 rj::Document dom = deserialize_json_to_dom(f);
569 if (dom.HasParseError())
570 throw std::runtime_error("render_vsids:: error parsing file: " + f);
571
572 // Show biases.
573 strings btags = values_from_field(dom, "bias-tags");
574 sids bias_ids;
575 for (const auto& s: btags)
576 bias_ids.insert(make_tuple(1, s));
577 const uint bidx = k::find_id_index("bias");
578 point_2t bp = to_point_in_matrix(obj._M_area, bidx, k::metamargin);
579 render_ids_text_bias(obj, bp, bias_ids, r);
580
581 // Show reception tags.
582 strings rpostags = values_from_field(dom, "reception-positive");
583 strings rnegtags = values_from_field(dom, "reception-negative");
584
585 sids rpos_ids;
586 for (const auto& s: rpostags)
587 rpos_ids.insert(make_tuple(1, s));
588 sids rneg_ids;
589 for (const auto& s: rnegtags)
590 rneg_ids.insert(make_tuple(1, s));
591 const uint ridx = k::find_id_index("reception");
592 point_2t rp = to_point_in_matrix(obj._M_area, ridx, k::metamargin);
593 render_ids_text_reception(obj, rp, rpos_ids, rneg_ids, r);
594
595 // Finish up placement of area on larger bleed area, if be.
596 if (bleedp)
597 {
598 obj.finish_element();
599 svg_element objb = augment_metadata_base_text_r(f, prefix, ableed);
600
601 // Paint edges.
602 const string namef = canonical_media_name(dom);
603 paint_edges_with_char_index(objb, ableed, namef.front());
604
605 objb.add_element(obj);
606 }
607 }
608}

611/// Entry point for identity tag output, per aggregate
612/// Input @dimensions are vsids.
613svg_element
614render_vsids_aggregate(const string aggname,
615 const vsids& dimensions, const vsids& dimensionsp,
616 const vul& counts, const int r, const area<> a)
617{
618 const string name("dimension-graph-" + aggname + "-aggregate");
619 svg_element obj = augment_metadata_background(name, a);
620
621 const int rspace = r / 5; // -> 12
622
623 // Set label spaces to (maxvalue / 10) + 1
624 const auto spaceso = svg::get_label_spaces();
625 const auto maxcountiter = std::max_element(counts.begin(), counts.end());
626 double maxcount = *maxcountiter;
627 svg::set_label_spaces(svg::significant_digits_in(maxcount));
628
629 // Name

659 //render_ids_value(obj, tag, count, ids, origin, r);
660 render_ids_uvalue(obj, tag, count, ids, origin, r, r/5);

734/// Common parts.
735void
736analyze_metadata_aggregate_common(const area<> a, const strings& files,
737 const string aggname, const int radius,
738 const string wfield, const string wvalue)
739{
740 vsids dimensions;
741 vsids dimensionsp;
742 get_vsids_from_metadata(files, dimensions, dimensionsp, wfield, wvalue);
743 vul counts = get_vul_from_metadata(files, wfield, wvalue);
744
745 // Verbose.
746 const bool verbosep = true;
747 if (verbosep)
748 {
749 std::cout << "found " << files.size()
750 << " files for aggregate: " << aggname << std::endl;
751 if (!wfield.empty() && !wvalue.empty())
752 {
753 std::cout << "with match field/value of: " << wfield << "/" << wvalue;
754 std::cout << std::endl;
755 }
756 for (const string& s : files)
757 std::cout << s << std::endl;
758
759 std::cout << std::endl;
760 ulong stotal = std::accumulate(counts.begin(), counts.end(), ulong(0));
761 std::cout << "accumulated: " << stotal << " across "
762 << k::id_dimensions.size() << " identity dimension(s)"
763 << std::endl;
764 }
765
766 // Render glyph elements and titles.
767 const string outname = mangle_metadata_aggregate(aggname, wfield, wvalue);
768 svg_element obj = render_vsids_aggregate(outname, dimensions, dimensionsp,
769 counts, radius, a);
770
771 // Convert files to vector of name-keys or name-key-canonical.
772 strings namekeys;
773 msstrings biases;
774 msstrings recpos;
775 msstrings recneg;
776 for (const string& f : files)
777 {
778 rj::Document dom = deserialize_json_to_dom(f);
779 if (dom.HasParseError())
780 throw std::runtime_error("analyze_metadata_aggregate_common: error");
781
782 // Find canonical name-key fields.
783 // Things like "Star Wars Last Jedi" is a subset, so look for
784 // the canonical version first.
785 string namek = canonical_media_name(dom);
786 namekeys.push_back(namek);
787
788 // Find bias-tag entries.
789 strings btags = values_from_field(dom, "bias-tags");
790 biases.insert(btags.begin(), btags.end());
791
792 // Find reception-positive entries.
793 strings rpostags = values_from_field(dom, "reception-positive");
794 recpos.insert(rpostags.begin(), rpostags.end());
795
796 // Find reception-negative entries.
797 strings rnegtags = values_from_field(dom, "reception-negative");
798 recneg.insert(rnegtags.begin(), rnegtags.end());
799 }
800
801 // Show biases.
802 sids bias_ids = multiset_to_set_of_ids(biases);
803 const uint bidx = k::find_id_index("bias");
804 point_2t bp = to_point_in_matrix(obj._M_area, bidx, k::metamargin);
805 render_ids_text_bias(obj, bp, bias_ids, radius);
806
807 // Show reception tags.
808 sids rpos_ids = multiset_to_set_of_ids(recpos);
809 sids rneg_ids = multiset_to_set_of_ids(recneg);
810 const uint ridx = k::find_id_index("reception");
811 point_2t rp = to_point_in_matrix(obj._M_area, ridx, k::metamargin);
812 render_ids_text_reception(obj, rp, rpos_ids, rneg_ids, radius);
813
814 // Show "short names" aka name-keys for all media objects in aggregate.
815 const auto titleoffx = k::metamargin + (36 * 2);
816 render_metadata_aggregate_namekeys(obj, namekeys, titleoffx, 200);
817}

828 vsids dimensionsp;
829 get_vsids_from_metadata(files, dimensions, dimensionsp, wfield, wvalue);

837 << " files for aggregate: " << aggname << std::endl;

845

847 ulong stotal = std::accumulate(counts.begin(), counts.end(), ulong(0));
848 std::cout << "accumulated: " << stotal << " across "
849 << k::id_dimensions.size() << " identity dimension(s)"
850 << std::endl;
851 }
852
853 // Convert files to vector of name-keys or name-key-canonical.
854 strings namekeys;
855 msstrings biases;
856 msstrings recpos;
857 msstrings recneg;
858 for (const string& f : files)
859 {
860 rj::Document dom = deserialize_json_to_dom(f);
861 if (dom.HasParseError())
862 throw std::runtime_error("analyze_metadata_aggregate_common: error");
863
864 // Find bias-tag entries.
865 strings btags = values_from_field(dom, "bias-tags");
866 biases.insert(btags.begin(), btags.end());
867
868 // Find reception-positive entries.
869 strings rpostags = values_from_field(dom, "reception-positive");
870 recpos.insert(rpostags.begin(), rpostags.end());
871
872 // Find reception-negative entries.
873 strings rnegtags = values_from_field(dom, "reception-negative");
874 recneg.insert(rnegtags.begin(), rnegtags.end());
875 }
876
877 // Render glyph elements and titles.
878 const int rspace = radius / 5; // -> 12
879 const string outbase = mangle_metadata_aggregate(aggname, wfield, wvalue);
880
881 for (const string& s : k::xdimensions)
882 {
883 const uint i = k::find_id_index(s);
884 const string oname = mangle_metadata_aggregate(outbase, to_string(i), s);
885 const string name("dimension-graph-" + oname);
886
887 // Make SVG file.
888 svg_element obj(name, a);
889
890 // Put down klr as background.
891 style styl = get_metadata_style();
892 styl._M_fill_color = color::gray20;
893 point_2d_to_rect(obj, 0, 0, styl, a._M_width, a._M_height);
894
895 const auto cp = obj.center_point();
896 if (i < dimensions.size())
897 {
898 const auto& ids = dimensions[i];
899 const auto& idsp = dimensionsp[i];
900 const ulong count = counts[i];
901
902 // Combine ids and idsp, they don't interesect.
903 sids idsflat = strings_to_set_of_ids(ids, idsp);
904
905 switch (i)
906 {
907 case 0:
908 {
909 // age
910 render_ids_ages(obj, s, count, ids, idsp, cp, radius, rspace);
911 break;
912 }
913 case 1:
914 {
915 // genders
916 }
917 case 2:
918 {
919 // sexualities
920 }
921 case 3:
922 {
923 // ethnicities
924 if (!idsflat.empty())
925 render_ids_uvalue(obj, s, count, idsflat, cp, radius, rspace);
926 break;
927 }
928 case 4:
929 {
930 // nationalities
931 render_id_base(obj, s, count, cp, radius, rspace, 0, false);
932 render_ids_text_x(obj, count, cp, ids, idsp, radius, rspace);
933 break;
934 }
935 case 5:
936 {
937 // z
938 if (!idsflat.empty())
939 render_ids_uvalue(obj, s, count, idsflat, cp, radius, rspace);
940 break;
941 }
942 }
943 }
944 else
945 {
946 switch (i)
947 {
948 case 6:
949 {
950 // bias
951 sids bias_ids = multiset_to_set_of_ids(biases);
952 render_ids_text_bias(obj, cp, bias_ids, radius);
953 break;
954 }
955 case 7:
956 {
957 // reception
958 sids rpos_ids = multiset_to_set_of_ids(recpos);
959 sids rneg_ids = multiset_to_set_of_ids(recneg);
960 render_ids_text_reception(obj, cp, rpos_ids, rneg_ids, radius);
961 break;
962 }
963 default:
964 {
965 string m("analyze_metadata_by_dimension:: error out of range");
966 throw std::out_of_range(m);
967 }
968 }
969 }
970 }
971}
