25/**
26 Color x collection
27 Return a semi-unique color based on mode characteristics.
28 And by semi-unique, we mean: up to 40ish per region in spectrum palette.
29*/
30svg::color_qi
31color_by_mode_finite(resolution_mode rm, encoding_mode em [[gnu::unused]])
32{
33 using color = svg::color;
34 using color_qi = svg::color_qi;
35
36 // Entry points into svg color coding table, by encoding and resolution.
37 static color c4k = color::orange;
38 static color c1080 = color::purple;
39 static color c720 = color::blue;
40 static color csd = color::green; // 264, 265, xvid
41
42 // Add colors for each of these torrent files based on this order:
43 // 1080p, 720p, x264, sd, 264|265|xvid, remaining
44 color_qi retk(color::gray50);
45 if (rm != resolution_mode::rsd)
46 {
47 if (rm == resolution_mode::r4k)
48 {
49 retk = c4k;
50 c4k = next_color(c4k);
51 }
52 if (rm == resolution_mode::r1080p)
53 {
54 retk = c1080;
55 c1080 = next_color(c1080);
56 }
57 if (rm == resolution_mode::r720p)
58 {
59 retk = c720;
60 c720 = next_color(c720);
61 }
62 }
63 else
64 {
65 retk = csd;
66 csd = next_color(csd);
67 }
68 return retk;
69}

72/**
73 Return a semi-unique color based on mode characteristics.
74 And by semi-unique, we mean: up to 40ish per region in 112 palette.
75*/
76svg::color_qi
77color_by_mode_and_colorband(resolution_mode rm, encoding_mode em [[gnu::unused]])
78{
79 // Add colors for each of these torrent files based on this order:
80 // 1080p, 720p, x264, xvid, remaining
81 svg::color_qi retk(svg::color::gray50);
82 if (rm != resolution_mode::rsd)
83 {
84 if (rm == resolution_mode::r4k)
85 retk = next_in_color_band(svg::cband_o);
86 if (rm == resolution_mode::r1080p)
87 retk = next_in_color_band(svg::cband_p);
88 if (rm == resolution_mode::r720p)
89 retk = next_in_color_band(svg::cband_b);
90 }
91 else
92 retk = next_in_color_band(svg::cband_g);
93
94 return retk;
95}

98// Generated/infinite color set, derived from existing named colors.
99void
100initialize_colors(collection& cll, svg::render_state& rs,
101 const bool finitep)
102{
103 // Switch coloring mode function.
104 auto fklrfinite = &color_by_mode_finite;
105 auto fklrbnd = &color_by_mode_and_colorband;
106 auto fklr = finitep ? fklrfinite : fklrbnd;
107
108 for (const auto& mrkup : cll.members)
109 {
110 auto [ name, tid, resolution, encoding ] = mrkup;
111 auto klr = (*fklr)(resolution, encoding);
112 rs.klrs.insert(make_pair(name, klr));
113 }
114 rs.color_generated = true;
115}
