24/**
25 Frame
26
27 Output size, display size, ie bounded area into which the
28 projection, legend, title, and other symbology are drawn to
29 create the map.
30*/
31struct frame
32{
33 using atype = space_type; // area<>::atype;
34
35 /// Map size in pixels.
36 atype width;
37 atype height;
38
39 /// Projection coordinates in frame, at origin (0,0).
40 double moriginx;
41 double moriginy;
42
43 frame(const atype w, const atype h,
44 const double x = 0.0, const double y = 0.0)
45 : width(w), height(h), moriginx(x), moriginy(y) { }
46
47 /// Frame with explicit projection coordinates given.
48 frame(const frame& f, const double x, const double y)
49 : width(f.width), height(f.height), moriginx(x), moriginy(y) { }
50
51 frame(const frame&) = default;
52
53 void
54 center(const projection_base& p)
55 {
56 moriginx = (width - p.width) / 2;
57 moriginy = ((height - p.height) / 2);
58 }
59
60 frame
61 bottom_tile_frame() const
62 {
63 frame ret(*this);
64 ret.moriginy -= ret.height;
65 return ret;
66 }
67};

216/**
217 Margin is required space from edge of the frame for a given output device.
218 For reductions (say 50%), one needs 2x the space....
219 const int fmargin = 50; // viz-65
220 const int fmargin = 75; // Createspace try v13 -
221 const int fmargin = 100; // Createspace try v67 +
222 */
223constexpr int
224frame_margin()
225{
226 if constexpr (dframe_scale == frame_scale::print)
227 return 120; // Createspace try v73 +, at 96dp 1.25"
228 if constexpr (dframe_scale == frame_scale::video)
229 return 60;
230}
