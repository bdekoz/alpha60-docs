235int main(int argc, char* argv[])
236{
237 using namespace a60::media_objects;
238 using namespace a60::metadata;
239
240 // Sanity check.
241 if (argc > 2)
242 {
243 fputs(usage().c_str(), stderr);
244
245 int i = argc;
246 while (i - 1 > 0)
247 {
248 --i;
249 std::cout << i + 1 << a60::k::tab << argv[i] << std::endl;
250 }
251 }
252
253 // Dump text logs of various data queries.
254 constexpr bool analysisp(true);
255 if (analysisp)
256 initial_metadata_characteristics();
257
258 // Canvas/page size.
259 // const area<> csz = svg::k::letter_096_h;
260 // const area<> csz = svg::k::a4_096_h;
261 // const area<> csz = svg::k::a4_096_h;
262
263 // const area<> csz = svg::k::instagram_v;
264 // const area<> csz = svg::k::instagram_s;
265 // const area<> csz = svg::k::instagram_4ks;
266
267 // const area<> csz = svg::k::square17b_096_v;
268 //const area<> csz = svg::k::square8bis_096;
269 const area<> csz = svg::k::square8_096;
270 const area<> cszb = svg::k::square8bisb_096_v;
271
272 // Initialize render details, including page sizes, circle radius,
273 // and spacing.
274 const int radius = 16; // 50 if engc
275 svg::set_label_spaces(12); // For 6 point, space 6 min, all but pose 18
276 svg::set_min_ring_size(2); // For 6 point
277
278 // 6pt type / 8 vmax / 5.5 is (15.3 deg)
279 // 6pt type / 8 vmax / 5.2 is (14.8 deg)
280 // 6pt type / 8 vmax / 4.8 is (14.2 deg)
281 // 6pt type / 8 vmax / 4.2 is (13.3 deg)
282 // 6pt type / 8 vmax / 3.6 is (12.3 deg)
283 // 6pt type / 8 vmax / 3.2 is (11.5 deg) // No overlap but not even
284 // 6pt type / 8 vmax / 3.0 is (11.32 deg) // No overlap but not even
285 svg::set_min_satellite_distance(2.85);
286 init_id_render_state_cache();
287
288 if (argc == 1)
289 analyze_metadata_variations(radius, csz, cszb);
290 else
291 {
292 std::string file(argv[1]);
293 analyze_metadata_media_object(file, radius, csz, cszb);
294 }
295
296 return 0;
297}
