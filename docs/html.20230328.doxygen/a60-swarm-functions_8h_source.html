62/// Accumulate peers all files in all subdirectories of dir matching dirm.
63/// Where all files matching filem with additional match are selected.
64/// Assumed filem is partial filename to look for in filesystem.
65/// If match is empty, then all files.
66/// If match is not empty, then only files matching.
67swarm_accumulator
68usunique_peers(string dir, string dirm, string filem, string match,
69 string peertype)
70{
71 namespace rj = rapidjson;
72
73 // Get all the peer info.
74 swarm_accumulator npeers;
75
76 // Minimal error checking.
77 strings subdirs = io::populate_sub_directories(dir, dirm);
78 if (subdirs.empty())
79 {
80 string m("usunique_peers:: ");
81 m += "no subdirectories found in ";
82 m += dir;
83 m += " matching: ";
84 m += dirm;
85 throw std::runtime_error(m);
86 }
87
88 for (string adir: subdirs)
89 {
90 /*
91 May need to separate out suffix trees, depending on the file match.
92
93 GOT_Season_07_DUB
94 GOT_Season_07_DUB_720
95 GOT_Season_07_DUB_1080
96
97 Where searching for the first gets all three variants.
98 To work around this, affix the file name here to name-YEAR, ie
99 using knowledge of the serialzed filename:
100
101 NAME-DATESTAMP-at-0h-0m.json
102 */
103 string filemhy(filem + ("-" + find_unique_year(adir)));
104 strings pfiles = io::populate_files(adir, ".json", filemhy);
105 strings files;
106 if (pfiles.empty())
107 {
108 string m("usunique_peers:: ");
109 m += "no json file(s) found in ";
110 m += adir;
111 m += " with match plus year of: ";
112 m += filemhy;
113 std::cout << m << std::endl;
114 continue;
115 }
116 else
117 {
118 // Second edit, match.
119 if (match.empty())
120 files = pfiles;
121 else
122 {
123 // Look on each filename for match. Filename only, not path.
124 for (string& pf: pfiles)
125 {
126 // Strip off path component.
127 string f;
128 auto pospath = pf.rfind('/');
129 if (pospath != string::npos)
130 f = pf.substr(pospath);
131 else
132 f = pf;
133
134 if (f.find(match) != string::npos)
135 files.push_back(pf);
136 }
137 }
138 std::cout << filemhy << " usunique_peers found in directory: "
139 << adir << std::endl;
140
141 std::cout << pfiles.size() << " files found" << std::endl;
142 if (!match.empty())
143 {
144 std::cout << files.size() << " files matching " << match
145 << std::endl;
146 }
147 std::cout << std::endl;
148 }
149
150
151 for (std::string file : files)
152 {
153 std::cout << file << std::endl;
154
155 // Do it.
156 rj::Document dom = deserialize_json_to_dom_object(file);
157 vmeta_peers cpeers = search_dom_for_meta_peers(dom, peertype);
158 npeers.peers.insert(cpeers.begin(), cpeers.end());
159
160 // Accumulate.
161 npeers.samplen += 1;
162 npeers.samplepeersn += cpeers.size();
163 }
164
165 uint fsize = static_cast<uint>(files.size());
166 npeers.maxfsize = std::max(npeers.maxfsize,fsize);
167 npeers.minfsize = std::min(npeers.minfsize,fsize);
168 }
169 return npeers;
170}

184/// Union of all peers from swarmA and swarmB. (a + b)
185swarm_setaside
186swarm_union(const swarm_setaside swarma, const swarm_setaside swarmb)
187{
188 swarm_setaside ret;
189
190 std::set_union(swarma.peers.begin(), swarma.peers.end(),
191 swarmb.peers.begin(), swarmb.peers.end(),
192 std::inserter(ret.peers, ret.peers.begin()));
193
194 std::set_union(swarma.seeds.begin(), swarma.seeds.end(),
195 swarmb.seeds.begin(), swarmb.seeds.end(),
196 std::inserter(ret.seeds, ret.seeds.begin()));
197
198 return ret;
199}

220/**
221 Look through cache directory for matching files.
222 Exclude persistent cache files.
223
224 dir is the cache directory
225 durationm is the duration match, ie day, week
226 filem is the match
227*/
228swarm_setaside
229sunique_peers(string dir, string durationm, string filem, uint ddigits)
230{
231 const string dompeert = "peers";
232 const string domseedt = "seeds";
233
234 namespace rj = rapidjson;
235
236 // All the data files in dir matching durationm.
237 strings tfiles = io::populate_files(dir, ".json", filem, durationm);
238
239 // Second edit, match (filem) and remove "persistent".
240 strings files;
241 for (std::string f: tfiles)
242 {
243 bool persistp = f.find(k::persist_s) != string::npos;
244 if (!persistp)
245 files.push_back(f);
246 }
247
248 std::cout << "sunique_peers " << durationm << " " << dir << std::endl;
249 std::cout << "sunique_peers " << filem
250 << " files: " << files.size() << std::endl;
251
252 // Iff matching files, find peers, seeds.
253 swarm_setaside swarmc { };
254 if (!files.empty())
255 {
256 for (std::string file: files)
257 {
258 // Do it.
259 rj::Document dom = deserialize_json_to_dom_object(file);
260
261 vmeta_peers cpeers = search_dom_for_meta_peers(dom, dompeert);
262 swarmc.peers.insert(cpeers.begin(), cpeers.end());
263
264 vmeta_peers cseeds = search_dom_for_meta_peers(dom, domseedt);
265 swarmc.seeds.insert(cseeds.begin(), cseeds.end());
266 }
267
268 // Given the duration in (durationm) and tid/id match in (filem),
269 // find the duration count. As in:
270 // durationm is 2017-06-01 and filem is house-of-cards-05-t001-day
271 // the duration count will be the next part of the string, ie
272 // -0001 from remaining -0001-YYYY-MM-DD.json
273 // -00001 from remaining -00001-YYYY-MM-DD.json
274 std::string file = files.front();
275 auto pos = file.find(filem);
276 if (pos != std::string::npos)
277 {
278 file = file.substr(pos + filem.size() + 1, ddigits);
279 swarmc.dcount = std::stoul(file);
280 }
281 else
282 swarmc.dcount = 0;
283
284 swarmc.dtype = durationm;
285 }
286
287 std::cout << "sunique_peers " << swarmc.dcount << std::endl;
288 std::cout << std::endl;
289
290 return swarmc;
291}
