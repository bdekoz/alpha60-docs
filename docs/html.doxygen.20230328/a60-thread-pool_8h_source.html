31template<typename Task>
32class base_thread_pool
33{
34 protected:
35 /**
36 * Wraps tasks in a executor function and wraps the promise which will
37 * receive the return value of the function.
38 *
39 * @param f the function to call when executing the task
40 * @param args the arguments to pass to the function
41 *
42 * @return the future used to wait on the task and get the result
43 */
44 template<typename Fn, typename Ret, typename... Args>
45 std::pair<std::function<void()>,std::future<Ret>>
46 package(Fn f, Args... args){
47 std::promise<Ret> *p = new std::promise<Ret>;
48
49 // Create a function to package as a task.
50 auto task_wrapper = std::bind([p, f{std::move(f)}](Args... args){
51 if constexpr (std::is_same<Ret,void>::value) {
52 f(std::move(args)...);
53 p->set_value();
54 } else {
55 p->set_value(std::move(f(std::move(args)...)));
56 }
57 }, std::move(args)...);
58
59 // Create a function to package as a future for the user to wait on.
60 auto ret_wrapper = [p]() -> Ret{
61 if constexpr (std::is_same<Ret,void>::value) {
62 p->get_future().get();
63 delete p;
64 } else {
65 auto temp = std::move(p->get_future().get());
66 delete p;
67 return std::move(temp);
68 }
69 };
70 return make_pair(task_wrapper,
71 std::async(std::launch::deferred, ret_wrapper));
72 }

91 /**
92 * Manages thread execution. This is the function that threads actually run.
93 * It pulls a task out of the queue and executes it.
94 */
95 void
96 thread_func()
97 {
98 // Can't call get_task until parent class is constructed.
99 init_mutex.lock();
100 init_mutex.unlock();
101 for (;;)
102 {
103 auto task = get_task();
104
105 // If there's nothing to do and we're not ready to join, just
106 // yield.
107 if (!task && !join)
108 {
109 std::this_thread::yield();
110 continue;
111 }
112 // If there's tasks waiting, do one.
113 else if (task)
114 {
115 handle_task(std::move(*task));
116 }
117 // If there's no tasks and we're ready to join, then exit the
118 // function (effectively joining).
119 else if (join)
120 {
121 return;
122 }
123 }
124 }

126 /**
127 * Creates threads for the thread pool.
128 */
129 void
130 init_threads()
131 {
132 task_mutex.lock();
133 for (uint i = 0;i < num_threads;i++)
134 {
135 auto f = std::bind(&base_thread_pool::thread_func, this);
136 threads.push_back(std::move(std::thread(f)));
137 }
138 task_mutex.unlock();
139 }
140
141 /**
142 * Waits for threads to exit. Leaves thread pool in unusable state. Used by
143 * destructors.
144 */
145 void
146 wait()
147 {
148 task_mutex.lock();
149 join = true;
150 task_mutex.unlock();
151 while (threads.size() > 0)
152 {
153 auto &t = threads.back();
154 t.join();
155 threads.pop_back();
156 }
157 }

159 /**
160 * Returns the next task, if there is one. None if there isn't.
161 */
162 virtual std::optional<Task> get_task() = 0;
163
164 /**
165 * Executes a task.
166 */
167 virtual void handle_task(Task) = 0;

169 /// Must be unlocked in child constructors.
170 std::mutex init_mutex;
171 std::mutex task_mutex;
172 private:
173 bool join = false;
174 unsigned int num_threads;
175 std::list<std::thread> threads;
176};

178class thread_pool
179: public base_thread_pool<std::future<void>>
180{
181 public:
182 thread_pool(unsigned int);
183 virtual ~thread_pool();

185 /**
186 * Pushes a new task to the queue.
187 *
188 * @param f the function to call when executing the task
189 * @param args the arguments to pass to the function
190 *
191 * @return the future used to wait on the task and get the result
192 */
193 template<typename Fn, typename... Args>
194 auto async(Fn f, Args... args)
195 {
196 auto p = package<Fn, decltype(f(args...)), Args...>(f, args...);
197 return std::move(add_task(std::move(p)));
198 }
199
200protected:
201 virtual std::optional<std::future<void>> get_task() override;
202 virtual void handle_task(std::future<void>) override;
203
204 auto add_task(auto p)
205 {
206 auto t = std::async(std::launch::deferred, p.first);
207 task_mutex.lock();
208 tasks.emplace(std::move(t));
209 task_mutex.unlock();
210 return std::move(p.second);
211 }
212 private:
213 std::queue<std::future<void>> tasks;
214};
