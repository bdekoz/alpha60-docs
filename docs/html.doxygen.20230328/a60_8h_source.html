126space_type
127distance_point_2t(const point_2t& a, const point_2t& b)
128{
129 auto [ax, ay] = a;
130 auto [bx, by] = b;
131 space_type x = ax - bx;
132 space_type y = ay - by;
133 double dist(sqrt(pow(x, 2) + pow(y, 2)));
134 return dist;
135}

224/// Convert MaxMind GeoIP information from ISO-8859-1 encoding to UTF-8.
225string
226convert_8859_to_utf8(string in)
227{
228 using namespace std;
229
230 using result = codecvt_base::result;
231 using int_type = char;
232 using ext_type = char;
233 using state_type = __gnu_cxx::encoding_state;
234 using ccutf8_codecvt = codecvt<int_type, ext_type, state_type>;
235
236 string out;
237 const int_type* internal_ptr;
238 ext_type* external_ptr;
239 ext_type* buf = new int_type[in.size() * 3];
240
241 // Use iconv on linux, ie `iconv --list`
242
243 // Internal encoding is ISO 8859 and external is UTF-8.
244 state_type state("ISO-8859-16", "UTF-8", 0, 0);
245
246 locale loc(locale::classic(), new ccutf8_codecvt);
247 const ccutf8_codecvt& cvt = use_facet<ccutf8_codecvt>(loc);
248 result r = cvt.out(state, in.c_str(), in.c_str() + in.size(), internal_ptr,
249 buf, buf + in.size() * 3, external_ptr);
250 if (r == codecvt_base::ok)
251 out = string(buf, external_ptr);
252 else
253 {
254 if (r == codecvt_base::partial)
255 {
256 ext_type* n_ptr = external_ptr;
257 while (n_ptr > external_ptr)
258 {
259 external_ptr = n_ptr;
260 r = cvt.in(state, in.c_str(), in.c_str() + in.size(),
261 internal_ptr, buf, buf + in.size() * 3, n_ptr);
262 }
263
264 if (r == codecvt_base::ok)
265 out = string(buf, n_ptr);
266 else
267 {
268 std::cerr << "convert_to_utf8: partial at "
269 << ulong(n_ptr - in.c_str()) << " "
270 << in << std::endl;
271 out = in;
272 }
273 }
274 if (r == codecvt_base::error)
275 {
276 std::cerr << "convert_to_utf8: error " << in << std::endl;
277 out = in;
278 }
279 }
280
281 // Any escape characters may have to be wrapped for HTML with &#[0][0][0];
282
283 return out;
284}

292/// Encapsulated timer with reporting.
293struct timer
294{
295 using clock_type = std::chrono::high_resolution_clock;
296 using time_point = clock_type::time_point;
297 using minutes = std::chrono::minutes;
298 using seconds = std::chrono::seconds;
299 using milliseconds = std::chrono::milliseconds;
300
301 time_point begin;
302 time_point end;
303
304 timer() = default;
305 ~timer() = default;
306
307 void
308 start() { begin = clock_type::now(); }
309
310 void
311 stop() { end = clock_type::now(); }
312
313 string
314 report()
315 {
316 minutes em = std::chrono::duration_cast<minutes>(end - begin);
317 seconds es = std::chrono::duration_cast<seconds>(end - begin);
318 milliseconds ems = std::chrono::duration_cast<milliseconds>(end - begin);
319
320 std::ostringstream ostr;
321 if (em.count() > 0)
322 ostr << em.count() << " minutes";
323 else
324 {
325 if (es.count() > 0)
326 ostr << es.count() << " seconds";
327 else
328 ostr << ems.count() << " milliseconds";
329 }
330 ostr << std::endl;
331 return ostr.str();
332 }
333};

354/// ISO date time stamp as default.
355string
356time_point_as_string(const std::chrono::system_clock::time_point& tp,
357 bool stampp = true)
358{
359 using namespace std::chrono;
360 std::time_t tp_c = system_clock::to_time_t(tp);
361 std::ostringstream odates;
362
363 if (stampp)
364 odates << std::put_time(std::localtime(&tp_c), "%Y-%m-%d-at-%H-%M");
365 else
366 odates << std::ctime(&tp_c);
367 return odates.str();
368}
