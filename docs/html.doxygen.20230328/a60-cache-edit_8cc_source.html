37// For sub-collections like "blade-runner-2049"
38// (A subset of the eoy-curiosities collection)
39// This will remove non-blade-runner files from the cache, such that
40// by-day and by-week, by-hour will work with existing csv scripts.
41
42// Coalesce json files for sub-collections,
43// aka Blade Runner subset from EOY 2017-2018
44// Generate the unique files, and then move those matching sub-collection to
45// separate directory. Then run the 'convert-swarm-json-to-csv.sh' script
46// on the separate directory.
47void
48edit_cache_files_for_episode(collection& cll, const std::string cdir,
49 string& odir, const string dmatch = "")
50{
51 // Get files in cache directory.
52 strings cfiles = io::populate_files(cdir, ".json", dmatch);
53
54 if (cfiles.empty())
55 return;
56 std::cout << cll.slice_name_mangled() << std::endl;
57 std::cout << cfiles.size() << " cache files in " << cdir << std::endl;
58
59 // Output directory for matching.
60 string matchdir("matching-" + cll.matchepisode + "/");
61 const string odirmatch(io::end_path(odir) + matchdir);
62 io::sanity_check_output_directory(odirmatch);
63
64 for (const string& f : cfiles)
65 {
66 rj::Document dom = deserialize_json_to_dom_object(f);
67 string name = search_dom_for_string_field(dom, "torrent-name");
68 if (name.find(cll.matchepisode) != string::npos)
69 {
70 string fname = io::prune_path(f);
71 std::cout << "match: " << fname << std::endl;
72 std::ifstream ifs(f);
73 std::ofstream ofs(odirmatch + fname, std::ios_base::out);
74 ofs << ifs.rdbuf();
75 }
76 }
77}
