59/// Strip off path component from full-qualified filename.
60string
61prune_path(const string& p, bool prunepathsep = false)
62{
63 string f;
64 auto pospath = p.rfind('/');
65 if (pospath != string::npos)
66 {
67 if (prunepathsep)
68 f = p.substr(pospath + 1);
69 else
70 f = p.substr(pospath);
71 }
72 else
73 f = p;
74 return f;
75}

208/// Takes a string representing a directory, returns a sorted list of
209/// contents as a std::vector<string>.
210strings
211populate_files(string dir, string ext = "", string cllm = "",
212 string filem = "")
213{
214 strings ret;
215 path target_path(dir);
216 if (exists(target_path) && is_directory(target_path))
217 {
218 directory_iterator iter(target_path);
219 directory_iterator end_iter;
220 for (; iter != end_iter ; ++iter)
221 {
222 if (is_regular_file(iter->status()))
223 {
224 path p = iter->path();
225 string ipath = p.string();
226 if (ext.empty() && cllm.empty() && filem.empty())
227 ret.push_back(ipath);
228 else
229 {
230 string ifilename = p.filename().string();
231 auto pext = ifilename.find(ext);
232 auto pcllm = ifilename.find(cllm);
233 auto pfilem = ifilename.find(filem);
234 bool extp = ext.empty() || pext != string::npos;
235 bool cllp = cllm.empty() || pcllm != string::npos;
236 bool filep = filem.empty() || pfilem != string::npos;
237 if (extp && filep && cllp)
238 ret.push_back(ipath);
239 }
240 }
241 }
242 }
243 std::sort(ret.begin(), ret.end());
244 return ret;
245}

277/// Takes a string representing a directory, returns a list of contents
278/// as a std::vector<string>.
279strings
280populate_sub_directories(string dir, string dirm = "",
281 bool recursep = true)
282{
283 strings ret;
284 path target_path(dir);
285 if (exists(target_path) && is_directory(target_path))
286 {
287 directory_iterator iter(target_path);
288 for (; iter != directory_iterator(); ++iter)
289 {
290 // Find first level of sub directories.
291 if (is_directory(iter->status()))
292 {
293 string idir = iter->path().string();
294 if (dirm.empty())
295 ret.push_back(idir);
296 else
297 {
298 auto pos = idir.find(dirm);
299 if (pos != string::npos)
300 ret.push_back(idir);
301 }
302
303 if (recursep)
304 {
305 strings rets = populate_sub_directories(idir, dirm);
306 copy(rets.begin(), rets.end(), back_inserter(ret));
307 }
308 }
309 }
310 }
311 else
312 {
313 std::ostringstream ostr;
314 ostr << "populate_sub_directories:: " << dir << " not found";
315 ostr << std::endl;
316 throw std::runtime_error(ostr.str());
317 }
318
319 std::sort(ret.begin(), ret.end());
320 return ret;
321}

324void
325append_to_files(string dir, string inclusion, const string ext)
326{
327 strings subdirs = populate_sub_directories(dir);
328 for (string dir : subdirs)
329 {
330 strings files = populate_files(dir, ext);
331 for (string file : files)
332 {
333 auto mode = std::ios_base::app | std::ios_base::out;
334 std::ofstream ofs(file, mode);
335 ofs << std::endl << inclusion << std::endl;
336 }
337 }
338}
