28/// Resolution bitmask type
29enum class resolution_mode : unsigned char
30 {
31 none = 1u << 0,
32
33 r720p = 1u << 1,
34 r1080p = 1u << 2,
35 r4k = 1u << 3,
36 r8k = 1u << 4,
37
38 rhd = r720p | r1080p | r4k | r8k,
39 rsd = 1u << 6,
40
41 all = 1u << 7
42 };
43
44/// Conversion of resolution_mode to string.
45inline const std::string
46to_string(const resolution_mode e)
47{
48 using map_type = std::map<resolution_mode, std::string>;
49
50 static map_type enum_map;
51 if (enum_map.empty())
52 {
53 enum_map[resolution_mode::all] = "all";
54 enum_map[resolution_mode::rhd] = "hd";
55 enum_map[resolution_mode::rsd] = "sd";
56 enum_map[resolution_mode::r8k] = "4320";
57 enum_map[resolution_mode::r4k] = "2160";
58 enum_map[resolution_mode::r1080p] = "1080";
59 enum_map[resolution_mode::r720p] = "720";
60 enum_map[resolution_mode::none] = "none";
61 }
62 return enum_map[e];
63}

275 find_data_dates() const
276 {
277 strings sdirs = find_data_dates_times();
278
279 if (archivelayoutdeprecatedp)

281 // From compressed ISO datestamps, without separators between year-day.
282 for (string& s: sdirs)
283 {
284 s.insert(4, "-");
285 s.insert(7, "-");
286 s = s.substr(0, 10);
287 }
288 }
289 else
290 {
291 // Remove "-at-" part for editing to datestamp only.
292 const string timestart("-at-");
293 for (string& s: sdirs)
294 s = s.substr(0, s.find(timestart));
295 }
296
297 // Sort, make unique.
298 sort(sdirs.begin(), sdirs.end());
299 sdirs.erase(std::unique(sdirs.begin(), sdirs.end()), sdirs.end());
300 return sdirs;

467 static string
468 trim_string(string in, string match, bool multiepisodep = false)
469 {
470 if (!in.empty() && !match.empty())
471 {
472 string lcin = to_lowercase_string(in);
473 string lcmatch = to_lowercase_string(match);
474
475 // Simplify file name match and remove extraneous cruft
476 // previously needed to insure unique search results.
477 // NB: Be careful about "[rutor.is]westworld_s02" and "s02"
478 auto pos = lcin.find(lcmatch);
479 if (pos != string::npos)
480 {
481 auto sizereq = pos + lcmatch.size();
482 if (multiepisodep || sizereq + 3 > lcin.size())
483 in = in.substr(sizereq);
484 else
485 in = in.substr(sizereq + 3);
486 }
487 }
488
489 const auto c = in.front();
490 if (c == '.' || c == ' ' || c == '/' || c == '-')
491 in = in.substr(1);
492
493 return in;
494 }

715 /// Adjust internal member data (members) to operate on a subset of
716 /// the total collection determined by @rmode and
717 /// @emode. Sub-partition is active after call exits.
718 void
719 partition_by_mode(resolution_mode rmode, encoding_mode emode)
720 {
721 // By default, all modes are active. If this is the requested
722 // change, do nothing.
723 bool ta = rmode == resolution_mode::all && emode == encoding_mode::all;
724 if (!ta)
725 {
726 tmarkups tmpmembers;
727 for (tmarkup& tm: members)
728 {
729 bool tr = tm.resolution == rmode || rmode == resolution_mode::all;
730 bool te = tm.encoding == emode || emode == encoding_mode::all;
731 if (tr && te)
732 tmpmembers.push_back(tm);
733 }
734
735 resolution = rmode;
736 encoding = emode;
737
738 members = tmpmembers;
739 }
740 }

742 /// Adjust internal member data (members) to operate on a subset of
743 /// the total collection based on the duplicate list provided by
744 /// @duplicates. Sub-partition is active after call exits.
745 /// Returns the non-matching names.
746 tmarkups
747 partition_by_names(cached_strings& duplicates)
748 {
749 // By default, all modes are active. If this is the requested
750 // change, do nothing.
751 tmarkups membersno;
752 if (!duplicates.empty())

754 tmarkups membersyes;

760 else
761 membersno.push_back(tm);

764 members = membersyes;
765 }

1124 /// Sort the accumulated elements so that the torrents with the
1125 /// largest number of (peers, seeds) is first, and the least number
1126 /// of (peers, seeds) is last. Assume peers and seeds are sorted in a
1127 /// similar manner previous to this point, based on the current
1128 /// ordering of the collection's data member (members).
1129 void
1130 sort_by_swarm_mode_and_size(swarm_mode mode)
1131 {
1132 /*
1133 Order the index first. It starts as tm.size() with an ad-hoc schema.
1134 First, express as a tuple of <rank, size>.
1135 */
1136 using ulong_tuple = std::tuple<uint, ulong>;
1137 using index_type = std::vector<ulong_tuple>;
1138
1139 const bool peerp = mode == swarm_mode::peer;
1140 vul& sortedby = peerp ? peer_totals : seed_totals;
1141
1142 // Rank selected swarm type by size.
1143 index_type indx;
1144 for (uint i = 0; i < members.size(); ++i)
1145 {
1146 ulong psz = sortedby[i];
1147 ulong_tuple ix(i, psz);
1148 indx.push_back(ix);
1149 }
1150
1151 // Sort index by size.
1152 auto lindexgt = [](const ulong_tuple& p1, const ulong_tuple& p2)
1153 { return std::get<1>(p1) > std::get<1>(p2); };
1154 std::sort(indx.begin(), indx.end(), lindexgt);
1155
1156 // Use the index to re-sort both other data items.
1157 vswgeo_peers speer_wkeys;
1158 vul speer_totals;
1159 vumgeo_peers speer_geos;
1160 vumgeo_ranks speer_weights;
1161
1162 vswgeo_peers sseed_wkeys;
1163 vul sseed_totals;
1164 vumgeo_peers sseed_geos;
1165 vumgeo_ranks sseed_weights;
1166
1167 collection::tmarkups sortedmembers;
1168 for (ulong_tuple& ix: indx)
1169 {
1170 uint oldrank = std::get<0>(ix);
1171 collection::tmarkup& t = members[oldrank];
1172 sortedmembers.push_back(t);
1173
1174 auto& p1 = peer_wkeys[oldrank];
1175 speer_wkeys.push_back(p1);
1176 auto& p2 = peer_totals[oldrank];
1177 speer_totals.push_back(p2);
1178 auto& p3 = peer_geos[oldrank];
1179 speer_geos.push_back(p3);
1180 auto& p4 = peer_weights[oldrank];
1181 speer_weights.push_back(p4);
1182
1183 auto& s1 = seed_wkeys[oldrank];
1184 sseed_wkeys.push_back(s1);
1185 auto& s2 = seed_totals[oldrank];
1186 sseed_totals.push_back(s2);
1187 auto& s3 = seed_geos[oldrank];
1188 sseed_geos.push_back(s3);
1189 auto& s4 = seed_weights[oldrank];
1190 sseed_weights.push_back(s4);
1191 }
1192
1193 members = sortedmembers;
1194
1195 peer_wkeys = speer_wkeys;
1196 peer_totals = speer_totals;
1197 peer_geos = speer_geos;
1198 peer_weights = speer_weights;
1199
1200 seed_wkeys = sseed_wkeys;
1201 seed_totals = sseed_totals;
1202 seed_geos = sseed_geos;
1203 seed_weights = sseed_weights;
1204
1205 sort_mode = mode;
1206 }

1210 /// parts of it as determined by resolution and encoding.
1211 swarm_gcache
1212 partition_cache_by_mode(resolution_mode rmode, encoding_mode emode)
1213 {

1215 // this is the requested change, do nothing.
1216 bool ta = rmode == resolution_mode::all && emode == encoding_mode::all;
1217 if (!ta)
1218 {
1219 // Setup a sub-partition.
1220 collection c(*this);
1221 c.partition_by_mode(rmode, emode);
1222 swarm_gcache pwcll(c);
1223
1224 // Go through and copy the matching caches.
1225 uint pi = 0;
1226 for (uint i = 0; i < members.size(); ++i)
1227 {
1228 tmarkup& tm = members[i];
1229 bool tr = tm.resolution == rmode || rmode == resolution_mode::all;
1230 bool te = tm.encoding == emode || emode == encoding_mode::all;
1231 if (tr && te)
1232 {
1233 // IIF matches requested partition modes, copy.
1234 pwcll.members[pi] = members[i];
1235
1236 pwcll.peer_geos[pi] = peer_geos[i];
1237 pwcll.peer_weights[pi] = peer_weights[i];
1238 pwcll.peer_totals[pi] = peer_totals[i];
1239
1240 if (weighedp)
1241 pwcll.peer_wkeys[pi] = peer_wkeys[i];
1242
1243 pwcll.seed_geos[pi] = seed_geos[i];
1244 pwcll.seed_weights[pi] = seed_weights[i];
1245 pwcll.seed_totals[pi] = seed_totals[i];
1246
1247 if (weighedp)
1248 pwcll.seed_wkeys[pi] = seed_wkeys[i];
1249
1250 ++pi;
1251 }
1252 }
1253
1254 pwcll.resolution = rmode;
1255 pwcll.encoding = emode;
1256
1257 pwcll.key_mask = key_mask;
1258 pwcll.datestamp = datestamp;
1259 pwcll.sort_mode = sort_mode;
1260 pwcll.weighedp = weighedp;
1261
1262 string modestr = to_string(rmode) + k::space + to_string(emode);
1263 if (pwcll.matchslice.empty())
1264 pwcll.matchslice = modestr;
1265 else
1266 pwcll.matchslice += k::space + modestr;

1268 return pwcll;
1269 }

1453 /**
1454 Combined summaries over topn peers or seeds over (largest topn of) members.
1455
1456 After this, sum_wkeys, sum_geos, and sum_weights will be set.
1457 */
1458 void
1459 summarize_swarm(ulong topn = 250)
1460 {
1461 const bool peerp(sort_mode == swarm_mode::peer);
1462
1463 sum_geos.clear();
1464 sum_wkeys.clear();
1465 sum_weights.clear();
1466 for (ulong i = 0; i < members.size(); ++i)
1467 {
1468 swgeo_peers& wkeys = peerp ? peer_wkeys[i] : seed_wkeys[i];
1469 umgeo_peers& geos = peerp ? peer_geos[i] : seed_geos[i];
1470 uint j = 0;
1471 for (auto itr = wkeys.begin(); itr != wkeys.end(); ++itr)
1472 {
1473 if (j >= topn)
1474 break;
1475 ++j;
1476
1477 auto key = std::get<1>(*itr);
1478 if (sum_weights.count(key) == 1)
1479 {
1480 // Increase weight for key.
1481 const int fweight = std::get<0>(*itr);
1482 const int sweight = sum_weights.at(key);
1483 sum_weights.erase(key);
1484 sum_weights.insert(make_pair(key, sweight + fweight));
1485
1486 // Replace with ranking.
1487 auto itr = sum_wkeys.find(std::make_tuple(sweight, key));
1488 if (itr != sum_wkeys.end())
1489 {
1490 sum_wkeys.erase(itr);
1491 sum_wkeys.insert(make_tuple(sweight + fweight, key));
1492 }
1493 }
1494 else
1495 {
1496 // Get geo peer. Mangle to key. Add to maps. Add key
1497 // to keys.
1498 geo::geo_peer& gp = geos.at(key);
1499 sum_geos.insert(make_pair(key, gp));
1500 sum_wkeys.insert(std::make_tuple(1, key));
1501 sum_weights.insert(make_pair(key, 1));
1502 }
1503 }
1504 }
1505 }
