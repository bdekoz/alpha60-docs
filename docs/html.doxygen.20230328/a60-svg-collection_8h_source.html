42/// First layer.
43void
44place_map_in_frame(svg_element& obj, const carto::cartography<auto>& cartog,
45 projection_base::raster_mode v = projection_base::filled)
46{
47 using atype = svg_element::atype;
48
49 atype width = cartog.p.width;
50 atype height = cartog.p.height;
51 atype xoff = cartog.f.moriginx;
52 atype yoff = cartog.f.moriginy;
53
54 group_element gpre;
55 gpre.start_element("cartographic-image");
56 obj.add_element(gpre);
57
58 image_element i;
59 std::string image_file = cartog.p.image_filename(v);
60 image_element::data di = { image_file, xoff, yoff, width, height };
61 i.start_element();
62 i.add_data(di);
63 i.finish_element();
64 obj.add_element(i);
65
66 group_element gpost;
67 gpost.finish_element();
68 obj.add_element(gpost);
69
70}

135/// Get min, max of weight in weighed peers.
136auto
137find_weight_range(const geo::swgeo_peers& wpeers, const uint maxn)
138{
139 // Current range.
140 uint rmin [[gnu::unused]] = 1;
141 uint rmax [[gnu::unused]] = 1;
142
143
144 auto ritr = wpeers.begin();
145 auto robj = *ritr;
146 rmax = std::get<0>(robj);
147
148 if (wpeers.size() > maxn)
149 {
150 uint n(0);
151 while (n++ < maxn)
152 ++ritr;
153 }
154 else
155 {
156 ritr = wpeers.end();
157 --ritr;
158 }
159 robj = *ritr;
160 rmin = std::get<0>(robj);
161
162 return std::make_tuple(rmin, rmax);
163}

276/// Parameterize scaling normalizes on range (nfloor, nceil).
277/// nfloor == 2, 48
278/// nceil == 396, 10000 (then divide by 25).
279void
280augment_wgeo_peers_location(svg_element& obj, const geo::umgeo_peers& peers,
281 const geo::swgeo_peers& wpeers,
282 const carto::cartography<auto>& cartog,
283 svg::typography& typo, const uint threshold,
284 const uint maxn, std::tuple<uint, uint>& range)
285{
286 // Compute range.
287 auto [ nfloor, nceil ] = range;
288 auto [ rmin, rmax ] = find_weight_range(wpeers, maxn);
289
290 if (!wpeers.empty())
291 {
292 uint topn = 0;
293 for (const auto& wp: wpeers)
294 {
295 uint weight = std::get<0>(wp);
296 const string key = std::get<1>(wp);
297 const geo::geo_peer& p = peers.at(key);
298 if (weight >= threshold && topn++ < maxn)
299 {
300 // Normalize on new range.
301 // Typographic range varies.
302 // 1-160 is large and graphic but not as crazy as the old ago.
303 int nwr = normalize_on_range(weight, rmin, rmax, nfloor, nceil);
304 augment_peer_location(obj, p, cartog, typo, nwr);
305 }
306 }
307 }

539 const carto::cartography<auto>& cartog,
540 string odir, swarm_mode m,
541 string type, render_state& tr)
542{
543 // Output naming.
544 string smode = to_string(m);

553 std::ostringstream ostr;
554 ostr << wcll.slice_name_mangled();
555 ostr << "-x-" << wcll.members.size() << "-torrents-";
556 ostr << smode << "-" << type
557 << "-render-mode-" << tr.weigh << !outlinevec << "-"
558 << viscarto << vistext << visvec << visnet << visnode;
559
560 string lc = to_lowercase_string(cartog.p.name);
561 ostr << "-" << to_replace_spaces_string(lc);
562
563 if (!wcll.postfix.empty())
564 ostr << "-" << to_lowercase_string(wcll.postfix);
565
566 string name = odir + ostr.str();

585svg_element
586augment_base(const collection& wcll, const carto::cartography<auto>& cartog,
587 string odir, swarm_mode m, string type,
588 render_state& rs)
589{
590 // Swarm type.
591 bool peerp = m == swarm_mode::peer;
592
593 if (m == swarm_mode::unknown)
594 throw std::runtime_error("augment_base:: unsorted");
595
596 // Render svg.
597 string name;
598 if (!rs.alt)
599 name = augment_base_map_name(wcll, cartog, odir, m, type, rs);
600 else
601 name = augment_base_info_name(wcll, odir, type);
602
603 std::cout << "slice name: " << wcll.slice_name_mangled() << std::endl;
604 std::cout << "postfix: " << wcll.postfix << std::endl;
605 std::cout << "svg filename: " << name << std::endl;
606
607 svg_element obj(peerp ? make_map_base(name, rs, cartog)
608 : make_map_inverse_base(name, rs, cartog));
609 return obj;
610}

1876 // Split collection into sparse and dense sub-collections.

1878 swarm_gcache wdense(wcll);
1879
1880 // Clear cache arrays but save date.
1881 wsparse.clear();
1882 wdense.clear();
1883 wsparse.datestamp = datet;

1955 dweights.push_back(denseweights);
1956
1957 // Re-count totals for peer_totals/seed_totals;
1958 ulong sparsen = wsparse.summarize_weights(sparseweights);
1959 stotals[i] = sparsen;
