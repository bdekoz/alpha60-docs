57 /// Use nodejs to run D3 projection via d3-geo-polygon
58 a60::point_2t
59 meridians_to_point_2d_via_js(double lt, double lng) const
60 {
61 const string ofname("./js-output.txt");
62 const string script("a60-carto-projection-cahill-keyes.js");
63 const auto& rtr = io::get_run_time_resources();
64
65 // Call script + args, pipe to tmp file
66 std::ostringstream cli;
67 cli << "node";
68 cli << k::space;
69 cli << "--experimental-modules"; // loads modules
70 cli << k::space;
71 cli << rtr.prefix << "/src/" << script;
72 cli << k::space;
73 cli << height / 2;
74 cli << k::space;
75 cli << lng;
76 cli << k::space;
77 cli << lt;
78 cli << k::space;
79 cli << ">> " << ofname;
80
81 int status = system(cli.str().c_str());
82 if (status == -1)
83 {
84 std::ostringstream oss;
85 oss << "system call failed with status:" << status << std::endl;
86 oss << cli.str() << std::endl;
87 throw std::runtime_error(oss.str());
88 }
89
90 double ckx(0);
91 double cky(0);
92 std::ifstream ifs(ofname, std::ios::in);
93 if (!ifs.good())
94 {
95 std::ostringstream oss;
96 oss << "system call output failed: " << ofname << std::endl;
97 throw std::runtime_error(oss.str());
98 }
99 ifs >> ckx;
100 ifs >> cky;
101
102 // Remove temporary file.
103 std::remove(ofname.c_str());
104
105 // ..add to origin point to get final destination.
106 double x = longitude_zero_x + ckx;
107 double y = latitude_zero_y - cky;
108 return std::make_tuple(x, y);
109 }

113 template<int _Height>
114 struct ckcache
115 {
116 static ump2tduce&
117 get_cache()
118 {
119 // Cached pairs of (lat, long) to (x, y)
120 // This has to be unique for every height used, as this changes
121 // the values returned from the projection formula.
122 static ump2tduce xy;
123 return xy;
124 }
125 };
126
127 ump2tduce&
128 get_cache_per() const
129 {
130 // Get per-height cache.
131 if (height == 960)
132 return ckcache<960>::get_cache();
133 if (height == 1056)
134 return ckcache<1056>::get_cache();
135 if (height == 1556)
136 return ckcache<1002>::get_cache();
137 if (height == 1002)
138 return ckcache<1920>::get_cache();
139 if (height == 2112)
140 return ckcache<2112>::get_cache();
141 if (height == 2492)
142 return ckcache<2492>::get_cache();
143 if (height == 2640)
144 return ckcache<2640>::get_cache();
145 if (height == 3297)
146 return ckcache<3297>::get_cache();
147 if (height == 4224)
148 return ckcache<4224>::get_cache();
149 if (height == 8250)
150 return ckcache<8250>::get_cache();
151 if (height == 13163)
152 return ckcache<13163>::get_cache();
153
154 string m("ckproj::get_cache_per height unknown for height value ");
155 m += to_string(height);
156 throw std::runtime_error(m);
157 }
158
159 // Cached version of above.
160 a60::point_2t
161 meridians_to_point_2d(double lt, double lng) const
162 {
163 ump2tduce& cachexy = get_cache_per();
164
165 // Use it.
166 a60::point_2t latlngval = std::make_pair(lt, lng);
167 a60::point_2t retval;
168 auto i = cachexy.find(latlngval);
169 if (i != cachexy.end())
170 retval = i->second;
171 else
172 {
173 retval = meridians_to_point_2d_via_js(lt, lng);
174 cachexy.insert(std::make_pair(latlngval, retval));
175 }
176 return retval;
177 }
178 };

181 /**
182 Cahill-Keyes-as-per visionscarto.net's Cahill-Keyes Projection.
183
184 vc.net == visioncarto.net
185 https://vc.net/carte-di-base
186 https://vc.net/public/fonds-de-cartes-it/visionscarto-cahillkeyes.svg
187
188 https://observablehq.com/@fil/cahill-keyes-projection
189
190 In the Cahill-Keyes octant assembly, y is the distance from the
191 equator to the north pole, -y is the distance from the equator to
192 the south pole, and the total circumfrence of the globe is a
193 distance of 4y.
194
195 For rendering to screen with this projection, derive y via
196 4y = projection.width
197
198 Since the rendered images are full-width but not full-height (for
199 the octant grid where 2y = height.
200 */
