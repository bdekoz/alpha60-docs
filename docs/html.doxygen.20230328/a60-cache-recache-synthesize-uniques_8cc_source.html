43/// Source-local layer editor.
44void
45render_reconsider_local(const double op = get_render_state().opacity,
46 const select son = select::none,
47 const select soff = select::none)
48{
49 // Clear/override any specific render_state characteristics from
50 // initialize_color_and_visibility.
51 render_state& current = get_render_state();
52
53 // For select layers only, suitable for trimming the files size of
54 // the star-x output. Can set/clear/flip any of these.
55 if (son != select::none)
56 current.flip(current.visible_mode, son);
57 if (soff != select::none)
58 current.clear(current.visible_mode, soff);
59
60 current.opacity = op;
61}

63/// Augmentation switch for cartography and slices.
64void
65augment_unique_collection_ck(cachecll& wcllu, const string& odir,
66 const bool oncep,
67 const bool slicelongituderangesp = true)
68{
69 // If slices are to use the whole earth collection without longitude
70 // ranges. Slower, every slice renders all peers. Faster, each
71 // slice only renders peers in longitude range.
72 const bool sliceuseallp [[gnu::unused]] = slicelongituderangesp ? 0 : 1;
73 const bool vdatap [[gnu::unused]] = slicelongituderangesp ? 1 : 0;

76 const auto& ckwecarto = ckwecarto_engc;
77 //const auto& ckwecarto = ckwecarto_qa4;
78 // const auto& ckwecarto = ckwecarto_1080p;

81 const bool metadatap = true;
82 if (metadatap)
83 {
84 augment_stateless(wcllu, ckwecarto, odir);

87 // Cahill-Keyes
88 // Whole Earth
89 const bool wholeearthp = false;
90 if (wholeearthp)
91 {
92 augment_weighed_cumulative_state(wcllu, ckwecarto, ckwestate,

97 const bool quartop = true;

114 {
115 // Slices, h no v
116 carto_state cstate = ckwestate;
117 const auto& ckh3carto = ckh3carto_engc;

199/// Serialization top level.
200void
201meta_recache_serialize(collection cll, const string& odir,
202 const k::sduration zduration,
203 const bool detailp = false)
204{
205 using namespace a60::k;
206 const string cdir = io::get_run_time_resources().cache;
207 const string prefix("unique-btih-");
208
209 if (is_sduration(zduration, sduration::hour))
210 serialize_btiha_per(cll, odir, cdir, hour_s, sduration::hour);
211
212 if (is_sduration(zduration, sduration::day))
213 serialize_btiha_per(cll, odir, cdir, day_s, sduration::day, detailp);
214
215 if (is_sduration(zduration, sduration::week))
216 serialize_btiha_per(cll, odir, cdir, week_s, sduration::week, detailp);
217
218 if (is_sduration(zduration, sduration::cumulative))
219 {
220 const bool detailcumup = true;
221 if (detailcumup)
222 {
223 // Cumulative summary with more extensive JSON, including
224 // additional cumulative per-duration (week, day) totals.
225 // NB: Be careful about enabling detail with large durations
226 // as this will severly bloat the size JSON file...
227
228 // const sduration zdurationd = zduration;
229 // const sduration zdurationd(sduration::day | sduration::week);
230 const sduration zdurationd = sduration::week;
231
232 serialize_unique_collection_duration(cll, odir, cumulative_s,
233 zdurationd);
234 }
235 else
236 {
237 // Swarm minimal-summary for just cumulative duration.
238 cachecll wcll = coalesce_with_unique_btiha(cll, cumulative_s);
239 serialize_unique_collection(wcll, odir, prefix + cumulative_s);
240 }
241
242 }

248meta_render(collection cll, const string& odir,
249 const k::sduration rduration, const swarm_mode m,
250 const bool unique_btiha_p)
251{
252 using namespace a60::k;
253 const string cdir = io::get_run_time_resources().cache;

257

260 const projection_type proj_t = cahillkeyes;
261
262 // Augment cumulative.
263 if (is_sduration(rduration, sduration::cumulative))
264 {
265 cachecll wcll = coalesce_with_unique_btiha(cll, cumulative_s);
266 weigh_sort_unique_collection(wcll, m);
267 augment_unique_collection(wcll, odir, oncep, proj_t);
268 }
269
270 // Setup cartography for durations.
271#if 0
272 carto_state cstate = ckh2state;
273 const auto& cartog = ckh2carto_1080p;
274#else
275 carto_state cstate = ckwestate;
276 const auto& cartog = ckwecarto_1080p;
277#endif
278
279 // NB: Other setup required but not automated or necessarily checked....
280 // don't use ubtiha
281 // no slices support
282 // a60_SINGLE
283 // dcomparefn / eq_previous_p
284 // available cache-persistent and cache/cumulative for this to work...
285
286 // Week
287 if (is_sduration(rduration, sduration::week))
288 {
289 std::cout << "begin week duration render" << std::endl;
290 scoped_timer tstart(std::cout, "elapsed");
291
292#if 1
293 // WIP Slice support.
294 // Find range of weeks sampled.
295 const string d = week_s;
296 strings files = cll.match_files(cdir, d);
297 strings durations = find_unique_durations(files, d);

300
301 for (uint i = 0; i < durations.size(); ++i)

303 const string& curd = durations[i];

306 sduration::week, curd, m,
307 equal_and_previous_p);
308 }
309#else
310 // Whole Earth, no slices, detailed metrics and color bar.
311 augment_weighed_duration_since(cll, cartog, odir, m, week_s,
312 equal_and_previous_p,
313 svg::color::asamaorange, cstate, true);
314#endif
315
316 std::cout << "end week duration render, ";
317 }
318
319
320 // Day
321 if (is_sduration(rduration, sduration::day) && !unique_btiha_p)
322 {
323 std::cout << "begin day duration render" << std::endl;
324 scoped_timer tstart(std::cout, "elapsed");

326 augment_weighed_duration_since(cll, cartog, odir, m, day_s,

328 svg::color::kissmepink, cstate, true);
329
330 std::cout << "end day duration render, ";

336 std::cout << "begin hour duration render" << std::endl;

341 svg::color::asamaorange, cstate, true);

351meta_ip_analysis(collection cll, const string dur, bool detailp)
352{
353 swarm swrm(coalesce_with_unique_btiha(cll, dur, detailp));

355 std::cout << "swarm datestamp: " << swrm.datestamp << std::endl;

357
358 auto xnodesz = swrm.exit_nodes.size();
359 std::cout << "tor exit nodes size: " << xnodesz << std::endl;
360 std::cout << std::endl;
361
362 // peer
363 ulong psz(0);
364 for (auto& s: swrm.peers)
365 psz += s.size();
366 std::cout << "swarm upeers total: " << psz << std::endl;
367 auto& peer_anomalies = swrm.peer_anomalies;

372 double pxnodesz = peer_anomalies.exit_node_tids.size();

374 std::cout << k::tab << "tor-exit-nodes: " << pxnodesz << k::tab << pperc;
375 std::cout << std::endl;
376 std::cout << std::endl;
377
378 // seed

397int main(int argc, char* argv[])
398{
399 using namespace a60;
400 using namespace a60::k;
401 using namespace a60::media_objects;
402
403 // Sanity check.
404 if (argc > 2)
405 {
406 fputs(usage().c_str(), stderr);
407
408 int i = argc;
409 while (i - 1 > 0)
410 {
411 --i;
412 std::cout << i + 1 << a60::k::tab << argv[i] << std::endl;
413 }
414 }
415
416 // Set up output environment.
417 // Iff argument is passed for output directory, use it. Else, pwd + tmp.
418 std::string odir;
419 if (argc == 2)
420 {
421 odir = argv[1];
422 io::get_output_prefix(odir);
423 }
424 else
425 odir = io::get_output_directory("tmp");
426 io::sanity_check_output_directory(odir);
427
428 // Report.
429 std::cout << "sample using: " << odir << " for output" << std::endl;
430
431 const std::string cdir = a60::k::cache_s;
432 const std::string cdirp = a60::k::cache_persistent_s;
433 io::set_run_time_resources(cdir, cdirp);
434
435 // Choices start here.
436
437 const swarm_mode m = swarm_mode::peer;
438
439 // Initialize collection from btiha or ubtiha (unique).
440 // If duration week/day/hour, btiha so false. Else ubtiha true (the default).
441 bool unique_btiha_p = true;
442 bool generatedk = true;
443 if (generatedk)
444 {
445 // Default color path is unique btiha, pick from
446 // resolution-deliniated style and generate random colorband of size 300.
447 setup_render(cll, m, unique_btiha_p, false);
448 }
449 else
450 {
451 // Discrete color exceptions.
452 // cband_brown, cband_y, cband_bw, cband_gray, cband_r
453 setup_render(cll, m, unique_btiha_p, true, svg::cband_bw);
454 }
455
456 // Duration Serialization is on by default.
457 const sduration zduration = sduration_dwc;
458 if (zduration != sduration::none)
459 meta_recache_serialize(cll, odir, zduration);
460
461 // IP checking.
462 constexpr bool ipanalysisp = true;
463 if (ipanalysisp)
464 meta_ip_analysis(cll, cumulative_s, true);
465
466 // Duration Rendering is off by default.
467 const sduration rduration = sduration::cumulative;
468 if (rduration != sduration::none)
469 meta_render(cll, odir, rduration, m, unique_btiha_p);
470
471 return 0;
472}
