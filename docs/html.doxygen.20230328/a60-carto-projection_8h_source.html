24 /// Projection types.
25 enum projection_type { cahillkeyes, equirectangular };

27 /**
28 Projection: the visual representation, mapping 3D to 2D.
29
30 Transforming spherical shapes to 2D surfaces is not trivial.
31
32 Doing this correctly requires
33 1. 2D representation
34 2. math representing the 3D -> 2D transformation, ie the projection.
35
36 Potential projections for world maps are: Voroni, Cahill-Keyes,
37 Robinson, Winkel Tripel, Natural Earth, AuthaGraph WorldMap, etc.
38 */
39 struct projection_base
40 {
41 /**
42 Raster versions.
43
44 3 base maps for all viz
45 cgphy - gray land white outline map on white
46 cgphyout - white land black outline map on white aka (outline)
47 cgphyinv - black land white outline map on black
48 aka (outline x inverse)
49 */
50 enum raster_mode { filled, outline, inverse, grid, glitch };
51
52 /// Map size in pixels.
53 double width;
54 double height;
55
56 /// Meridian center point (0,0) and origin in pixels for drawing.
57 double longitude_zero_x;
58 double latitude_zero_y;
59
60 /// Projection name.
61 std::string name;
62 };

77 virtual a60::point_2t

79 };
