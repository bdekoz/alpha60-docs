27/// Cache swarm data, if not already done.
28/// cdir == cache file directory
29/// cdirp == persistent cache file directory
30void
31cache_collection(const collection& cll, const string cdir, const string cdirp,
32 const k::sduration sdur,
33 const bool cachedp = false, const bool persistp = false)
34{
35 using a60::k::sduration;
36
37 // Coalesce log.
38 std::ofstream ofscoa(cdir + cll.match + "-coalesce.log");
39 auto coutbuf = std::cout.rdbuf();
40 std::cout.rdbuf(ofscoa.rdbuf());
41
42 const uint ddigits [[gnu::unused]] = cll.mangle_duration_significant_digits();
43
44 // Warning: can substantially increase cache runtimes.
45 if (is_sduration(sdur, sduration::sample))
46 {
47 scoped_timer t(ofscoa, cll.match + " coalesce per sample");
48 coalesce_torrents_per_sample(cll, cdir);
49 }
50
51 // HOUR
52 if (is_sduration(sdur, sduration::hour))
53 {
54 scoped_timer t(ofscoa, cll.match + " coalesce per hour");
55 coalesce_torrents_per_hour(cll, cdir, cachedp);
56
57 if (persistp)
58 {
59 strings durations = cll.find_data_dates_times();
60 strings dts = find_unique_date_hour_stamps(durations);
61
62 // Don't compute this by default.
63 // Easy to make miserably long run times here....
64 // Either turn off subsample generation for this duration, or
65 // change it to (1 hour). Generated subsample duration makes no sense.
66 bool persisthour(false);
67 if (persisthour)
68 {
69 scoped_timer t(ofscoa, cll.match + " persistence per hour");
70 persistent_collection_per(cll, k::hour_s, dts, cdir, cdirp);
71 }
72
73 // Don't compute this by default.
74 // Smaller version of above, that assumes caches and persistent
75 // caches are already built, and is just for re-computing the
76 // swarm summaries computed immediately above.
77 bool swarmsumhour(false);
78 if (!persisthour && swarmsumhour)
79 {
80 scoped_timer t(ofscoa, cll.match + " swarm persistence per hour");
81 summarize_collection_per(cll, k::hour_s, dts, cdir, cdirp);
82 }
83 }
84 }
85
86 // DAY
87 if (is_sduration(sdur, sduration::day))
88 {
89 scoped_timer t(ofscoa, cll.match + " coalesce per " + k::day_s);
90 coalesce_torrents_per_day(cll, cdir);
91
92 if (persistp)
93 {
94 scoped_timer t(ofscoa, cll.match + " persistence per " + k::day_s);
95 strings files = cll.cache_files_by(cdir, k::day_s);
96 const strings dts = find_unique_date_stamps(files, ddigits);
97 persistent_collection_per(cll, k::day_s, dts, cdir, cdirp);
98 }
99 }
100
101 // WEEK
102 if (is_sduration(sdur, sduration::week))
103 {
104 scoped_timer t(ofscoa, cll.match + " coalesce per " + k::week_s);
105 coalesce_torrents_per_week(cll, cdir);
106
107 if (persistp)
108 {
109 scoped_timer t(ofscoa, cll.match + " persistence per " + k::week_s);
110 strings files = cll.cache_files_by(cdir, k::week_s);
111 strings durations;
112 durations = find_unique_duration_date_stamps(files, k::week_s);
113 const strings dts = find_unique_date_stamp_ranges(durations,
114 k::week_s, ddigits);
115 persistent_collection_per(cll, k::week_s, dts, cdir, cdirp);
116 }
117 }
118
119 // CUMULATIVE
120 if (is_sduration(sdur, sduration::cumulative))
121 {
122 scoped_timer t(ofscoa, cll.match + " coalesce " + k::cumulative_s);
123 coalesce_torrents_cumulative(cll, cdir);
124
125 if (persistp)
126 {
127 scoped_timer t(ofscoa, cll.match + " persistence per " + k::cumulative_s);
128 strings files = cll.cache_files_by(cdir, k::cumulative_s);
129 strings durations;
130 durations = find_unique_duration_date_stamps(files, k::cumulative_s);
131 const strings dts = find_unique_date_stamp_ranges(durations,
132 k::cumulative_s,
133 ddigits);
134
135 // Should be just one.
136 std::cout << "durations size: " << dts.size() << std::endl;
137 for (const string& s: dts)
138 std::cout << s << std::endl;
139
140 // Just swarm data as there is no persistence for cumlative data.
141 summarize_collection_per(cll, k::cumulative_s, dts, cdir, cdirp);
142 }
143 }
144
145 std::cout.rdbuf(coutbuf);
146}
