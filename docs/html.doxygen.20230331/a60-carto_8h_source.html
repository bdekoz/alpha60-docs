39
40 - projection (representation of Earth)

54 Data Slice.
55

57 be a geographic range, say longitudes from -175 to -30 (aka

59

62 USA

66 Longitude Ranges
67

74*/

80/**
81 Render Slices, Enlargement, & Tiling.
82
83 Some (slice) renderings use more than one frame to complete the
84 whole image. This is called tiling, where the image is broken up
85 in segments over a couple of frames, aka pages.
86
87 1x == single || none || whole_earth
88 1x all longitude values
89
90 2x == duo (C-K) [landscape diptych engc]
91 2x two logitude ranges
92 1 = pacifica americas
93 2 = afro eur asia
94
95 [3,6]x == trio (UN) [landscape 1,2,3 tiles x north,south engc]
96 1 = afro eur asia
97 2 = americas
98 3 = asia pacifica
99
100 4x == quarto (C-K) [portrait 1,2,3,4 tiles engc]
101 1 = pacifica america
102 2 = americas
103 3 = afro eur asia
104 4 = asia pacifica
105*/
106enum class slice_mode
107 {
108 none,
109 whole_earth,
110 country_code,
111 h_duo,
112 h_duo_1, h_duo_2,
113 h_trio,
114 h_trio_1, h_trio_2, h_trio_3,
115 h_quarto,
116 h_quarto_1, h_quarto_2, h_quarto_3, h_quarto_4,
117 v_duo,
118 v_duo_north, v_duo_south,
119 all
120 };

123/// Cartographic state type.
124struct carto_state
125{
126 projection_type proj;
127 slice_mode hslyc;
128 slice_mode vslyc;
129
130 carto_state(projection_type p, slice_mode h, slice_mode v = slice_mode::none)
131 : proj(p), hslyc(h), vslyc(v) { }
132
133 bool
134 is_h_slice() const
135 {
136 const bool b2all(hslyc == slice_mode::h_duo);
137 const bool b2(hslyc == slice_mode::h_duo_1 || hslyc == slice_mode::h_duo_2);
138 const bool b3all(hslyc == slice_mode::h_trio);
139 const bool b31(hslyc == slice_mode::h_trio_1);
140 const bool b32(hslyc == slice_mode::h_trio_2);
141 const bool b33(hslyc == slice_mode::h_trio_3);
142 const bool b3(b31 || b32 || b33);
143 const bool b4all(hslyc == slice_mode::h_quarto);
144 const bool b41(hslyc == slice_mode::h_quarto_1);
145 const bool b42(hslyc == slice_mode::h_quarto_2);
146 const bool b43(hslyc == slice_mode::h_quarto_3);
147 const bool b44(hslyc == slice_mode::h_quarto_4);
148 const bool b4(b41 || b42 || b43 || b44);
149 return bool(b2all || b2 || b3all || b3 || b4all || b4);
150 }
151
152 bool
153 is_v_slice() const
154 {
155 const bool vn(vslyc == slice_mode::v_duo_north);
156 const bool vs(vslyc == slice_mode::v_duo_south);
157 return bool(vslyc == slice_mode::v_duo || vn || vs);
158 }
159};

180/**
181 Cartography,
182 meta object for all components of visual representation of a
183 geosphere.
184
185 Includes the cartographic projection, and the frame in which it
186 will be placed. Together, these two items are called a
187 cartographic plate, or, simply the "plate."
188
189 Projections that span more than one plate are "tiled plates" or "tiles."
190*/
191template<typename _Proj>
192struct cartography
193{
194 using projection = _Proj;
195
196 frame f;
197 projection p;
198
199 // Offsets for horizontal and vertical tiles, used in slices and
200 // larger displays.
201 vd h_tiles;
202 vd v_tiles;
203
204 cartography(const frame& fin, const projection& pin, const bool cp = false)
205 : f(fin), p(pin)
206 {
207 if (cp)
208 f.center(p);
209 }
210
211 cartography(const frame& fin, const projection& pin,
212 const vd& htile, const vd& vtile)
213 : f(fin), p(pin), h_tiles(htile), v_tiles(vtile)
214 { }
215
216 // Slices with offsets.
217 cartography(const frame& fin, const projection& pin,
218 const double x, const double y)
219 : f(fin.width, fin.height, x, y), p(pin)
220 { }
221
222 cartography(const cartography& other)
223 : f(other.f), p(other.p), h_tiles(other.h_tiles), v_tiles(other.v_tiles)
224 { }
225
226 const frame&
227 get_frame() const
228 { return f; }
229
230 const projection&
231 get_projection() const
232 { return p; }
233
234 void
235 set_frame(const frame& fin)
236 { f = fin; }
237
238 void
239 set_frame_portrait(void)
240 {
241 if (f.width > f.height)
242 {
243 auto oldw = f.width;
244 auto oldh = f.height;
245 f.width = oldh;
246 f.height = oldw;
247 }
248 }
249
250 void
251 set_projection(const projection& pin)
252 { p = pin; }
253
254 void
255 set(const cartography& crt)
256 {
257 f = crt.f;
258 p = crt.p;
259 }

261 a60::point_2t
262 to_point_2d(double lt, double lng) const
263 {
264 auto [ x, y ] = p.meridians_to_point_2d(lt, lng);
265
266 // Position map in frame.
267 x += f.moriginx;
268 y += f.moriginy;
269
270 return std::make_tuple(x, y);
271 }

273 /**
274 Proceeding left to right from the leftmost edge of the projection:
275
276 quad 1 == x [0, quartowidth)
277 quad 2 == x [quartowidth, quartowidth * 2)
278 quad 3 == x [quartowidth * 2, quartowidth * 3)
279 quad 4 == x [quartowidth * 3, quartowidth * 4)
280
281 Cahill-Keyes Butterfly has has 8 octants that have longitude ranges:
282 (160-180 + -180 to -110), (-110, 0), (0, 70), (70, 160)
283
284 Thus:
285 "leftmost" is longitude 160
286 "rightmost" is longitude 159
287 */
288 slice_mode
289 point_to_x_quadrant(const point_2t po) const
290 {
291 const auto quartowidth = p.width / 4;
292 const auto xo = f.moriginx;
293 auto [x, y] = po;
294
295 slice_mode m = slice_mode::none;
296 if (xo <= x && x < xo + quartowidth)
297 m = slice_mode::h_quarto_1;
298 if (xo + quartowidth <= x && x < xo + (quartowidth * 2))
299 m = slice_mode::h_quarto_2;
300 if (xo + (quartowidth * 2) <= x && x < xo + (quartowidth * 3))
301 m = slice_mode::h_quarto_3;
302 if (xo + (quartowidth * 3) <= x && x < xo + (quartowidth * 4))
303 m = slice_mode::h_quarto_4;
304 return m;
305 }
306
307 /// Top (h1) and bottom (h2) vertical.
308 slice_mode
309 point_to_y_half(const point_2t po) const
310 {
311 const auto halfy = p.height / 2;
312 const auto yo = f.moriginy;
313 auto [x, y] = po;
314
315 slice_mode m = slice_mode::none;
316 if (yo <= y && y < yo + halfy)
317 m = slice_mode::v_duo_north;
318 else
319 m = slice_mode::v_duo_south;
320 return m;
321 }
322};

368/// Slice base type
369struct slice_base
370{
371 using srange = a60::point_2t;
372 static constexpr srange longr_all = std::make_tuple(-180.0, 180.0);
373
374 /// Longitude ranges trio.
375 static constexpr srange longr_3_asiap = std::make_tuple(60.0, 180.0);
376 static constexpr srange longr_3_afroe = std::make_tuple(-29.99, 59.99);
377 static constexpr srange longr_3_amecs = std::make_tuple(-175.0, -30.0);
378
379 /// Longitude ranges quadro.
380 /// Horizontal slices from Cahill-Keyes octets
381 /// (using adjacent north/south pairs of octets)
382 /// NB: For this projection, want both 160 to 180 and -110 to -180
383 static constexpr srange longr_4_1 = std::make_tuple(70.0, 159.99);
384 static constexpr srange longr_4_2 = std::make_tuple(-20, 69.99);
385 static constexpr srange longr_4_3 = std::make_tuple(-110.0, -19.99);
386 static constexpr srange longr_4_4 = std::make_tuple(160.0, -110.01);
387 static constexpr srange longr_4_4a = std::make_tuple(160.0, 180.0);
388 static constexpr srange longr_4_4b = std::make_tuple(-180, -110.01);
389
390 /// Longitude ranges duo, idealized and including overlap.
391
392 // Without overlap: longr_4_4a + longr_4_4b + longr_4_3
393 static constexpr srange longr_2_1 = std::make_tuple(-20, 159.99);
394
395 // Without overlap: longr_4_2 + longr_4_1
396 static constexpr srange longr_2_2 = std::make_tuple(160, -19.99);
397
398
399 /// Latitude ranges.
400 static constexpr srange latr_all = std::make_tuple(90.0, -90.0);
401
402 static constexpr srange latr_n = std::make_tuple(90.0, 0.0);
403 static constexpr srange latr_s = std::make_tuple(0.0, -90.0);
404};

407/// Slice type
408template<typename _Proj>
409struct slice: slice_base
410{
411 using projection = _Proj;
412 using carto_type = cartography<projection>;
413
414 string name;
415
416 /// Longitude, (-180 to 180)
417 /// NB: Each range in longr should not overlap.
418 vrange longr;
419
420 /// Latitude
421 /// (90 to 0) == Northern Hemisphere
422 /// (0 to -90) == Southern Hemisphere
423 /// Each range in longr should not overlap.
424 vrange latr;
425
426 carto_type plate;
427
428 slice(const string nm, const srange lng1, const srange lt1,
429 const carto_type& crto)
430 : name(nm), longr({lng1}), latr({lt1}), plate(crto)
431 { }
432
433 slice(const slice&) = default;
434};

448{
449 using slice_type = ckproj_slices::value_type;

453 const slice_mode hslyc = cstate.hslyc;

467 slice_base::longr_4_3, slice_base::latr_all, carto2);
468 s.push_back(s2);
469 }
470 if (hslyc == slice_mode::h_quarto || hslyc == slice_mode::h_quarto_3)
471 {
472 const cartography carto3(carto.f, carto.p, xoff[2], yoff);
473 slice s3(prefix + "Afro Eur Asia",
474 slice_base::longr_4_2, slice_base::latr_all, carto3);
475 s.push_back(s3);
476 }
477 if (hslyc == slice_mode::h_quarto || hslyc == slice_mode::h_quarto_4)

482 s.push_back(s4);

675/// Cahill-Keyes slices.
676template<>
677slices<ckproj>
678get_slices(const cartography<ckproj>& carto, const carto_state cstate)
679{
680 slices<ckproj> s;
681 const slice_mode hslyc = cstate.hslyc;
682 const slice_mode vslyc = cstate.vslyc;
683
684 const bool pv(vslyc == slice_mode::all || vslyc == slice_mode::v_duo);
685
686 /// Quattro
687 const bool p41(hslyc == slice_mode::h_quarto_1);
688 const bool p42(hslyc == slice_mode::h_quarto_2);
689 const bool p43(hslyc == slice_mode::h_quarto_3);
690 const bool p44(hslyc == slice_mode::h_quarto_4);
691 const bool all4p(hslyc == slice_mode::h_quarto);
692 if (p41 || p42 || p43 || p44 || all4p)
693 {
694 // No vertical slice.
695 if (vslyc == slice_mode::none)
696 get_slices_quatro(s, carto, cstate);
697
698 // Two vertical slices: north, south, each a landscape plate.
699 if (pv || vslyc == slice_mode::v_duo_north)
700 get_slices_quatro(s, carto, cstate, "North ");
701 if (pv || vslyc == slice_mode::v_duo_south)
702 get_slices_quatro_v_s(s, carto, cstate);
703 }
704
705 /// Trio
706 const bool p31(hslyc == slice_mode::h_trio_1);
707 const bool p32(hslyc == slice_mode::h_trio_2);
708 const bool p33(hslyc == slice_mode::h_trio_3);
709 const bool all3p(hslyc == slice_mode::h_trio);
710 if (p31 || p32 || p33 || all3p)
711 {
712 // No vertical slice, plate is the size of 2 landscape plates.
713 if (vslyc == slice_mode::none)
714 {
715 frame fdoublev(carto.f);
716 fdoublev.height *= 2;
717 auto cartodoublev(carto);
718 cartodoublev.f = fdoublev;
719 get_slices_trio(s, cartodoublev, cstate);
720 }
721
722 // Two vertical slices: north, south, each a landscape plate.
723 if (pv || vslyc == slice_mode::v_duo_north)
724 get_slices_trio(s, carto, cstate, "North ");
725 if (pv || vslyc == slice_mode::v_duo_south)
726 get_slices_trio_v_s(s, carto, cstate);
727 }
728
729 /// Duo
730 const bool p21(hslyc == slice_mode::h_duo_1);
731 const bool p22(hslyc == slice_mode::h_duo_2);
732 const bool all2p(hslyc == slice_mode::h_duo);
733 if (p21 || p22 || all2p)
734 {
735 // No vertical slice.
736 if (vslyc == slice_mode::none)
737 get_slices_duo(s, carto, cstate);
738 else
739 {
740 string m("get_slices:: h duo does not have a vertical slice");
741 throw std::runtime_error(m);
742 }
743 }
744
745 if (hslyc == slice_mode::country_code)
746 get_slices_by_country(s, carto);
747 return s;
748}
