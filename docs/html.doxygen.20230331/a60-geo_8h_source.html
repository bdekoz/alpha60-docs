49/// Geo metadata.
50enum meta_mask
51 {
52 latitudem = 1L << 0,
53 longitudem = 1L << 1,
54 country_codem = 1L << 2,
55 countrym = 1L << 3,

60/// Geo metadata index.
61enum meta_idx
62 {
63 latitude,
64 longitude,

71/// Geo metadata type.
72using meta_type = std::tuple<double, double, string, string, string, string>;
73
74inline constexpr meta_mask
75operator&(meta_mask __a, meta_mask __b)
76{ return meta_mask(static_cast<int>(__a) & static_cast<int>(__b)); }
77
78inline constexpr meta_mask
79operator|(meta_mask __a, meta_mask __b)
80{ return meta_mask(static_cast<int>(__a) | static_cast<int>(__b)); }

124/// Straightforward statement on the unknown.
125constexpr const char* alpha_3_not_found = "Unknown or Legendary";
126
127const string
128alpha_3_to_country(const umstrings& a3country, string match)
129{
130 string country(alpha_3_not_found);
131 const auto foundi = a3country.find(match);
132 if (foundi != a3country.end())
133 country = foundi->second;
134 return country;
135}
136
137/// Tries harder.
138const string
139country_to_alpha_3(const umstrings& country2a3, string match)
140{
141 string a3(match);
142 const auto foundi = country2a3.find(match);
143 if (foundi != country2a3.end())
144 a3 = (foundi->second);
145 else
146 {
147 // Look for a partial match and if found, use that.
148 for (const auto& e: country2a3)
149 {
150 string country = e.first;
151 string code = e.second;
152 if (country.find(match) != string::npos)
153 a3 = code;
154 }
155 }
156 return a3;
157}

159/// Convert ISO 2 to ISO 3.
160const string
161alpha_2_to_alpha_3(const string a2)
162{
163 // Setup the map.
164 static umstrings a23 = deserialize_json_to_iso_3166_2_map();
165
166 string a3;
167 const auto foundi = a23.find(a2);
168 if (foundi != a23.end())
169 a3 = (foundi->second);
170 return a3;
171}

254/// Country mapping.
255struct country_map : map_base
256{
257 country_map()
258 : map_base(nullptr, "/usr/share/GeoIP/", "GeoLiteCountry.dat", true) { }
259
260 string
261 country_code_via_string(const string addr [[gnu::unused]]) const
262 {
263 string country;
264 if (is_map_open())
265 {
266#ifdef a60_USE_GEO
267 GeoIPLookup gl = { };
268 auto* p = GeoIP_country_code3_by_addr_gl(geop, addr.c_str(), &gl);
269 if (p == NULL)
270 p = GeoIP_country_code_by_addr_gl(geop, addr.c_str(), &gl);
271
272 if (p == NULL && verbose)
273 {
274 std::cerr << "country_map::country_code_via_string is null"
275 << std::endl
276 << k::tab << "for map name: " << name << std::endl
277 << k::tab << "for ip: " << addr << std::endl
278 << std::endl;
279 }
280
281 country = geo::safe_if_empty(p);
282#endif
283 }
284 return country;
285 }

318
319 city_map()
320 : map_base(nullptr, "/usr/share/GeoIP/", "GeoLiteCity.dat", true)
321 { }
322
323 record
324 open_record(const string addr [[gnu::unused]] )
325 {
326 record::record_ptr p(nullptr);
327#ifdef a60_USE_GEO
328 if (is_map_open())
329 {
330 auto rtmp = GeoIP_record_by_name(geop, addr.c_str());
331 if (rtmp == NULL && verbose)
332 {
333 std::cerr << "city_map::open_record:: error finding address " << addr
334 << " in database: " << name << std::endl;
335 }
336 else
337 p = rtmp;
338 }
339#endif
340 return record(p);
341 }
342
343 string
344 record_to_string(record& r [[gnu::unused]] )
345 {
346 string region;
347#ifdef a60_USE_GEO
348 if (r.is_record_open())
349 {
350
351 string s1(safe_if_empty(r.ptr->region));
352 string s2(safe_if_empty(r.ptr->city));
353 auto lat(r.ptr->latitude);
354 auto lon(r.ptr->longitude);
355
356 const char tab('\t');
357 std::ostringstream oss;
358 oss << std::setprecision(4);
359 oss << lat << tab;
360 oss << lon << tab;
361 oss << s1 << tab;
362 oss << s2 << tab;
363 oss << std::endl;
364 region = oss.str();
365 }
366#endif
367 return region;
368 }

372/// Address to IP/Geo.
373string
374display_ip_and_geo_info(string addr)
375{
376 using namespace std;
377
378 ostringstream oss;
379
380 const char tab('\t');
381 oss << "scraper geoip: " << addr << tab;
382
383#ifdef a60_USE_GEO
384 double lt(0);
385 double lng(0);
386 string region;
387 string country;
388 string city;
389
390 geo::country_map cmap;
391 cmap.open();
392 geo::city_map ctymap;
393 ctymap.open(GEOIP_INDEX_CACHE);
394
395 country = cmap.country_code_via_string(addr);
396
397 geo::city_map::record r = ctymap.open_record(addr);
398 if (r.is_record_open())
399 {
400 lt = r.ptr->latitude;
401 lng = r.ptr->longitude;
402 region = geo::safe_if_empty(r.ptr->region);
403 city = geo::safe_if_empty(r.ptr->city);
404 }
405
406 oss << country << tab;
407 oss << setprecision(5);
408 oss << lt << tab;
409 oss << lng << tab;
410 oss << region << tab;
411 oss << city;
412 oss << endl;
413#endif
414
415 return oss.str();
416}
