326/// Aggregate for slices.
327void
328augment_carto_slices(string odir, auto& cartog, const carto_state cstate)
329{
330 using proj_type = decltype(cartog.p);
331 const slices<proj_type>& sls = get_slices(cartog, cstate);
332 if (!sls.empty())
333 {
334 uint sn = 1;
335 for (const slice<proj_type>& sl : sls)
336 {
337 const char cspace = svg::k::space;
338 string slicenum = to_string(sn++) + cspace + to_string(sls.size());
339 string slicenbase = slicenum + cspace + sl.name;
340 string namelc = to_replace_string(to_lowercase_string(slicenbase));
341 augment_carto_variations(odir, sl.plate, namelc);
342 augment_carto_composite(odir, sl.plate, namelc);
343 }
344 }
