33/**
34 WCAG 2.0 level AA requires a contrast ratio of at least 4.5:1 for
35 normal text and 3:1 for large text.
36
37 WCAG 2.1 requires a contrast ratio of at least 3:1 for graphics and
38 user interface components (such as form input borders).
39
40 WCAG Level AAA requires a contrast ratio of at least 7:1 for normal
41 text and 4.5:1 for large text.
42
43 Large text is defined as 14 point (typically 18.66px) and bold or
44 larger, or 18 point (typically 24px) or larger.
45
46 Here are some examples of text with almost exactly 4.5:1 contrast.
47
48 Gray (#767676) on white (118)
49 Purple (#CC21CC) white
50 Blue (#000063) on gray (#808080)
51 Red (#E60000) on yellow (#FFFF47)
52
53 // Large text
54 Gray (#949494) 18 point text on white (148)
55 Purple (#C86ff1) 14 point + bold text on white
56
57 Contrast ratios:
58 black 21:1
59 blue 8.6:1
60 red 4:1
61 green 1.4:1
62
63 Base style for metadata is either white or dark gray.
64 Default is dark gray.
65*/
66style
67get_metadata_style(const bool isblackp = true)
68{
69 const style hcb_styl = { color::wcag_dgray, 1.0, color::white, 0.0, 1 };
70 const style hcw_styl = { color::white, 1.0, color::wcag_dgray, 0.0, 1 };
71 return isblackp ? hcb_styl : hcw_styl;
72}

98/// Typographic defaults for text labels.
99typography
100get_metadata_typo(const space_type fsize = 10, const bool boldp = false,
101 const bool centerp = false)
102{
103 typography typo = svg::k::hyperl_typo;
104 typo._M_size = fsize;
105 typo._M_style = get_metadata_style();
106
107 if (centerp)
108 {
109 typo._M_anchor = typography::anchor::middle;
110 typo._M_align = typography::align::center;
111 }
112 else
113 {
114 typo._M_align = typography::align::left;
115 typo._M_anchor = typography::anchor::start;
116 }
117
118 if (boldp)
119 typo._M_w = typography::weight::bold;
120 return typo;
121}

124/// Dump text files for dimensions over selected aggregate.
125void
126initial_metadata_characteristics()
127{
128 // Auto-assign the rest.
129 // Find all the metadata files.
130 const string idir = io::get_run_time_resources().metadata;
131 strings files = io::populate_files(idir, ".json");
132 std::cout << files.size() << " metadata files found" << std::endl;
133
134 // Store summary info from processing all files in memory.
135
136 // SHOW MEASURES
137 // SHOW KEYS
138 // Lists of all tag values in each idenity dimension...
139 vsstrings vuvalues;
140 for (const string& tag : k::id_dimensions)
141 {
142 sstrings uvalues;
143 for (const string& f: files)
144 {
145 rj::Document dom = deserialize_json_to_dom_object(f);
146 msstrings vals = values_from_fields_matching(dom, tag);
147 uvalues.insert(vals.begin(), vals.end());
148 }
149 vuvalues.push_back(uvalues);
150
151 string ofname(tag + "-values-" + std::to_string(files.size()));
152 std::ofstream ofs(ofname + ".txt");
153 for (const string& s: uvalues)
154 ofs << s << std::endl;
155 }
156
157 // Lists of all tag values for reception +-
158 // distribution-tags, production-tags, genre-tags
159 strings sfields = { "reception-positive", "reception-negative",
160 "distribution-tags", "production-tags", "genre-tags" };
161 for (const string& stag : sfields)
162 {
163 sstrings uvalues;
164 for (const string& f: files)
165 {
166 rj::Document dom = deserialize_json_to_dom_object(f);
167 strings vals = values_from_field(dom, stag);
168 uvalues.insert(vals.begin(), vals.end());
169 }
170
171 // Check to make sure some values were found.
172 if (uvalues.empty())
173 {
174 string m("initial_metadata_characteristics:: error no ");
175 m += stag;
176 m += k::newline;
177 throw std::runtime_error(m);
178 }
179
180 string ofname(stag + "-values-" + std::to_string(files.size()));
181 std::ofstream ofs(ofname + ".txt");
182 for (const string& s: uvalues)
183 ofs << s << std::endl;
184 }
185
186 // List of all media objects...
187 // List of all empty media object data files in "-empty" file...
188 // List of all media objects with production-geo != usa.
189 constexpr bool serialize_media_object_names_p = true;
190 if (serialize_media_object_names_p)
191 {
192 sstrings sgeoothervalues;
193 sstrings sgeocanadavalues;
194 sstrings sgeoukvalues;
195 sstrings snotusavalues;
196 sstrings smovalues;
197 sstrings snovalues;
198 for (const string& f : files)
199 {
200 rj::Document dom = deserialize_json_to_dom_object(f);
201 auto agevals = values_from_fields_matching(dom, k::id_dimensions.front());
202 string cname = canonical_media_name(dom);
203
204 if (!cname.empty())
205 {
206 std::cout << "canonical media name: " << cname << std::endl;
207 smovalues.insert(cname);
208
209 strings prodvs = values_from_field(dom, "production-geo");
210 if (!prodvs.empty())
211 {
212 auto usapos = find(prodvs.begin(), prodvs.end(), "usa");
213 if (usapos == prodvs.end())
214 {
215 // Add to not usa.
216 snotusavalues.insert(cname);
217
218 auto ukpos = find(prodvs.begin(), prodvs.end(), "uk");
219 if (ukpos != prodvs.end())
220 sgeoukvalues.insert(cname);
221
222 auto capos = find(prodvs.begin(), prodvs.end(), "canada");
223 if (capos != prodvs.end())
224 sgeocanadavalues.insert(cname);
225
226 if (ukpos == prodvs.end() && capos == prodvs.end())
227 sgeoothervalues.insert(cname);
228 }
229 }
230 }
231 else
232 {
233 string m("canonical media name not found in provided dom, ");
234 if (bool(dom.HasParseError()))
235 m += "parse error.";
236 else
237 m += "most probably indicating an empty data file.";
238 m += k::newline;
239
240 m += "file: ";
241 m += f;
242 m += k::newline;
243
244 std::cout << m;
245 snovalues.insert(f);
246 }
247
248 // If there are no age values for any character, then it's empty.
249 if (agevals.empty())
250 snovalues.insert(f);
251 }
252
253 string ofname("media-objects-" + std::to_string(files.size()));
254
255 std::ofstream ofspro(ofname + "-geo-non-usa-production.txt");
256 for (const string& s: snotusavalues)
257 ofspro << s << std::endl;
258
259 std::ofstream ofspro2(ofname + "-geo-uk-production.txt");
260 for (const string& s: sgeoukvalues)
261 ofspro2 << s << std::endl;
262
263 std::ofstream ofspro3(ofname + "-geo-canada-production.txt");
264 for (const string& s: sgeocanadavalues)
265 ofspro3 << s << std::endl;
266
267 std::ofstream ofspro4(ofname + "-geo-non-anglophone-production.txt");
268 for (const string& s: sgeoothervalues)
269 ofspro4 << s << std::endl;
270
271 std::ofstream ofs(ofname + ".txt");
272 for (const string& s: smovalues)
273 ofs << s << std::endl;
274
275 std::ofstream ofsno(ofname + "-empty.txt");
276 for (const string& s: snovalues)
277 ofsno << s << std::endl;
278 }

307/// Initialize special states, per matching id.
308/// Return an array of special/customized ids.
309void
310init_id_render_state_cache()
311{
312 using select = svg::k::select;
313
314 // Set radial range for this viz.
315 point_2t& rrange = get_radial_range();
316 rrange = { 0, 359 };
317
318 // Default style is 2-pixel black color stroke, 50% opacity 1 pixel stroke.
319 const style dstyl = { color::wcag_lgray, 1, color::wcag_lgray, 0, 0.66 };
320
321 // Default layer selections are vector, glyph and text.
322 const select dviz = select::glyph | select::vector | select::text;
323
324 // Set empty identifiers (aka "") to default layer selection and style.
325 add_to_id_render_state_cache("", dstyl, dviz);
326
327 // Specialize for ethnicities
328 style estyl(dstyl);
329 estyl._M_fill_opacity = 1.0;
330 estyl._M_stroke_opacity = 0.0;
331
332 // Specialize for black.
333 style blackstyl(estyl);
334 blackstyl.set_colors(color::duboisred1);
335 add_to_id_render_state_cache("black", blackstyl, dviz);
336
337 // Specialize for white
338 style whitestyl(estyl);
339 whitestyl.set_colors(color::duboisbrown1);
340 add_to_id_render_state_cache("white", whitestyl, dviz);
341
342 // Specialize for asian
343 style asianstyl(estyl);
344 asianstyl.set_colors(color::duboisblue1);
345 add_to_id_render_state_cache("asian pacific", asianstyl, dviz);
346
347 // Specialize for latinx
348 style latinxstyl(estyl);
349 latinxstyl.set_colors(color::duboisyellow2);
350 add_to_id_render_state_cache("latinx", latinxstyl, dviz);
351
352 // Specialize for middle/near eastern
353 style mnestyl(estyl);
354 mnestyl.set_colors(color::duboisblue2);
355 add_to_id_render_state_cache("middle/near eastern", mnestyl, dviz);
356
357 // Specialize for indigenous
358 style istyl(estyl);
359 istyl.set_colors(color::duboisgreen4);
360 add_to_id_render_state_cache("indigenous", istyl, dviz);
361
362 // Specialize for multi/mixed
363 style multistyl(estyl);
364 multistyl.set_colors(color::duboisred2);
365 add_to_id_render_state_cache("multi", multistyl, dviz);
366
367 // Specialize for ambiguous
368 style ambstyl(estyl);
369 ambstyl.set_colors(color::duboisgreen2);
370 add_to_id_render_state_cache("ambiguous", ambstyl, dviz);
371
372 // Specialize for unknown
373 style unknstyl(estyl);
374 unknstyl.set_colors(color::duboisbrown5);
375 add_to_id_render_state_cache("unknown", unknstyl, dviz);
376
377
378 // Specialize for genders.
379 // For male/female/transgender, use the unicode code points in the font.
380 // Scale this via the circle part of each glyph, not the whole glyph.
381 // These svg files are
382 // - precomputed for rotation, aligned to 0 north.
383 // - 100 pixels x 100 pixels square
384 // - circle part of the glyph is centered on page with radius = 35.74 px
385 id_render_state_umap& cache = get_id_render_state_cache();
386 auto& rtr = io::get_run_time_resources();
387 const select gviz = select::glyph | select::svg;
388
389 // female
390 string ffile = "identity/a60-identity-female-glyph.svg";
391 const string femalefile(io::end_path(rtr.data) + ffile);
392 id_render_state female_state(get_metadata_style(), femalefile, 0, 35.74);
393 female_state.set(female_state.visible_mode, gviz);
394 cache.insert(std::make_pair(k::cfemale_value, female_state));
395
396 // alien female is same but ambiguous color
397 female_state.styl.set_colors(color::duboisgreen2);
398 cache.insert(std::make_pair(k::afemale_value, female_state));
399
400 // male
401 string mfile = "identity/a60-identity-male-glyph.svg";
402 const string malefile(io::end_path(rtr.data) + mfile);
403 id_render_state male_state(get_metadata_style(), malefile, 0, 35.74);
404 male_state.set(male_state.visible_mode, gviz);
405 cache.insert(std::make_pair(k::cmale_value, male_state));
406
407 // alien male is same but ambiguous color
408 male_state.styl.set_colors(color::duboisgreen2);
409 cache.insert(std::make_pair(k::amale_value, male_state));
410
411 // trans
412 string tfile = "identity/a60-identity-transgender-glyph.svg";
413 const string transfile(io::end_path(rtr.data) + tfile);
414 id_render_state trans_state(get_metadata_style(), transfile, 0, 35.74);
415 trans_state.set(trans_state.visible_mode, gviz);
416 cache.insert(std::make_pair(k::tfemale_value, trans_state));
417 cache.insert(std::make_pair(k::tmale_value, trans_state));
418
419 // Add playing.
420 add_to_id_render_state_cache(k::playing, get_playing_style(), dviz);
421 add_to_id_render_state_cache(k::playingabv, get_playing_style(), dviz);
422}
