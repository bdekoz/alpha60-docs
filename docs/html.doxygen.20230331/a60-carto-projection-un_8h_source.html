60 /// Assumes projection with no distortion, which is most certainly
61 /// not correct.
62 a60::point_2t
63 meridians_to_point_2d(double lt, double lng) const
64 {
65 bool ltp = lt >= 0.0;
66 bool lngp = lng >= 0.0;
67
68 // Position point on map.
69 double x0 = this->longitude_zero_x;
70 double xdelta = std::abs((width / 360) * lng);
71 double y0 = this->latitude_zero_y;
72 double ydelta = std::abs((height / 180) * lt);
73
74 if (ltp)
75 y0 -= ydelta;
76 else
77 y0 += ydelta;
78
79 if (lngp)
80 x0 += xdelta;
81 else
82 x0 -= xdelta;
83
84 x0 += k_x;
85 y0 += k_y;
86
87 return std::make_tuple(x0, y0);
88 }
89 };
