56/// Get baseline idea about how to color, and what is visible.
57///
58/// Color strategy defaults:
59/// color band if collection size is more than color_array size.
60///
61/// If @pickspecificp is true, start at an implementation-defined color.
62///
63/// NB: cll vs. unique cll, pick only one. And decide before this call...
64void
65initialize_color_and_visibility(collection& cll, render_state& rs,
66 const bool pickspecificp = true,
67 const svg::colorband cb = svg::cband_y)
68{
69 if (!pickspecificp)
70 {
71 // Option 1.
72 // Native color palette is an array of n (currrently 122) RGB colors.
73 // If the torrent collection BTIHA has more elements than that,
74 // generate (infinitely) more colors with the colorband option.
75 if (cll.members.size() <= std::tuple_size<svg::color_array>::value)
76 initialize_colors(cll, rs, true);
77 else
78 initialize_colors(cll, rs, false);
79 }
80 else
81 {
82 // Option 2.
83 // Start at specific color band and walk down colors.
84 initialize_colors(cll, rs, cb);
85 }
86
87 // Set defaults for rendering: weighed, and default visible layers.
88 rs.weigh = true;
89 rs.flip(rs.visible_mode, svg::k::select::vector);
90 rs.flip(rs.visible_mode, svg::k::select::cloud);
91 rs.flip(rs.visible_mode, svg::k::select::echo);
92 rs.flip(rs.visible_mode, svg::k::select::cartography);
93 rs.flip(rs.visible_mode, svg::k::select::telecom);
94 rs.flip(rs.visible_mode, svg::k::select::exitnode);
95 rs.flip(rs.visible_mode, svg::k::select::text);
96}

99/// Render state defaults: render layer visibility, opacity, size.
100void
101initialize_scale_and_opacity(cachecll& wcllu, render_state& rs,
102 const double oo = 0.05)
103{
104 // Get baseline idea of how to size.
105 const ulong sz = wcllu.summarize_totals();
106
107#if 1
108 if (sz > 0 && sz < 100000)
109 rs.scale_mode = svg::k::scale::xxsmall;
110 if (sz >= 100000 && sz < 500000)
111 rs.scale_mode = svg::k::scale::xsmall;
112 if (sz >= 500000 && sz < 1000000)
113 rs.scale_mode = svg::k::scale::small;
114 if (sz >= 1000000 && sz < 4000000)
115 rs.scale_mode = svg::k::scale::medium;
116 if (sz >= 4000000 && sz < 10000000)
117 rs.scale_mode = svg::k::scale::large;
118 if (sz >= 10000000 && sz < 20000000)
119 rs.scale_mode = svg::k::scale::xlarge;
120 if (sz >= 20000000)
121 rs.scale_mode = svg::k::scale::xxlarge;
122#else
123 // 13 steps from [0, 10 * baseline ]
124 // 0 to 5M in 5 steps.
125 // 5M to 10 M in 5 steps.
126 // 0 .83M 1.6666M 2.5M 3.333M 4.16M 5M
127 constexpr double baseline1(1000000.00); // single unit 1-10
128 constexpr double baseline5(baseline1 * 5);
129 constexpr double delta5(baseline1 * 5.0/6.0) ;
130
131 using scale = svg::k::scale;
132 if (sz < baseline5)
133 {
134 // .83
135 if (sz < delta5 * 1)
136 rs.scale_mode = scale::r5s;
137 // 1.6
138 if (sz >= delta5 * 1 && sz < delta5 * 2)
139 rs.scale_mode = scale::r4s;
140 // 2.5
141 if (sz >= delta5 * 2 && sz < delta5 * 3)
142 rs.scale_mode = scale::r3s;
143 // 3.33
144 if (sz >= delta5 * 3 && sz < delta5 * 4)
145 rs.scale_mode = scale::r2s;
146 // 4.16
147 if (sz >= delta5 * 4 && sz < delta5 * 5)
148 rs.scale_mode = scale::r1s;
149 // 5
150 if (sz >= delta5 * 5 && sz < baseline5)
151 rs.scale_mode = scale::baseline;
152 }
153 else
154 {
155 if (sz >= baseline5 && sz < baseline5 + (delta5 * 1))
156 rs.scale_mode = scale::e1s;
157 if (sz >= baseline5 + (delta5 * 1) && sz < baseline5 + (delta5 * 2))
158 rs.scale_mode = scale::e2s;
159 if (sz >= baseline5 + (delta5 * 2) && sz < baseline5 + (delta5 * 3))
160 rs.scale_mode = scale::e3s;
161 if (sz >= baseline5 + (delta5 * 3) && sz < baseline5 + (delta5 * 4))
162 rs.scale_mode = scale::e4s;
163 if (sz >= baseline5 + (delta5 * 4) && sz < baseline5 + (delta5 * 5))
164 rs.scale_mode = scale::e5s;
165 }
166#endif
167
168 const double opacity = oo;
169 // opacity = 0.33; // echo v 1.3
170 // opacity = 0.5; // echo v 1.4, good for echo on 3d guns (1.5M)
171 // opacity = 0.20;
172 // opacity = 0.10; // v90x
173 // opacity = 0.05; // v913-25x, v1.1, LKG
174 // opacity = 0.04; // v913-25x, v1.1, maybe a bit to light at 10M
175 // opacity = 0.025; // v1.2, too light for mandalorian (6M)
176 rs.opacity = opacity;
177
178 // Output for log.
179 std::cout << "initialize_scale_and_opacity:: assessed defaults" << std::endl
180 << "swarm size is " << sz << " peers total" << std::endl
181 << "scale is " << to_string(rs.scale_mode) << std::endl
182 << "opacity is " << rs.opacity << std::endl
183 << std::endl << std::endl;
184}

187/// Setup environment for rendering.
188cachecll
189setup_render(collection& cll, const swarm_mode swarm,
190 const bool useuniquebtihap = true,
191 const bool pickspecificp = false,
192 const svg::colorband cb = svg::cband_y)
193{
194 // Initialize torrent collection.
195 cll.initialize_ids_and_members(cll.torrent_files());
196
197 cachecll wcll(cll);
198 if (useuniquebtihap)
199 {
200 // Find unique btiha, and use that to setup colors.
201 wcll = coalesce_with_unique_btiha(cll, a60::k::cumulative_s);
202 weigh_sort_unique_collection(wcll, swarm);
203 }
204 else
205 {
206 // NB: Durations day and week need this for per-day output.
207 // Find non-unique btiha, and use that to setup colors.
208 const io::run_time_resources& rtr = io::get_run_time_resources();
209 const string cdir = rtr.cache;
210 wcll = get_swarm_gcache_matching(cll, cdir, k::cumulative_s);
211 weigh_sort_unique_collection(wcll, swarm);
212 }
213
214 // Get render state, set default characteristics desired.
215 render_state& rs = get_render_state();
216 initialize_color_and_visibility(wcll, rs, pickspecificp, cb);
217 initialize_scale_and_opacity(wcll, rs);
218
219 return wcll;
220}

233/// Augment peers/seeds for all d(days, weeks) that have been coalesced.
234void
235augment_weighed_by_torrent(cachecll& wcll,
236 const carto::cartography<auto>& cartog,
237 string odir)
238{
239 // Augmentation log
240 ios_base::openmode om = ios_base::out | ios_base::app;
241 string logstem(wcll.match + "-augment-weighed-by-torrent");
242 std::ofstream ofsaug(odir + logstem + ".log", om);
243
244 bool peerp = wcll.sort_mode == swarm_mode::peer;
245
246 // Common typographics.
247 svg::typography typo = apercu_typo;
248 typo._M_style = peerp ? b_style : w_style;
249 typo._M_w = svg::typography::weight::medium;
250
251 svg::typography typov = apercu_typo;
252 typov._M_anchor = svg::typography::anchor::start;
253 typov._M_w = svg::typography::weight::xbold;
254
255 svg::typography typotxt = apercu_typo;
256 typotxt._M_anchor = svg::typography::anchor::start;
257 typotxt._M_align = svg::typography::align::left;
258 typotxt._M_w = svg::typography::weight::medium;
259 typotxt._M_p = svg::typography::property::normal;
260
261 render_state tr = get_render_state();
262 tr.set(tr.visible_mode, svg::k::select::cartography | svg::k::select::vector);
263 tr.outline_mode = svg::k::select::cartography;
264 tr.weigh = true;
265 tr.opacity = 0.10;
266
267 group_element gpost;
268 gpost.finish_element();
269
270 svg::color cbytor = svg::color::asamablue;
271 try
272 {
273 const collection::tmarkups& tm = wcll.members;
274 ofsaug << "augment_weighed_by_torrent: " << tm.size() << endl;
275 for (uint count = 0; count < tm.size(); ++count)
276 {
277 const collection::tmarkup& mrkup = tm[count];
278 auto key = mrkup.name;
279 ulong tid = wcll.get_tid(key);
280
281 std::ostringstream osst;
282 osst << "t" << std::setfill('0') << std::setw(3) << tid;
283 osst << "-cumulative";
284 string matchtid = osst.str();
285
286 // Sanity check the torrent has any information.
287 auto peertotalsn = wcll.peer_totals[count];
288 ofsaug << matchtid << " " << peertotalsn << endl;
289 if (peertotalsn == 0)
290 continue;
291
292 // Make SVG
293 svg_element obj = augment_base(wcll, cartog, odir,
294 wcll.sort_mode, matchtid, tr);
295 const auto [ width, height] = obj._M_area;
296
297 // Style details.
298 svg::style styl = peerp ? b_style : w_style;
299 styl._M_fill_opacity = 0.33;
300 styl._M_fill_color = cbytor;
301
302
303 // Vector Graphics group start
304 group_element gpreg;
305 gpreg.start_element("graphic");
306 obj.add_element(gpreg);
307
308
309 // Set radius to 1, 2, 3, 4.
310 augment_wgeo_peers(obj, wcll.peer_geos[count], wcll.peer_wkeys[count],
311 cartog, styl, 3);
312
313 obj.add_element(gpost);
314
315 // Put down solid color on bottom edge.
316 augment_collection_color_bar(obj, wcll.sort_mode, cbytor, 100);
317
318 // Put summary numbers down.
319 int fmargin = carto::frame_margin();
320 glyph sumg;
321 sumg.d1 = wcll.peer_totals[count];
322 sumg.d2 = wcll.seed_totals[count];
323 augment_sum_glyph(obj, typo, sumg, wcll.sort_mode, width - fmargin,
324 height - (fmargin / 2));
325
326 // Media title.
327 svg::typography typog = typotxt;
328 typog._M_style._M_fill_color = svg::color::gray25;
329 sized_text_r(obj, typog, 24, wcll.media_object_name(),
330 fmargin, fmargin / 2, 90);
331
332 // Torrent id.
333 sized_text(obj, typov, 24, matchtid, fmargin, height - (fmargin / 2));
334
335 // Torrent file name title.
336 sized_text(obj, typo, 20, key, width / 2, height - (fmargin / 2));
337 }
338 }
339 catch (const std::runtime_error& e)
340 {
341 // No files found for given match, ie reached the end of the sequence.
342 std::cout << "finished: " << e.what() << endl;
343 }
344}

347/// Augment peers/seeds for each partition in the torrent collection.
348/// NB: Assumes 1x cartography.
349void
350augment_weighed_by_partition(cachecll& wcll,
351 const carto::cartography<auto>& cartog,
352 string odir, render_state& tr)
353{
354 // Augmentation log
355 ios_base::openmode om = ios_base::out | ios_base::app;
356 string logstem(wcll.match + "-augment-weighed-weighed-by-partition");
357 std::ofstream ofsaug(odir + logstem + ".log", om);
358
359 resolution_mode rmodesd(resolution_mode::rsd);
360 encoding_mode emodea(encoding_mode::all);
361
362 // Draw maps of each partition only if cartography flag.
363 const bool cartop = tr.is_visible(svg::k::select::cartography);
364 uint slicen(1);
365 const string slicestr("cumulative-slice-");
366
367 // 0: hd, 4k
368 cachecll c4k = wcll.partition_cache_by_mode(resolution_mode::r4k, emodea);
369 string name0 = slicestr + to_string(slicen++) + "-" + c4k.partition_name();
370 c4k.summarize_swarm(std::numeric_limits<ulong>::max());
371 if (cartop && !c4k.sum_geos.empty())
372 augment_collection(c4k, cartog, odir, name0, tr);
373
374 // 1: hd, 1080p
375 cachecll c1080 = wcll.partition_cache_by_mode(resolution_mode::r1080p, emodea);
376 string name1 = slicestr + to_string(slicen++) + "-" + c1080.partition_name();
377 c1080.summarize_swarm(std::numeric_limits<ulong>::max());
378 if (cartop && !c1080.sum_geos.empty())
379 augment_collection(c1080, cartog, odir, name1, tr);
380
381 // 2: hd, 720p
382 cachecll c720 = wcll.partition_cache_by_mode(resolution_mode::r720p, emodea);
383 string name2 = slicestr + to_string(slicen++) + "-" + c720.partition_name();
384 c720.summarize_swarm(std::numeric_limits<ulong>::max());
385 if (cartop && !c720.sum_geos.empty())
386 augment_collection(c720, cartog, odir, name2, tr);
387
388 // 3: sd
389 cachecll cx = wcll.partition_cache_by_mode(rmodesd, emodea);
390 string name3 = slicestr + to_string(slicen++) + "-" + cx.partition_name();
391 cx.summarize_swarm(std::numeric_limits<ulong>::max());
392 if (cartop && !cx.sum_geos.empty())
393 augment_collection(cx, cartog, odir, name3, tr);
394
395 // Make summary ovoid with parition info as per circle area.
396 // Make glyps for variations.
397
398 glyphs glys;
399 glys.push_back(sum_glyphs(make_glyphs(c4k), c4k.partition_name()));
400 glys.push_back(sum_glyphs(make_glyphs(c1080), c1080.partition_name()));
401 glys.push_back(sum_glyphs(make_glyphs(c720), c720.partition_name()));
402 glys.push_back(sum_glyphs(make_glyphs(cx), cx.partition_name()));
403
404 // Sort glyphs in decreasing size, so that biggest layer is lowest.
405 auto lglyphweigh = [](const glyph& g1, const glyph& g2)
406 { return g1.weight > g2.weight; };
407 std::sort(glys.begin(), glys.end(), lglyphweigh);
408
409 // Draw maps of each partition only if text flag.
410 const bool textp = tr.is_visible(svg::k::select::text);
411 if (textp)
412 {
413 svg::typography typo = svg::k::apercu_typo;
414 typo._M_align = svg::typography::align::left;
415 typo._M_anchor = svg::typography::anchor::start;
416
417 render_state txtotr = tr;
418 tr.clear(tr.visible_mode, svg::k::select::cartography);
419 tr.clear(tr.visible_mode, svg::k::select::legend);
420 txtotr.alt = true;
421 string tt("cumulative-peer-4-ovoid-partitions-text");
422
423 svg_element obj = augment_collection_ovoid(wcll, cartog,
424 glys, odir, tt, txtotr);
425
426 // Add in total bytes.
427 const int fmargin = carto::frame_margin();
428 auto y = obj._M_area._M_height / 2;
429
430 for (glyph& g: glys)
431 {
432 if (g.bytes > 0)
433 {
434 std::ostringstream oss;
435 oss.imbue(std::locale(""));
436 oss << std::fixed << std::setprecision(2);
437 oss << "size in TB: " << double(g.bytes) / k::b_in_gb / k::gb_in_tb;
438 string psize(g.name + k::space + oss.str());
439
440 sized_text(obj, typo, 18, psize, fmargin, y);
441 y += 22;
442 }
443 }
444
445 // Total in collection.
446 glyph sumg = sum_glyphs(make_glyphs(wcll));
447 double tbtotal = double((sumg.bytes / k::b_in_gb) / k::gb_in_tb);
448
449 std::ostringstream oss;
450 oss.imbue(std::locale(""));
451 oss << "total size in TB: "
452 << std::fixed << std::setprecision(2) << tbtotal;
453
454 typo._M_w = svg::typography::weight::medium;
455 sized_text(obj, typo, 22, oss.str(), fmargin,
456 obj._M_area._M_height - fmargin);
457 }
458}

1169 to previous entity. For hours, that is the previous hour. For

1173 duration.

1175 The removestr parameter filters out files matching argument. This
1176 is needed to remove the hour/sample mangling issue, where
1177 per-sample files have both an hour and sample id. (As per attempt
1178 to file sort same-mode in directory.) For hour, use removestr =
1179 "-sample-".
1180*/

1183augment_weighed_duration_since(const collection& cll,
1184 const carto::cartography<_Proj>& cartog,
1185 string odir, const swarm_mode m,
1186 string d, const bool eq_previous_p,
1187 svg::color klr, const carto_state cstate,
1188 bool comparep = false,
1189 const string removestr = "")
1190{
1191 // Get current cache, cache-persistent directories.
1192 const io::run_time_resources& rtr = io::get_run_time_resources();

1201

1204 if (!removestr.empty())
1205 ocll.remove_matching_strings(files, removestr);
1206 if (files.empty())
1207 {
1208 string s("no cache files found in " + cdir + " for match " + d);
1209 std::cout << s << endl;
1210 return;
1211 }
1212
1213 // Split into persistent, swarm, and regular cache files.
1214 strings pfiles = ocll.match_files(cpdir, d);
1215 strings sfiles = cll.remove_matching_strings(pfiles, k::swarm_s);
1216
1217 ofsaug << "augment_weighed_duration_since: starting" << endl;
1218 ofsaug << "finding: " << d << endl;

1220 << "cache persistent files: " << pfiles.size() << k::space
1221 << "cache swarm files: " << sfiles.size()
1222 << "dcompare function: " << eq_previous_p
1223 << endl;
1224
1225 // Durations like (day-0002, week-0041, etc.)
1226 strings durations = find_unique_durations(sfiles, d);
1227 if (durations.empty())
1228 throw std::runtime_error("weighed_duration_since:: no durations for " + d);
1229
1230 // Log results.
1231 ofsaug << durations.size() << " durations found:" << endl;
1232 for (const string& s: durations)
1233 ofsaug << s << endl;
1234 ofsaug << endl;

1245 for (const string& sfile : sfiles)
