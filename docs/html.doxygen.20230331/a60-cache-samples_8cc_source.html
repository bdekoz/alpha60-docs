87
88 // Set up cached/output environment.

91 string cdir;
92 if (argc >= 2)
93 cdir = argv[1];
94 else
95 cdir = a60::k::cache_s;
96 io::end_path(cdir);
97 io::sanity_check_output_directory(cdir);
98
99 string cdirp;
100 if (argc >= 3)
101 cdirp = argv[2];
102 else
103 cdirp = a60::k::cache_persistent_s;
104 io::end_path(cdirp);
105 io::sanity_check_output_directory(cdirp);
106
107 io::set_run_time_resources(cdir, cdirp);
108
109 // Report.
110 std::cout << "output: " << odirbase << std::endl;
111 std::cout << "cache data: " << cdir << std::endl;
112 std::cout << "cache persistent data: " << cdirp << std::endl;
113
114 coalesce_single(cll, cdir, cdirp);
115
116 return 0;
